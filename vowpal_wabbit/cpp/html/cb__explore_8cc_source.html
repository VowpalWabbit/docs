158  size_t counter = data.counter;

174  base.learn(ec);
175 
176  // Now update oracles
177 
178  // 1. Compute loss vector
179  data.cs_label.costs.clear();
180  float norm = min_prob * num_actions;
181  ec.l.cb = data.cb_label;
182  data.cbcs.known_cost = get_observed_cost(data.cb_label);
183  gen_cs_example<false>(data.cbcs, ec, data.cb_label, data.cs_label);
184  for (uint32_t i = 0; i < num_actions; i++) probabilities[i] = 0;

187  // 2. Update functions
188  for (size_t i = 0; i < cover_size; i++)

192  {
193  float pseudo_cost =
194  data.cs_label.costs[j].x - data.psi * min_prob / (max(probabilities[j], min_prob) / norm) + 1;
195  data.second_cs_label.costs[j].class_index = j + 1;
196  data.second_cs_label.costs[j].x = pseudo_cost;
197  }
198  if (i != 0)
199  data.cs->learn(ec, i + 1);
200  if (probabilities[predictions[i] - 1] < min_prob)
201  norm += max(0, additive_probability - (min_prob - probabilities[predictions[i] - 1]));
202  else
203  norm += additive_probability;
204  probabilities[predictions[i] - 1] += additive_probability;
205  }
206  }
207
