128 inline uint64_t hash_space(vw& all, std::string s)
129 {
130  substring ss;
131  ss.begin = (char*)s.c_str();
132  ss.end = ss.begin + s.length();
133  return all.p->hasher(ss, all.hash_seed);
134 }
135 inline uint64_t hash_space_static(std::string s, std::string hash)
136 {
137  substring ss;
138  ss.begin = (char*)s.c_str();
139  ss.end = ss.begin + s.length();
140  return getHasher(hash)(ss, 0);
141 }
142 // Then use it as the seed for hashing features.
143 inline uint64_t hash_feature(vw& all, std::string s, uint64_t u)
144 {
145  substring ss;
146  ss.begin = (char*)s.c_str();
147  ss.end = ss.begin + s.length();
148  return all.p->hasher(ss, u) & all.parse_mask;
149 }
150 inline uint64_t hash_feature_static(std::string s, uint64_t u, std::string h, uint32_t num_bits)
151 {
152  substring ss;
153  ss.begin = (char*)s.c_str();
154  ss.end = ss.begin + s.length();
155  size_t parse_mark = (1 << num_bits) - 1;
156  return getHasher(h)(ss, u) & parse_mark;
157 }
