30 void print_result(int f, priority_queue<scored_example, vector<scored_example>, compare_scored_examples>& pr_queue)
31 {
32  if (f >= 0)
33  {
34  char temp[30];
35  std::stringstream ss;

38  {

71 template <bool is_learn>
72 void predict_or_learn(topk& d, LEARNER::single_learner& base, multi_ex& ec_seq)
73 {
74  for (auto example : ec_seq)
75  {
76  auto ec = *example;
77 
78  if (is_learn)

82 
83  if (d.pr_queue.size() < d.K)
84  d.pr_queue.push(make_pair(ec.pred.scalar, ec.tag));
85  else if (d.pr_queue.top().first < ec.pred.scalar)
86  {
87  d.pr_queue.pop();
88  d.pr_queue.push(make_pair(ec.pred.scalar, ec.tag));
89  }
90
