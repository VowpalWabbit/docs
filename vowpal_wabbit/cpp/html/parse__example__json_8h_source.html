612  BaseState<audit>* Ignore(Context<audit>& ctx, rapidjson::SizeType length)
613  {
614  // fast ignore
615  // skip key + \0 + "
616  char* head = ctx.stream->src_ + length + 2;
617  if (head >= ctx.stream_end || *head != ':')
618  {
619  ctx.error() << "Expected ':' found '" << *head << "'";
620  return nullptr;
621  }
622  head++;
623 
624  // scan for ,}
625  // support { { ... } }
626  int depth = 0, sq_depth = 0;
627  bool stop = false;
628  while (!stop)
629  {
630  switch (*head)
631  {
632  case '\0':
633  ctx.error() << "Found EOF";
634  return nullptr;
635  case '"':
636  {
637  // skip strings
638  bool stopInner = false;
639  while (!stopInner)
640  {
641  head++;
642  switch (*head)
643  {
644  case '\0':
645  ctx.error() << "Found EOF";
646  return nullptr;
647  case '\\':
648  head++;
649  break;
650  case '"':
651  stopInner = true;
652  break;
653  }
654  }
655  break;
656  }
657  case '{':
658  depth++;
659  break;
660  case '}':
661  if (depth == 0 && sq_depth == 0)
662  stop = true;
663  else
664  depth--;
665  break;
666  case '[':
667  sq_depth++;
668  break;
669  case ']':
670  if (depth == 0 && sq_depth == 0)
671  stop = true;
672  else
673  sq_depth--;
674  break;
675  case ',':
676  if (depth == 0 && sq_depth == 0)
677  stop = true;
678  break;
679  }
680  head++;
681  }
682 
683  // skip key + \0 + ":
684  char* value = ctx.stream->src_ + length + 3;
685  if (value >= ctx.stream_end)
686  {
687  ctx.error() << "Found EOF";
688  return nullptr;
689  }
690 
691  *value = '0';
692  value++;
693  memset(value, ' ', head - value - 1);
694 
695  return &ctx.ignore_state;
696  }

699  {
700  ctx.key = str;
701  ctx.key_length = length;
702 
703  if (length > 0 && str[0] == '_')
704  {

714  else

718  }
719  }
