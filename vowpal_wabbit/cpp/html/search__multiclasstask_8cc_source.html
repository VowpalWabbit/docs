42 void run(Search::search& sch, multi_ex& ec)
43 {
44  task_data* my_task_data = sch.get_task_data<task_data>();
45  size_t gold_label = ec[0]->l.multi.label;
46  size_t label = 0;
47  size_t learner_id = 0;
48 
49  for (size_t i = 0; i < my_task_data->num_level; i++)
50  {
51  size_t mask = UINT64_ONE << (my_task_data->num_level - i - 1);
52  size_t y_allowed_size = (label + mask + 1 <= my_task_data->max_label) ? 2 : 1;
53  action oracle = (((gold_label - 1) & mask) > 0) + 1;
54  size_t prediction = sch.predict(*ec[0], 0, &oracle, 1, nullptr, nullptr, my_task_data->y_allowed.begin(),
55  y_allowed_size, nullptr, learner_id); // TODO: do we really need y_allowed?
56  learner_id = (learner_id << 1) + prediction;
57  if (prediction == 2)
58  label += mask;
59  }
60  label += 1;
61  sch.loss(!(label == gold_label));
62  if (sch.output().good())
63  sch.output() << label << ' ';
64 }
