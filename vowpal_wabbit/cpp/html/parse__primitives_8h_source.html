83 inline float parseFloat(char* p, char** end, char* endLine = nullptr)
84 {
85  char* start = p;
86  bool endLine_is_null = endLine == nullptr;
87 
88  if (!*p)
89  {
90  *end = p;
91  return 0;
92  }
93  int s = 1;
94  while ((*p == ' ') && (endLine_is_null || p < endLine)) p++;
95 
96  if (*p == '-')
97  {
98  s = -1;
99  p++;
100  }
101 
102  float acc = 0;
103  while (*p >= '0' && *p <= '9' && (endLine_is_null || p < endLine)) acc = acc * 10 + *p++ - '0';
104 
105  int num_dec = 0;
106  if (*p == '.')
107  {
108  while (*(++p) >= '0' && *p <= '9' && (endLine_is_null || p < endLine))
109  {
110  if (num_dec < 35)
111  {
112  acc = acc * 10 + (*p - '0');
113  num_dec++;
114  }
115  }
116  }
117 
118  int exp_acc = 0;
119  if ((*p == 'e' || *p == 'E') && (endLine_is_null || p < endLine))
120  {
121  p++;
122  int exp_s = 1;
123  if (*p == '-' && (endLine_is_null || p < endLine))
124  {
125  exp_s = -1;
126  p++;
127  }
128  while (*p >= '0' && *p <= '9' && (endLine_is_null || p < endLine)) exp_acc = exp_acc * 10 + *p++ - '0';
129  exp_acc *= exp_s;
130  }
131  if (*p == ' ' || *p == '\n' || *p == '\t' || p == endLine) // easy case succeeded.
132  {
133  acc *= powf(10, (float)(exp_acc - num_dec));
134  *end = p;
135  return s * acc;
136  }
137  else
138  return (float)strtod(start, end);
139 }
