90 float binary_search(float fhat, float delta, float sens, float tol = 1e-6)
91 {
92  const float maxw = min(fhat / sens, FLT_MAX);
93 
94  if (maxw * fhat * fhat <= delta)
95  return maxw;
96 
97  float l = 0;
98  float u = maxw;
99  float w, v;
100 
101  for (int iter = 0; iter < B_SEARCH_MAX_ITER; iter++)
102  {
103  w = (u + l) / 2.f;
104  v = w * (fhat * fhat - (fhat - sens * w) * (fhat - sens * w)) - delta;
105  if (v > 0)
106  u = w;
107  else
108  l = w;
109  if (fabs(v) <= tol || u - l <= tol)
110  break;
111  }
112 
113  return l;
114 }

325 {

327  // sort top_actions and data.action_probs by the order induced in data.scores.

420  const uint32_t num_actions = (uint32_t)preds.size();

425  probs.clear();

432  size_t tied_actions = fill_tied(data, preds);
433  for (size_t i = 0; i < tied_actions; ++i) probs[preds[i].action].score += additive_probability / tied_actions;
434  }
