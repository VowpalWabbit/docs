147 static int make_hot_sv(svm_params& params, size_t svi)
148 {
149  svm_model* model = params.model;
150  size_t n = model->num_support;
151  if (svi >= model->num_support)
152  params.all->trace_message << "Internal error at " << __FILE__ << ":" << __LINE__ << endl;
153  // rotate params fields
154  svm_example* svi_e = model->support_vec[svi];
155  int alloc = svi_e->compute_kernels(params);
156  float svi_alpha = model->alpha[svi];
157  float svi_delta = model->delta[svi];
158  for (size_t i = svi; i > 0; --i)
159  {
160  model->support_vec[i] = model->support_vec[i - 1];
161  model->alpha[i] = model->alpha[i - 1];
162  model->delta[i] = model->delta[i - 1];
163  }
164  model->support_vec[0] = svi_e;
165  model->alpha[0] = svi_alpha;
166  model->delta[0] = svi_delta;
167  // rotate cache
168  for (size_t j = 0; j < n; j++)
169  {
170  svm_example* e = model->support_vec[j];
171  size_t rowsize = e->krow.size();
172  if (svi < rowsize)
173  {
174  float kv = e->krow[svi];
175  for (size_t i = svi; i > 0; --i) e->krow[i] = e->krow[i - 1];
176  e->krow[0] = kv;
177  }
178  else
179  {
180  float kv = svi_e->krow[j];
181  e->krow.push_back(0);
182  alloc += 1;
183  for (size_t i = e->krow.size() - 1; i > 0; --i) e->krow[i] = e->krow[i - 1];
184  e->krow[0] = kv;
185  }
186  }
187  return alloc;
188 }

332 float linear_kernel(const flat_example* fec1, const flat_example* fec2)
333 {
334  float dotprod = 0;
335 
336  features& fs_1 = (features&)fec1->fs;
337  features& fs_2 = (features&)fec2->fs;
338  if (fs_2.indicies.size() == 0)
339  return 0.f;
340 
341  int numint = 0;
342  for (size_t idx1 = 0, idx2 = 0; idx1 < fs_1.size() && idx2 < fs_2.size(); idx1++)
343  {
344  uint64_t ec1pos = fs_1.indicies[idx1];
345  uint64_t ec2pos = fs_2.indicies[idx2];
346  // params.all->opts_n_args.trace_message<<ec1pos<<" "<<ec2pos<<" "<<idx1<<" "<<idx2<<" "<<f->x<<" "<<ec2f->x<<endl;
347  if (ec1pos < ec2pos)
348  continue;
349 
350  while (ec1pos > ec2pos && ++idx2 < fs_2.size()) ec2pos = fs_2.indicies[idx2];
351 
352  if (ec1pos == ec2pos)
353  {
354  // params.all->opts_n_args.trace_message<<ec1pos<<" "<<ec2pos<<" "<<idx1<<" "<<idx2<<" "<<f->x<<"
355  // "<<ec2f->x<<endl;
356  numint++;
357  dotprod += fs_1.values[idx1] * fs_2.values[idx2];
358  ++idx2;
359  }
360  }
361  // params.all->opts_n_args.trace_message<<endl;
362  // params.all->opts_n_args.trace_message<<"numint = "<<numint<<" dotprod = "<<dotprod<<endl;
363  return dotprod;
364 }

502 bool update(svm_params& params, size_t pos)
503 {
504  // params.all->opts_n_args.trace_message<<"Update\n";
505  svm_model* model = params.model;
506  bool overshoot = false;
507  // params.all->opts_n_args.trace_message<<"Updating model "<<pos<<" "<<model->num_support<<" ";
508  svm_example* fec = model->support_vec[pos];
509  label_data& ld = fec->ex.l.simple;
510  fec->compute_kernels(params);
511  float* inprods = fec->krow.begin();
512  float alphaKi = dense_dot(inprods, model->alpha, model->num_support);
513  model->delta[pos] = alphaKi * ld.label / params.lambda - 1;
514  float alpha_old = model->alpha[pos];
515  alphaKi -= model->alpha[pos] * inprods[pos];
516  model->alpha[pos] = 0.;
517 
518  float proj = alphaKi * ld.label;
519  float ai = (params.lambda - proj) / inprods[pos];
520  // cout<<model->num_support<<" "<<pos<<" "<<proj<<" "<<alphaKi<<" "<<alpha_old<<" "<<ld.label<<"
521  // "<<model->delta[pos]<<" " << ai<<" "<<params.lambda<<endl;
522 
523  if (ai > fec->ex.l.simple.weight)
524  ai = fec->ex.l.simple.weight;
525  else if (ai < 0)
526  ai = 0;
527 
528  ai *= ld.label;
529  float diff = ai - alpha_old;
530 
531  if (fabs(diff) > 1.0e-06)
532  overshoot = true;
533 
534  if (fabs(diff) > 1.)
535  {
536  // params.all->opts_n_args.trace_message<<"Here\n";
537  diff = (float)(diff > 0) - (diff < 0);
538  ai = alpha_old + diff;
539  }
540 
541  for (size_t i = 0; i < model->num_support; i++)
542  {
543  label_data& ldi = model->support_vec[i]->ex.l.simple;
544  model->delta[i] += diff * inprods[i] * ldi.label / params.lambda;
545  }
546 
547  if (fabs(ai) <= 1.0e-10)
548  remove(params, pos);
549  else
550  model->alpha[pos] = ai;
551 
552  return overshoot;
553 }
