59 {

66 
67  while (reading_head < s.end)
68  {
69  char current_character = *reading_head++;
70 
71  if (in_escape)
72  {
73  *writing_head++ = current_character;
74  in_escape = false;
75  }
76  else
77  {
78  if (current_character == delim)
79  {
80  current.end = writing_head++;

90  }
