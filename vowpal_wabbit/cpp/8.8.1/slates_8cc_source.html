25 void slates_data::learn_or_predict(LEARNER::multi_learner& base, multi_ex& examples)
26 {
27  _stashed_labels.clear();
28  _stashed_labels.reserve(examples.size());
29  for (auto& example : examples)
30  {
31  _stashed_labels.push_back(std::move(example->l.slates));
32  }
33 
34  const size_t num_slots = std::count_if(examples.begin(), examples.end(),
35  [](const example* example) { return example->l.conditional_contextual_bandit.type == CCB::example_type::slot; });
36 
37  float global_cost = 0.f;
38  bool global_cost_found = false;
39  uint32_t action_index = 0;
40  size_t slot_index = 0;
41  std::vector<std::vector<uint32_t>> slot_action_pools(num_slots);
42  for (size_t i = 0; i < examples.size(); i++)
43  {
44  CCB::label ccb_label;
45  memset(&ccb_label, 0, sizeof(ccb_label));
46  CCB::ccb_label_parser.default_label(&ccb_label);
47  const auto& slates_label = _stashed_labels[i];
48  if (slates_label.type == slates::example_type::shared)
49  {
50  ccb_label.type = CCB::example_type::shared;
51  if (slates_label.labeled)
52  {
53  global_cost_found = true;
54  global_cost = slates_label.cost;
55  }
56  }
57  else if (slates_label.type == slates::example_type::action)
58  {
59  if (slates_label.slot_id >= num_slots)
60  {
61  THROW("slot_id cannot be larger than or equal to the number of slots");
62  }
63  ccb_label.type = CCB::example_type::action;
64  slot_action_pools[slates_label.slot_id].push_back(action_index);
65  action_index++;
66  }
67  else if (slates_label.type == slates::example_type::slot)
68  {
69  ccb_label.type = CCB::example_type::slot;
70  if (global_cost_found)
71  {
72  ccb_label.outcome = new CCB::conditional_contextual_bandit_outcome();
73  ccb_label.outcome->cost = global_cost;
74  ccb_label.outcome->probabilities = v_init<ACTION_SCORE::action_score>();
75  ccb_label.explicit_included_actions = v_init<uint32_t>();
76  for (const auto index : slot_action_pools[slot_index])
77  {
78  ccb_label.explicit_included_actions.push_back(index);
79  }
80  for (const auto& action_score : slates_label.probabilities)
81  {
82  // We need to convert from slate space which is zero based for
83  // each slot to CCB where all action indices are in the same space.
84  auto ccb_space_index = slot_action_pools[slot_index][action_score.action];
85  ccb_label.outcome->probabilities.push_back({ccb_space_index, action_score.score});
86  }
87  }
88  slot_index++;
89  }
90 
91  ccb_label.weight = slates_label.weight;
92  examples[i]->l.conditional_contextual_bandit = ccb_label;
93  }
94  LEARNER::multiline_learn_or_predict<is_learn>(base, examples, examples[0]->ft_offset);
95 
96  // Need to convert decision scores to the original index space. This can be
97  // done by going through the prediction for each slots and subtracting the
98  // number of actions seen so far.
99  uint32_t size_so_far = 0;
100  for (auto& action_scores : examples[0]->pred.decision_scores)
101  {
102  for (size_t i = 0; i < action_scores.size(); i++)
103  {
104  action_scores[i].action = action_scores[i].action - size_so_far;
105  }
106  size_so_far += static_cast<uint32_t>(action_scores.size());
107  }
108 
109  for (size_t i = 0; i < examples.size(); i++)
110  {
111  CCB::ccb_label_parser.delete_label(&examples[i]->l.conditional_contextual_bandit);
112  examples[i]->l.slates = std::move(_stashed_labels[i]);
113  }
114  _stashed_labels.clear();
115 }

130  size_t counter = 0;

136  const auto& label = slot->l.slates;

140  }

143  label_ss << delim << "?";

151  label_ss << delim << "...";

153  }
154  }

159 void output_example(vw& all, slates_data& /*c*/, multi_ex& ec_seq)
160 {
161  std::vector<example*> slots;
162  size_t num_features = 0;
163  float loss = 0.;
164  bool is_labelled = ec_seq[SHARED_EX_INDEX]->l.slates.labeled;
165  float cost = is_labelled ? ec_seq[SHARED_EX_INDEX]->l.slates.cost : 0.f;
166 
167  for (auto ec : ec_seq)
168  {
169  num_features += ec->num_features;
170 
171  if (ec->l.slates.type == slates::example_type::slot)
172  {
173  slots.push_back(ec);
174  }
175  }
176 
177  auto predictions = ec_seq[0]->pred.decision_scores;
178  for (size_t slot_index = 0; slot_index < slots.size(); slot_index++)
179  {
180  const auto& label_probabilties = slots[slot_index]->l.slates.probabilities;
181  if (is_labelled)
182  {
183  if (label_probabilties.empty())
184  {
185  THROW("Probabilities missing for labeled example");
186  }
187 
188  float l = CB_ALGS::get_cost_estimate(
189  label_probabilties[TOP_ACTION_INDEX], cost, predictions[slot_index][TOP_ACTION_INDEX].action);
190  loss += l * predictions[slot_index][TOP_ACTION_INDEX].score;
191  }
192  }
193 
194  bool holdout_example = is_labelled;
195  if (holdout_example != false)
196  {
197  for (const auto& example : ec_seq)
198  {
199  holdout_example &= example->test_only;
200  }
201  }
202 
203  all.sd->update(holdout_example, is_labelled, loss, ec_seq[SHARED_EX_INDEX]->weight, num_features);
204 
205  for (auto sink : all.final_prediction_sink)
206  {
207  VW::print_decision_scores(sink, ec_seq[SHARED_EX_INDEX]->pred.decision_scores);
208  }
209 
210  VW::print_update_slates(all, slots, predictions, num_features);
211 }
