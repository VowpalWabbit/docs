31 inline int compare_on_cost(const void *void_a, const void *void_b)
32 {
33  if (void_a == void_b)
34  return 0;
35  const beam_element<void> *a = (const beam_element<void> *)void_a;
36  const beam_element<void> *b = (const beam_element<void> *)void_b;
37  if (a->active && !b->active)
38  return -1; // active things come before inactive things
39  else if (!a->active && b->active)
40  return 1;
41  else if (!a->active && !b->active)
42  return 0;
43  else if (a->cost < b->cost)
44  return -1; // otherwise sort by cost
45  else if (a->cost > b->cost)
46  return 1;
47  else
48  return 0;
49 }
