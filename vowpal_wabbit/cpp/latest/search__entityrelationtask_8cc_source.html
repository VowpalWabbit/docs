141 size_t predict_entity(
142  Search::search& sch, example* ex, v_array<size_t>& /*predictions*/, ptag my_tag, bool isLdf = false)
143 {
144  task_data* my_task_data = sch.get_task_data<task_data>();
145  size_t prediction;
146  if (my_task_data->allow_skip)
147  {
148  v_array<uint32_t> star_labels = v_init<uint32_t>();
149  star_labels.push_back(ex->l.multi.label);
150  star_labels.push_back(LABEL_SKIP);
151  my_task_data->y_allowed_entity.push_back(LABEL_SKIP);
152  prediction = Search::predictor(sch, my_tag)
153  .set_input(*ex)
154  .set_oracle(star_labels)
155  .set_allowed(my_task_data->y_allowed_entity)
156  .set_learner_id(1)
157  .predict();
158  my_task_data->y_allowed_entity.pop();
159  }
160  else
161  {
162  if (isLdf)
163  {
164  for (uint32_t a = 0; a < 4; a++)
165  {
166  VW::copy_example_data(false, &my_task_data->ldf_entity[a], ex);
167  update_example_indicies(true, &my_task_data->ldf_entity[a], 28904713, 4832917 * (uint64_t)(a + 1));
168  CS::label& lab = my_task_data->ldf_entity[a].l.cs;
169  lab.costs[0].x = 0.f;
170  lab.costs[0].class_index = a;
171  lab.costs[0].partial_prediction = 0.f;
172  lab.costs[0].wap_value = 0.f;
173  }
174  prediction = Search::predictor(sch, my_tag)
175  .set_input(my_task_data->ldf_entity, 4)
176  .set_oracle(ex->l.multi.label - 1)
177  .set_learner_id(1)
178  .predict() +
179  1;
180  }
181  else
182  {
183  prediction = Search::predictor(sch, my_tag)
184  .set_input(*ex)
185  .set_oracle(ex->l.multi.label)
186  .set_allowed(my_task_data->y_allowed_entity)
187  .set_learner_id(0)
188  .predict();
189  }
190  }
191 
192  // record loss
193  float loss = 0.0;
194  if (prediction == LABEL_SKIP)
195  {
196  loss = my_task_data->skip_cost;
197  }
198  else if (prediction != ex->l.multi.label)
199  loss = my_task_data->entity_cost;
200  sch.loss(loss);
201  return prediction;
202 }
203 size_t predict_relation(Search::search& sch, example* ex, v_array<size_t>& predictions, ptag my_tag, bool isLdf = false)
204 {
205  char type;
206  int id1, id2;
207  task_data* my_task_data = sch.get_task_data<task_data>();
208  size_t hist[2];
209  decode_tag(ex->tag, type, id1, id2);
210  v_array<uint32_t> constrained_relation_labels = v_init<uint32_t>();
211  if (my_task_data->constraints && predictions[id1] != 0 && predictions[id2] != 0)
212  {
213  hist[0] = predictions[id1];
214  hist[1] = predictions[id2];
215  }
216  else
217  {
218  hist[0] = 0;
219  hist[1] = 0;
220  }
221  for (size_t j = 0; j < my_task_data->y_allowed_relation.size(); j++)
222  {
223  if (!my_task_data->constraints || hist[0] == (size_t)0 ||
224  check_constraints(hist[0], hist[1], my_task_data->y_allowed_relation[j]))
225  constrained_relation_labels.push_back(my_task_data->y_allowed_relation[j]);
226  }
227 
228  size_t prediction;
229  if (my_task_data->allow_skip)
230  {
231  v_array<uint32_t> star_labels = v_init<uint32_t>();
232  star_labels.push_back(ex->l.multi.label);
233  star_labels.push_back(LABEL_SKIP);
234  constrained_relation_labels.push_back(LABEL_SKIP);
235  prediction = Search::predictor(sch, my_tag)
236  .set_input(*ex)
237  .set_oracle(star_labels)
238  .set_allowed(constrained_relation_labels)
239  .set_learner_id(2)
240  .add_condition(id1, 'a')
241  .add_condition(id2, 'b')
242  .predict();
243  constrained_relation_labels.pop();
244  }
245  else
246  {
247  if (isLdf)
248  {
249  int correct_label = 0; // if correct label is not in the set, use the first one
250  for (size_t a = 0; a < constrained_relation_labels.size(); a++)
251  {
252  VW::copy_example_data(false, &my_task_data->ldf_relation[a], ex);
253  update_example_indicies(
254  true, &my_task_data->ldf_relation[a], 28904713, 4832917 * (uint64_t)(constrained_relation_labels[a]));
255  CS::label& lab = my_task_data->ldf_relation[a].l.cs;
256  lab.costs[0].x = 0.f;
257  lab.costs[0].class_index = constrained_relation_labels[a];
258  lab.costs[0].partial_prediction = 0.f;
259  lab.costs[0].wap_value = 0.f;
260  if (constrained_relation_labels[a] == ex->l.multi.label)
261  {
262  correct_label = (int)a;
263  }
264  }
265  size_t pred_pos = Search::predictor(sch, my_tag)
266  .set_input(my_task_data->ldf_relation, constrained_relation_labels.size())
267  .set_oracle(correct_label)
268  .set_learner_id(2)
269  .predict();
270  prediction = constrained_relation_labels[pred_pos];
271  }
272  else
273  {
274  prediction = Search::predictor(sch, my_tag)
275  .set_input(*ex)
276  .set_oracle(ex->l.multi.label)
277  .set_allowed(constrained_relation_labels)
278  .set_learner_id(1)
279  .predict();
280  }
281  }
282 
283  float loss = 0.0;
284  if (prediction == LABEL_SKIP)
285  {
286  loss = my_task_data->skip_cost;
287  }
288  else if (prediction != ex->l.multi.label)
289  {
290  if (ex->l.multi.label == R_NONE)
291  {
292  loss = my_task_data->relation_none_cost;
293  }
294  else
295  {
296  loss = my_task_data->relation_cost;
297  }
298  }
299  sch.loss(loss);
300  constrained_relation_labels.delete_v();
301  return prediction;
302 }

304 void entity_first_decoding(Search::search& sch, multi_ex& ec, v_array<size_t>& predictions, bool isLdf = false)
305 {
306  // ec.size = #entity + #entity*(#entity-1)/2
307  size_t n_ent = (size_t)(std::sqrt(ec.size() * 8 + 1) - 1) / 2;
308  // Do entity recognition first
309  for (size_t i = 0; i < ec.size(); i++)
310  {
311  if (i < n_ent)
312  predictions[i] = predict_entity(sch, ec[i], predictions, (ptag)i, isLdf);
313  else
314  predictions[i] = predict_relation(sch, ec[i], predictions, (ptag)i, isLdf);
315  }
316 }

318 void er_mixed_decoding(Search::search& sch, multi_ex& ec, v_array<size_t>& predictions)
319 {
320  // ec.size = #entity + #entity*(#entity-1)/2
321  uint32_t n_ent = (uint32_t)((std::sqrt(ec.size() * 8 + 1) - 1) / 2);
322  for (uint32_t t = 0; t < ec.size(); t++)
323  {
324  // Do entity recognition first
325  uint32_t count = 0;
326  for (ptag i = 0; i < n_ent; i++)
327  {
328  if (count == t)
329  {
330  predictions[i] = predict_entity(sch, ec[i], predictions, i);
331  break;
332  }
333  count++;
334  for (uint32_t j = 0; j < i; j++)
335  {
336  if (count == t)
337  {
338  ptag rel_index = (ptag)(n_ent + (2 * n_ent - j - 1) * j / 2 + i - j - 1);
339  predictions[rel_index] = predict_relation(sch, ec[rel_index], predictions, rel_index);
340  break;
341  }
342  count++;
343  }
344  }
345  }
346 }

348 void er_allow_skip_decoding(Search::search& sch, multi_ex& ec, v_array<size_t>& predictions)
349 {
350  task_data* my_task_data = sch.get_task_data<task_data>();
351  // ec.size = #entity + #entity*(#entity-1)/2
352  size_t n_ent = (size_t)(std::sqrt(ec.size() * 8 + 1) - 1) / 2;
353 
354  bool must_predict = false;
355  size_t n_predicts = 0;
356  size_t p_n_predicts = 0;
357  my_task_data->allow_skip = true;
358 
359  // loop until all the entity and relation types are predicted
360  for (ptag t = 0;; t++)
361  {
362  ptag i = t % (uint32_t)ec.size();
363  if (n_predicts == ec.size())
364  break;
365 
366  if (predictions[i] == 0)
367  {
368  if (must_predict)
369  {
370  my_task_data->allow_skip = false;
371  }
372  size_t prediction = 0;
373  if (i < n_ent) // do entity recognition
374  {
375  prediction = predict_entity(sch, ec[i], predictions, i);
376  }
377  else // do relation recognition
378  {
379  prediction = predict_relation(sch, ec[i], predictions, i);
380  }
381 
382  if (prediction != LABEL_SKIP)
383  {
384  predictions[i] = prediction;
385  n_predicts++;
386  }
387 
388  if (must_predict)
389  {
390  my_task_data->allow_skip = true;
391  must_predict = false;
392  }
393  }
394 
395  if (i == ec.size() - 1)
396  {
397  if (n_predicts == p_n_predicts)
398  {
399  must_predict = true;
400  }
401  p_n_predicts = n_predicts;
402  }
403  }

406 void run(Search::search& sch, multi_ex& ec)
407 {
408  task_data* my_task_data = sch.get_task_data<task_data>();
409 
410  v_array<size_t> predictions = v_init<size_t>();
411  for (size_t i = 0; i < ec.size(); i++)
412  {
413  predictions.push_back(0);
414  }
415 
416  switch (my_task_data->search_order)
417  {
418  case 0:
419  entity_first_decoding(sch, ec, predictions, false);
420  break;
421  case 1:
422  er_mixed_decoding(sch, ec, predictions);
423  break;
424  case 2:
425  er_allow_skip_decoding(sch, ec, predictions);
426  break;
427  case 3:
428  entity_first_decoding(sch, ec, predictions, true); // LDF = true
429  break;
430  default:
431  std::cerr << "search order " << my_task_data->search_order << "is undefined." << std::endl;
432  }
433 
434  for (size_t i = 0; i < ec.size(); i++)
435  {
436  if (sch.output().good())
437  sch.output() << predictions[i] << ' ';
438  }
439  predictions.delete_v();
440 }
