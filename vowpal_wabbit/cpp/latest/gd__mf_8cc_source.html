93 float mf_predict(gdmf& d, example& ec, T& weights)
94 {
95  vw& all = *d.all;
96  label_data& ld = ec.l.simple;
97  float prediction = ld.initial;
98 
99  for (std::string& i : d.all->pairs)
100  {
101  ec.num_features -= ec.feature_space[(int)i[0]].size() * ec.feature_space[(int)i[1]].size();
102  ec.num_features += ec.feature_space[(int)i[0]].size() * d.rank;
103  ec.num_features += ec.feature_space[(int)i[1]].size() * d.rank;
104  }
105 
106  // clear stored predictions
107  d.scalars.clear();
108 
109  float linear_prediction = 0.;
110  // linear terms
111 
112  for (features& fs : ec) GD::foreach_feature<float, GD::vec_add, T>(weights, fs, linear_prediction);
113 
114  // store constant + linear prediction
115  // note: constant is now automatically added
116  d.scalars.push_back(linear_prediction);
117 
118  prediction += linear_prediction;
119  // interaction terms
120  for (std::string& i : d.all->pairs)
121  {
122  if (ec.feature_space[(int)i[0]].size() > 0 && ec.feature_space[(int)i[1]].size() > 0)
123  {
124  for (uint64_t k = 1; k <= d.rank; k++)
125  {
126  // x_l * l^k
127  // l^k is from index+1 to index+d.rank
128  // float x_dot_l = sd_offset_add(weights, ec.atomics[(int)(*i)[0]].begin(), ec.atomics[(int)(*i)[0]].end(), k);
129  pred_offset x_dot_l = {0., k};
130  GD::foreach_feature<pred_offset, offset_add, T>(weights, ec.feature_space[(int)i[0]], x_dot_l);
131  // x_r * r^k
132  // r^k is from index+d.rank+1 to index+2*d.rank
133  // float x_dot_r = sd_offset_add(weights, ec.atomics[(int)(*i)[1]].begin(), ec.atomics[(int)(*i)[1]].end(),
134  // k+d.rank);
135  pred_offset x_dot_r = {0., k + d.rank};
136  GD::foreach_feature<pred_offset, offset_add, T>(weights, ec.feature_space[(int)i[1]], x_dot_r);
137 
138  prediction += x_dot_l.p * x_dot_r.p;
139 
140  // store prediction from interaction terms
141  d.scalars.push_back(x_dot_l.p);
142  d.scalars.push_back(x_dot_r.p);
143  }
144  }
145  }
146 
147  if (all.triples.begin() != all.triples.end())
148  THROW("cannot use triples in matrix factorization");
149 
150  // d.scalars has linear, x_dot_l_1, x_dot_r_1, x_dot_l_2, x_dot_r_2, ...
151 
152  ec.partial_prediction = prediction;
153 
154  all.set_minmax(all.sd, ld.label);
155 
156  ec.pred.scalar = GD::finalize_prediction(all.sd, all.logger, ec.partial_prediction);
157 
158  if (ld.label != FLT_MAX)
159  ec.loss = all.loss->getLoss(all.sd, ec.pred.scalar, ld.label) * ec.weight;
160 
161  if (all.audit)
162  mf_print_audit_features(d, ec, 0);
163 
164  return ec.pred.scalar;
165 }

183 template <class T>
184 void mf_train(gdmf& d, example& ec, T& weights)
185 {
186  vw& all = *d.all;
187  label_data& ld = ec.l.simple;
188 
189  // use final prediction to get update size
190  // update = eta_t*(y-y_hat) where eta_t = eta/(3*t^p) * importance weight
191  float eta_t = all.eta / powf((float)all.sd->t + ec.weight, (float)all.power_t) / 3.f * ec.weight;
192  float update = all.loss->getUpdate(ec.pred.scalar, ld.label, eta_t, 1.); // ec.total_sum_feat_sq);
193 
194  float regularization = eta_t * all.l2_lambda;
195 
196  // linear update
197  for (features& fs : ec) sd_offset_update<T>(weights, fs, 0, update, regularization);
198 
199  // quadratic update
200  for (std::string& i : all.pairs)
201  {
202  if (ec.feature_space[(int)i[0]].size() > 0 && ec.feature_space[(int)i[1]].size() > 0)
203  {
204  // update l^k weights
205  for (size_t k = 1; k <= d.rank; k++)
206  {
207  // r^k \cdot x_r
208  float r_dot_x = d.scalars[2 * k];
209  // l^k <- l^k + update * (r^k \cdot x_r) * x_l
210  sd_offset_update<T>(weights, ec.feature_space[(int)i[0]], k, update * r_dot_x, regularization);
211  }
212  // update r^k weights
213  for (size_t k = 1; k <= d.rank; k++)
214  {
215  // l^k \cdot x_l
216  float l_dot_x = d.scalars[2 * k - 1];
217  // r^k <- r^k + update * (l^k \cdot x_l) * x_r
218  sd_offset_update<T>(weights, ec.feature_space[(int)i[1]], k + d.rank, update * l_dot_x, regularization);
219  }
220  }
221  }
222  if (all.triples.begin() != all.triples.end())
223  THROW("cannot use triples in matrix factorization");
224 }

245 void save_load(gdmf& d, io_buf& model_file, bool read, bool text)
246 {
247  vw& all = *d.all;
248  uint64_t length = (uint64_t)1 << all.num_bits;
249  if (read)
250  {
251  initialize_regressor(all);
252  if (all.random_weights)
253  {
254  uint32_t stride = all.weights.stride();
255  if (all.weights.sparse)
256  all.weights.sparse_weights.set_default<uint32_t, set_rand_wrapper<sparse_parameters> >(stride);
257  else
258  all.weights.dense_weights.set_default<uint32_t, set_rand_wrapper<dense_parameters> >(stride);
259  }
260  }
261 
262  if (model_file.files.size() > 0)
263  {
264  uint64_t i = 0;
265  size_t brw = 1;
266  do
267  {
268  brw = 0;
269  size_t K = d.rank * 2 + 1;
270  std::stringstream msg;
271  msg << i << " ";
272  brw += bin_text_read_write_fixed(model_file, (char*)&i, sizeof(i), "", read, msg, text);
273  if (brw != 0)
274  {
275  weight* w_i = &(all.weights.strided_index(i));
276  for (uint64_t k = 0; k < K; k++)
277  {
278  weight* v = w_i + k;
279  msg << v << " ";
280  brw += bin_text_read_write_fixed(model_file, (char*)v, sizeof(*v), "", read, msg, text);
281  }
282  }
283  if (text)
284  {
285  msg << "\n";
286  brw += bin_text_read_write_fixed(model_file, nullptr, 0, "", read, msg, text);
287  }
288 
289  if (!read)
290  ++i;
291  } while ((!read && i < length) || (read && brw > 0));
292  }

332 
333  // Not supported, need to be checked to be false.

335  gf_md_options.add(
336  make_option("conjugate_gradient", conjugate_gradient).help("Option not supported by this reduction"));
337  options.add_and_parse(gf_md_options);
338 
339  if (!options.was_supplied("rank"))
340  return nullptr;
341 
342  if (options.was_supplied("adaptive"))
343  THROW("adaptive is not implemented for matrix factorization");
344  if (options.was_supplied("normalized"))
345  THROW("normalized is not implemented for matrix factorization");
346  if (options.was_supplied("exact_adaptive_norm"))
347  THROW("normalized adaptive updates is not implemented for matrix factorization");
348 
349  if (bfgs || conjugate_gradient)
350  THROW("bfgs is not implemented for matrix factorization");
351 
352  data->all = &all;
353  data->no_win_counter = 0;
354 
355  // store linear + 2*rank weights per index, round up to power of two
356  float temp = ceilf(logf((float)(data->rank * 2 + 1)) / logf(2.f));
357  all.weights.stride_shift((size_t)temp);
358  all.random_weights = true;
359 
360  if (!all.holdout_set_off)
361  {
362  all.sd->holdout_best_loss = FLT_MAX;
363  data->early_stop_thres = options.get_typed_option<size_t>("early_terminate").value();
364  }
365 
366  if (!options.was_supplied("learning_rate") && !options.was_supplied("l"))
367  all.eta = 10; // default learning rate to 10 for non default update rule
368 
369  // default initial_t to 1 instead of 0
370  if (!options.was_supplied("initial_t"))
371  {
372  all.sd->t = 1.f;
373  all.initial_t = 1.f;
374  }
375  all.eta *= powf((float)(all.sd->t), all.power_t);
376 
377  learner<gdmf, example>& l = init_learner(data, learn, predict, (UINT64_ONE << all.weights.stride_shift()));
378  l.set_save_load(save_load);
379  l.set_end_pass(end_pass);
380 
381  return make_base(l);
