135  template <class T, void (*f)(T&, const T&)>
136  void all_reduce(T* buffer, const size_t n)
137  { // register buffer
138  T** buffers = (T**)m_sync->buffers;
139  buffers[node] = buffer;
140  m_sync->waitForSynchronization();
141 
142  size_t blockSize = n / total;
143  size_t index;
144  size_t end;
145 
146  if (blockSize == 0)
147  {
148  if (node < n)
149  {
150  index = node;
151  end = node + 1;
152  }
153  else
154  { // more threads than bytes --> don't do any work
155  index = end = 0;
156  }
157  }
158  else
159  {
160  index = node * blockSize;
161  end = node == total - 1 ? n : (node + 1) * blockSize;
162  }
163 
164  for (; index < end; index++)
165  { // Perform transposed AllReduce to help data locallity
166  T& first = buffers[0][index];
167 
168  for (size_t i = 1; i < total; i++) f(first, buffers[i][index]);
169 
170  // Broadcast back
171  for (size_t i = 1; i < total; i++) buffers[i][index] = first;
172  }
173 
174  m_sync->waitForSynchronization();
175  }

205  template <class T, void (*f)(T&, const T&)>
206  void reduce(char* buffer, const size_t n)
207  {
208  fd_set fds;
209  FD_ZERO(&fds);
210  if (socks.children[0] != -1)
211  FD_SET(socks.children[0], &fds);
212  if (socks.children[1] != -1)
213  FD_SET(socks.children[1], &fds);
214 
215  socket_t max_fd = std::max(socks.children[0], socks.children[1]) + 1;
216  size_t child_read_pos[2] = {0, 0}; // First unread float from left and right children
217  int child_unprocessed[2] = {0, 0}; // The number of bytes sent by the child but not yet added to the buffer
218  char child_read_buf[2][ar_buf_size + sizeof(T) - 1];
219  size_t parent_sent_pos = 0; // First unsent float to parent
220  // parent_sent_pos <= left_read_pos
221  // parent_sent_pos <= right_read_pos
222 
223  if (socks.children[0] == -1)
224  {
225  child_read_pos[0] = n;
226  }
227  if (socks.children[1] == -1)
228  {
229  child_read_pos[1] = n;
230  }
231 
232  while (parent_sent_pos < n || child_read_pos[0] < n || child_read_pos[1] < n)
233  {
234  if (socks.parent != -1)
235  pass_up<T>(buffer, child_read_pos[0], child_read_pos[1], parent_sent_pos);
236 
237  if (parent_sent_pos >= n && child_read_pos[0] >= n && child_read_pos[1] >= n)
238  break;
239 
240  if (child_read_pos[0] < n || child_read_pos[1] < n)
241  {
242  if (max_fd > 0 && select((int)max_fd, &fds, nullptr, nullptr, nullptr) == -1)
243  THROWERRNO("select");
244 
245  for (int i = 0; i < 2; i++)
246  {
247  if (socks.children[i] != -1 && FD_ISSET(socks.children[i], &fds))
248  { // there is data to be left from left child
249  if (child_read_pos[i] == n)
250  THROW("I think child has no data to send but he thinks he has "
251  << FD_ISSET(socks.children[0], &fds) << " " << FD_ISSET(socks.children[1], &fds));
252 
253  size_t count = std::min(ar_buf_size, n - child_read_pos[i]);
254  int read_size = recv(socks.children[i], &child_read_buf[i][child_unprocessed[i]], (int)count, 0);
255  if (read_size == -1)
256  THROWERRNO("recv from child");
257 
258  addbufs<T, f>((T*)buffer + child_read_pos[i] / sizeof(T), (T*)child_read_buf[i],
259  (child_read_pos[i] + read_size) / sizeof(T) - child_read_pos[i] / sizeof(T));
260 
261  child_read_pos[i] += read_size;
262  int old_unprocessed = child_unprocessed[i];
263  child_unprocessed[i] = child_read_pos[i] % (int)sizeof(T);
264  for (int j = 0; j < child_unprocessed[i]; j++)
265  {
266  child_read_buf[i][j] =
267  child_read_buf[i][((old_unprocessed + read_size) / (int)sizeof(T)) * sizeof(T) + j];
268  }
269 
270  if (child_read_pos[i] == n) // Done reading parent
271  FD_CLR(socks.children[i], &fds);
272  }
273  else if (socks.children[i] != -1 && child_read_pos[i] != n)
274  FD_SET(socks.children[i], &fds);
275  }
276  }
277  if (socks.parent == -1 && child_read_pos[0] == n && child_read_pos[1] == n)
278  parent_sent_pos = n;
279  }
280  }
