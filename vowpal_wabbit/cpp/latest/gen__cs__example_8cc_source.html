85 void gen_cs_example_ips(cb_to_cs& c, CB::label& ld, COST_SENSITIVE::label& cs_ld, float clip_p)
86 {
87  // this implements the inverse propensity score method, where cost are importance weighted by the probability of the
88  // chosen action generate cost-sensitive example
89  cs_ld.costs.clear();
90  if (ld.costs.size() == 0 || (ld.costs.size() == 1 && ld.costs[0].cost != FLT_MAX))
91  // this is a typical example where we can perform all actions
92  {
93  // in this case generate cost-sensitive example with all actions
94  for (uint32_t i = 1; i <= c.num_actions; i++)
95  {
96  COST_SENSITIVE::wclass wc = {0., i, 0., 0.};
97  if (c.known_cost != nullptr && i == c.known_cost->action)
98  {
99  // use importance weighted cost for observed action, 0 otherwise
100  wc.x = c.known_cost->cost / safe_probability(std::max(c.known_cost->probability, clip_p));
101 
102  // ips can be thought as the doubly robust method with a fixed regressor that predicts 0 costs for everything
103  // update the loss of this regressor
104  c.nb_ex_regressors++;
105  c.avg_loss_regressors +=
106  (1.0f / c.nb_ex_regressors) * ((c.known_cost->cost) * (c.known_cost->cost) - c.avg_loss_regressors);
107  c.last_pred_reg = 0;
108  c.last_correct_cost = c.known_cost->cost;
109  }
110 
111  cs_ld.costs.push_back(wc);
112  }
113  }
114  else // this is an example where we can only perform a subset of the actions
115  {
116  // in this case generate cost-sensitive example with only allowed actions
117  for (auto& cl : ld.costs)
118  {
119  COST_SENSITIVE::wclass wc = {0., cl.action, 0., 0.};
120  if (c.known_cost != nullptr && cl.action == c.known_cost->action)
121  {
122  // use importance weighted cost for observed action, 0 otherwise
123  wc.x = c.known_cost->cost / safe_probability(std::max(c.known_cost->probability, clip_p));
124 
125  // ips can be thought as the doubly robust method with a fixed regressor that predicts 0 costs for everything
126  // update the loss of this regressor
127  c.nb_ex_regressors++;
128  c.avg_loss_regressors +=
129  (1.0f / c.nb_ex_regressors) * ((c.known_cost->cost) * (c.known_cost->cost) - c.avg_loss_regressors);
130  c.last_pred_reg = 0;
131  c.last_correct_cost = c.known_cost->cost;
132  }
133 
134  cs_ld.costs.push_back(wc);
135  }
136  }
137 }
