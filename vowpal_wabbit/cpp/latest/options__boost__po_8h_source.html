74 
75  // Not found, insert instead.

85  THROW(key + " option does not have a value.");

217 template <typename T>
218 po::typed_value<std::vector<T>>* options_boost_po::add_notifier(
219  std::shared_ptr<typed_option<T>>& opt, po::typed_value<std::vector<T>>* po_value)
220 {
221  return po_value->notifier([opt](std::vector<T> final_arguments) {
222  T result = final_arguments[0];
223 
224  // Due to the way options get added to the vector, the model options are at the end, and the
225  // command-line options are at the front. To allow override from command-line over model file,
226  // simply keep the first item, and suppress the error.
227  if (!opt->m_allow_override)
228  {
229  for (auto const& item : final_arguments)
230  {
231  if(item != result)
232  {
233  std::stringstream ss;
234  ss << "Disagreeing option values for '" << opt->m_name << "': '" << result << "' vs '" << item << "'";
235  THROW_EX(VW::vw_argument_disagreement_exception, ss.str());
236  }
237  }
238  }
239 
240  // Set the value for the listening location.
241  opt->m_location = result;
242  opt->value(result);
243  });
244 }
