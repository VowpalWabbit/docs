102 size_t create_circuit(ect& e, uint64_t max_label, uint64_t eliminations)
103 {
104  if (max_label == 1)
105  return 0;
106 
107  v_array<v_array<uint32_t>> tournaments = v_init<v_array<uint32_t>>();
108  v_array<uint32_t> t = v_init<uint32_t>();
109 
110  for (uint32_t i = 0; i < max_label; i++)
111  {
112  t.push_back(i);
113  direction d = {i, 0, 0, 0, 0, 0, false};
114  e.directions.push_back(d);
115  }
116 
117  tournaments.push_back(t);
118 
119  for (size_t i = 0; i < eliminations - 1; i++) tournaments.push_back(v_array<uint32_t>());
120 
121  e.all_levels.push_back(tournaments);
122 
123  size_t level = 0;
124 
125  uint32_t node = (uint32_t)e.directions.size();
126 
127  while (not_empty(e.all_levels[level]))
128  {
129  v_array<v_array<uint32_t>> new_tournaments = v_init<v_array<uint32_t>>();
130  tournaments = e.all_levels[level];
131 
132  for (size_t t = 0; t < tournaments.size(); t++)
133  {
134  v_array<uint32_t> empty = v_init<uint32_t>();
135  new_tournaments.push_back(empty);
136  }
137 
138  for (size_t t = 0; t < tournaments.size(); t++)
139  {
140  for (size_t j = 0; j < tournaments[t].size() / 2; j++)
141  {
142  uint32_t id = node++;
143  uint32_t left = tournaments[t][2 * j];
144  uint32_t right = tournaments[t][2 * j + 1];
145 
146  direction d = {id, t, 0, 0, left, right, false};
147  e.directions.push_back(d);
148  uint32_t direction_index = (uint32_t)e.directions.size() - 1;
149  if (e.directions[left].tournament == t)
150  e.directions[left].winner = direction_index;
151  else
152  e.directions[left].loser = direction_index;
153  if (e.directions[right].tournament == t)
154  e.directions[right].winner = direction_index;
155  else
156  e.directions[right].loser = direction_index;
157  if (e.directions[left].last)
158  e.directions[left].winner = direction_index;
159 
160  if (tournaments[t].size() == 2 && (t == 0 || tournaments[t - 1].empty()))
161  {
162  e.directions[direction_index].last = true;
163  if (t + 1 < tournaments.size())
164  new_tournaments[t + 1].push_back(id);
165  else // winner eliminated.
166  e.directions[direction_index].winner = 0;
167  e.final_nodes.push_back((uint32_t)(e.directions.size() - 1));
168  }
169  else
170  new_tournaments[t].push_back(id);
171  if (t + 1 < tournaments.size())
172  new_tournaments[t + 1].push_back(id);
173  else // loser eliminated.
174  e.directions[direction_index].loser = 0;
175  }
176  if (tournaments[t].size() % 2 == 1)
177  new_tournaments[t].push_back(tournaments[t].last());
178  }
179  e.all_levels.push_back(new_tournaments);
180  level++;
181  }
182 
183  e.last_pair = (uint32_t)((max_label - 1) * eliminations);
184 
185  if (max_label > 1)
186  e.tree_height = final_depth(eliminations);
187 
188  return e.last_pair + (eliminations - 1);
189 }

191 uint32_t ect_predict(ect& e, single_learner& base, example& ec)
192 {
193  if (e.k == (size_t)1)
194  return 1;
195 
196  uint32_t finals_winner = 0;
197 
198  // Binary final elimination tournament first
199  ec.l.simple = {FLT_MAX, 0., 0.};
200 
201  for (size_t i = e.tree_height - 1; i != (size_t)0 - 1; i--)
202  {
203  if ((finals_winner | (((size_t)1) << i)) <= e.errors)
204  {
205  // a real choice exists
206  uint32_t problem_number = e.last_pair + (finals_winner | (((uint32_t)1) << i)) - 1; // This is unique.
207 
208  base.learn(ec, problem_number);
209 
210  if (ec.pred.scalar > e.class_boundary)
211  finals_winner = finals_winner | (((size_t)1) << i);
212  }
213  }
214 
215  uint32_t id = e.final_nodes[finals_winner];
216  while (id >= e.k)
217  {
218  base.learn(ec, id - e.k);
219 
220  if (ec.pred.scalar > e.class_boundary)
221  id = e.directions[id].right;
222  else
223  id = e.directions[id].left;
224  }
225  return id + 1;
226 }

228 void ect_train(ect& e, single_learner& base, example& ec)
229 {
230  if (e.k == 1) // nothing to do
231  return;
232  MULTICLASS::label_t mc = ec.l.multi;
233 
234  label_data simple_temp;
235 
236  simple_temp.initial = 0.;
237 
238  e.tournaments_won.clear();
239 
240  uint32_t id = e.directions[mc.label - 1].winner;
241  bool left = e.directions[id].left == mc.label - 1;
242  do
243  {
244  if (left)
245  simple_temp.label = -1;
246  else
247  simple_temp.label = 1;
248 
249  ec.l.simple = simple_temp;
250  base.learn(ec, id - e.k);
251  float old_weight = ec.weight;
252  ec.weight = 0.;
253  base.learn(ec, id - e.k); // inefficient, we should extract final prediction exactly.
254  ec.weight = old_weight;
255 
256  bool won = (ec.pred.scalar - e.class_boundary) * simple_temp.label > 0;
257 
258  if (won)
259  {
260  if (!e.directions[id].last)
261  left = e.directions[e.directions[id].winner].left == id;
262  else
263  e.tournaments_won.push_back(true);
264  id = e.directions[id].winner;
265  }
266  else
267  {
268  if (!e.directions[id].last)
269  {
270  left = e.directions[e.directions[id].loser].left == id;
271  if (e.directions[id].loser == 0)
272  e.tournaments_won.push_back(false);
273  }
274  else
275  e.tournaments_won.push_back(false);
276  id = e.directions[id].loser;
277  }
278  } while (id != 0);
279 
280  if (e.tournaments_won.empty())
281  std::cout << "badness!" << std::endl;
282 
283  // tournaments_won is a bit vector determining which tournaments the label won.
284  for (size_t i = 0; i < e.tree_height; i++)
285  {
286  for (uint32_t j = 0; j < e.tournaments_won.size() / 2; j++)
287  {
288  bool left = e.tournaments_won[j * 2];
289  bool right = e.tournaments_won[j * 2 + 1];
290  if (left == right) // no query to do
291  e.tournaments_won[j] = left;
292  else // query to do
293  {
294  if (left)
295  simple_temp.label = -1;
296  else
297  simple_temp.label = 1;
298  simple_temp.weight = (float)(1 << (e.tree_height - i - 1));
299  ec.l.simple = simple_temp;
300 
301  uint32_t problem_number = e.last_pair + j * (1 << (i + 1)) + (1 << i) - 1;
302 
303  base.learn(ec, problem_number);
304 
305  if (ec.pred.scalar > e.class_boundary)
306  e.tournaments_won[j] = right;
307  else
308  e.tournaments_won[j] = left;
309  }
310  if (e.tournaments_won.size() % 2 == 1)
311  e.tournaments_won[e.tournaments_won.size() / 2] = e.tournaments_won[e.tournaments_won.size() - 1];
312  e.tournaments_won.end() = e.tournaments_won.begin() + (1 + e.tournaments_won.size()) / 2;
313  }
314  }
315 }
