148 void compute_recall_lbest(recall_tree& b, node* n)
149 {
150  if (n->n <= 0)
151  return;
152 
153  double mass_at_k = 0;
154 
155  for (node_pred* ls = n->preds.begin(); ls != n->preds.end() && ls < n->preds.begin() + b.max_candidates; ++ls)
156  {
157  mass_at_k += ls->label_count;
158  }
159 
160  float f = (float)mass_at_k / (float)n->n;
161  float stdf = std::sqrt(f * (1.f - f) / (float)n->n);
162  float diamf = 15.f / (std::sqrt(18.f) * (float)n->n);
163 
164  n->recall_lbest = std::max(0.f, f - std::sqrt(b.bern_hyper) * stdf - b.bern_hyper * diamf);
165 }

169 double updated_entropy(recall_tree& b, uint32_t cn, example& ec)
170 {
171  node_pred* ls = find(b, cn, ec);
172 
173  // entropy = -\sum_k (c_k/n) Log[c_k/n]
174  // c_0 <- c_0 + 1, n <- n + 1
175  // entropy <- + (c_0/n) Log[c_0/n]
176  // - n/(n+1) \sum_{k>0} (c_k/n) Log[c_k/n]
177  // - Log[n/(n+1)] \sum_{k>0} (c_k/(n+1))
178  // - ((c_0+1)/(n+1)) Log[(c_0+1)/(n+1)]
179 
180  double c0 = (ls == b.nodes[cn].preds.end()) ? 0 : ls->label_count;
181  double deltac0 = ec.weight;
182  double n = b.nodes[cn].n;
183 
184  double novernp1 = n / (deltac0 + n);
185  double lognovernp1 = (novernp1 == 0) ? 0 : log(novernp1);
186  double nminusc0overnp1 = (n - c0) / (n + deltac0);
187 
188  double newentropy = b.nodes[cn].entropy;
189 
190  newentropy += plogp(c0, n);
191  newentropy *= novernp1;
192  newentropy -= lognovernp1 * nminusc0overnp1;
193  newentropy -= plogp(c0 + deltac0, n + deltac0);
194 
195  return newentropy;
196 }

340 float train_node(recall_tree& b, single_learner& base, example& ec, uint32_t cn)
341 {
342  MULTICLASS::label_t mc = ec.l.multi;
343  uint32_t save_pred = ec.pred.multiclass;
344 
345  // minimize entropy
346  // better than maximize expected likelihood, and the proofs go through :)
347  double new_left = updated_entropy(b, b.nodes[cn].left, ec);
348  double new_right = updated_entropy(b, b.nodes[cn].right, ec);
349  double old_left = b.nodes[b.nodes[cn].left].entropy;
350  double old_right = b.nodes[b.nodes[cn].right].entropy;
351  double nl = b.nodes[b.nodes[cn].left].n;
352  double nr = b.nodes[b.nodes[cn].right].n;
353  double delta_left = nl * (new_left - old_left) + mc.weight * new_left;
354  double delta_right = nr * (new_right - old_right) + mc.weight * new_right;
355  float route_label = delta_left < delta_right ? -1.f : 1.f;
356  float imp_weight = fabs((float)(delta_left - delta_right));
357 
358  ec.l.simple = {route_label, imp_weight, 0.};
359  base.learn(ec, b.nodes[cn].base_router);
360 
361  // TODO: using the updated routing seems to help
362  // TODO: consider faster version using updated_prediction
363  // TODO: (doesn't play well with link function)
364  base.predict(ec, b.nodes[cn].base_router);
365 
366  float save_scalar = ec.pred.scalar;
367 
368  ec.l.multi = mc;
369  ec.pred.multiclass = save_pred;
370 
371  return save_scalar;
372 }

374 void learn(recall_tree& b, single_learner& base, example& ec)
375 {
376  predict(b, base, ec);
377 
378  if (b.all->training && ec.l.multi.label != (uint32_t)-1) // if training the tree
379  {
380  uint32_t cn = 0;
381 
382  while (b.nodes[cn].internal)
383  {
384  float which = train_node(b, base, ec, cn);
385 
386  if (b.randomized_routing)
387  which = (b._random_state->get_and_update_random() > to_prob(which) ? -1.f : 1.f);
388 
389  uint32_t newcn = descend(b.nodes[cn], which);
390  bool cond = stop_recurse_check(b, cn, newcn);
391  insert_example_at_node(b, cn, ec);
392 
393  if (cond)
394  {
395  insert_example_at_node(b, newcn, ec);
396  break;
397  }
398 
399  cn = newcn;
400  }
401 
402  if (!b.nodes[cn].internal)
403  insert_example_at_node(b, cn, ec);
404 
405  if (is_candidate(b, cn, ec))
406  {
407  MULTICLASS::label_t mc = ec.l.multi;
408  uint32_t save_pred = ec.pred.multiclass;
409 
410  add_node_id_feature(b, cn, ec);
411 
412  ec.l.simple = {1.f, 1.f, 0.f};
413  base.learn(ec, b.max_routers + mc.label - 1);
414  ec.l.simple = {-1.f, 1.f, 0.f};
415 
416  for (node_pred* ls = b.nodes[cn].preds.begin();
417  ls != b.nodes[cn].preds.end() && ls < b.nodes[cn].preds.begin() + b.max_candidates; ++ls)
418  {
419  if (ls->label != mc.label)
420  base.learn(ec, b.max_routers + ls->label - 1);
421  }
422 
423  remove_node_id_feature(b, cn, ec);
424 
425  ec.l.multi = mc;
426  ec.pred.multiclass = save_pred;
427  }
428  }
