42 void multiply(features& f_dest, features& f_src2, interact& in)
43 {
44  f_dest.clear();
45  features& f_src1 = in.feat_store;
46  vw* all = in.all;
47  uint64_t weight_mask = all->weights.mask();
48  uint64_t base_id1 = f_src1.indicies[0] & weight_mask;
49  uint64_t base_id2 = f_src2.indicies[0] & weight_mask;
50 
51  f_dest.push_back(f_src1.values[0] * f_src2.values[0], f_src1.indicies[0]);
52 
53  uint64_t prev_id1 = 0;
54  uint64_t prev_id2 = 0;
55 
56  for (size_t i1 = 1, i2 = 1; i1 < f_src1.size() && i2 < f_src2.size();)
57  {
58  // calculating the relative offset from the namespace offset used to match features
59  uint64_t cur_id1 = (uint64_t)(((f_src1.indicies[i1] & weight_mask) - base_id1) & weight_mask);
60  uint64_t cur_id2 = (uint64_t)(((f_src2.indicies[i2] & weight_mask) - base_id2) & weight_mask);
61 
62  // checking for sorting requirement
63  if (cur_id1 < prev_id1)
64  {
65  std::cout << "interact features are out of order: " << cur_id1 << " < " << prev_id1 << ". Skipping features."
66  << std::endl;
67  return;
68  }
69 
70  if (cur_id2 < prev_id2)
71  {
72  std::cout << "interact features are out of order: " << cur_id2 << " < " << prev_id2 << ". Skipping features."
73  << std::endl;
74  return;
75  }
76 
77  if (cur_id1 == cur_id2)
78  {
79  f_dest.push_back(f_src1.values[i1] * f_src2.values[i2], f_src1.indicies[i1]);
80  i1++;
81  i2++;
82  }
83  else if (cur_id1 < cur_id2)
84  i1++;
85  else
86  i2++;
87  prev_id1 = cur_id1;
88  prev_id2 = cur_id2;
89  }
90 }
