124 void parse_label(parser* p, shared_data* sd, void* v, v_array<VW::string_view>& words)
125 {
126  label* ld = (label*)v;
127  ld->costs.clear();
128 
129  // handle shared and label first
130  if (words.size() == 1)
131  {
132  float fx;
133  name_value(words[0], p->parse_name, fx);
134  bool eq_shared = p->parse_name[0] == "***shared***";
135  bool eq_label = p->parse_name[0] == "***label***";
136  if (!sd->ldict)
137  {
138  eq_shared |= p->parse_name[0] == "shared";
139  eq_label |= p->parse_name[0] == "label";
140  }
141  if (eq_shared || eq_label)
142  {
143  if (eq_shared)
144  {
145  if (p->parse_name.size() != 1)
146  std::cerr << "shared feature vectors should not have costs on: " << words[0] << std::endl;
147  else
148  {
149  wclass f = {-FLT_MAX, 0, 0., 0.};
150  ld->costs.push_back(f);
151  }
152  }
153  if (eq_label)
154  {
155  if (p->parse_name.size() != 2)
156  std::cerr << "label feature vectors should have exactly one cost on: " << words[0] << std::endl;
157  else
158  {
159  wclass f = {float_of_string(p->parse_name[1]), 0, 0., 0.};
160  ld->costs.push_back(f);
161  }
162  }
163  return;
164  }
165  }
166 
167  // otherwise this is a "real" example
168  for (unsigned int i = 0; i < words.size(); i++)
169  {
170  wclass f = {0., 0, 0., 0.};
171  name_value(words[i], p->parse_name, f.x);
172 
173  if (p->parse_name.size() == 0)
174  THROW(" invalid cost: specification -- no names on: " << words[i]);
175 
176  if (p->parse_name.size() == 1 || p->parse_name.size() == 2 || p->parse_name.size() == 3)
177  {
178  f.class_index = sd->ldict ? (uint32_t)sd->ldict->get(p->parse_name[0])
179  : (uint32_t)hashstring(p->parse_name[0].begin(), p->parse_name[0].length(), 0);
180  if (p->parse_name.size() == 1 && f.x >= 0) // test examples are specified just by un-valued class #s
181  f.x = FLT_MAX;
182  }
183  else
184  THROW("malformed cost specification on '" << (p->parse_name[0]) << "'");
185 
186  ld->costs.push_back(f);
187  }
188 }

239 void output_example(vw& all, example& ec)
240 {
241  label& ld = ec.l.cs;
242 
243  float loss = 0.;
244  if (!test_label(&ld))
245  {
246  // need to compute exact loss
247  size_t pred = (size_t)ec.pred.multiclass;
248 
249  float chosen_loss = FLT_MAX;
250  float min = FLT_MAX;
251  for (auto& cl : ld.costs)
252  {
253  if (cl.class_index == pred)
254  chosen_loss = cl.x;
255  if (cl.x < min)
256  min = cl.x;
257  }
258  if (chosen_loss == FLT_MAX)
259  std::cerr << "warning: csoaa predicted an invalid class. Are all multi-class labels in the {1..k} range?"
260  << std::endl;
261 
262  loss = (chosen_loss - min) * ec.weight;
263  // TODO(alberto): add option somewhere to allow using absolute loss instead?
264  // loss = chosen_loss;
265  }
266 
267  all.sd->update(ec.test_only, !test_label(&ld), loss, ec.weight, ec.num_features);
268 
269  for (int sink : all.final_prediction_sink)
270  if (!all.sd->ldict)
271  all.print_by_ref(sink, (float)ec.pred.multiclass, 0, ec.tag);
272  else
273  {
274  VW::string_view sv_pred = all.sd->ldict->get(ec.pred.multiclass);
275  all.print_text_by_ref(sink, sv_pred.to_string(), ec.tag);
276  }
277 
278  if (all.raw_prediction > 0)
279  {
280  std::stringstream outputStringStream;
281  for (unsigned int i = 0; i < ld.costs.size(); i++)
282  {
283  wclass cl = ld.costs[i];

286  outputStringStream << cl.class_index << ':' << cl.partial_prediction;
287  }

289  }
290 
291  print_update(all, test_label(&ec.l.cs), ec, nullptr, false, ec.pred.multiclass);
292 }
