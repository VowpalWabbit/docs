25 void AllReduceSync::waitForSynchronization()
26 {
27  std::unique_lock<std::mutex> l(*m_mutex);
28  m_count++;
29 
30  if (m_count >= m_total)
31  {
32  assert(m_count == m_total);
33 
34  m_cv->notify_all();
35 
36  // order of m_count before or after notify_all doesn't matter
37  // since the lock is still hold at this point in time.
38  m_count = 0;
39 
40  // flip for the next run
41  m_run = !m_run;
42  }
43  else
44  {
45  bool current_run = m_run;
46  // this predicate cannot depend on m_count, as somebody can race ahead and m_count++
47  // FYI just wait can spuriously wake-up
48  m_cv->wait(l, [this, current_run] { return m_run != current_run; });
49  }
50 }
