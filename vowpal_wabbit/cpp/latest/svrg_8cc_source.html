80 inline void update_inner_feature(update& u, float x, float& w)
81 {
82  float* ws = &w;
83  w -= u.eta * ((u.g_scalar_inner - u.g_scalar_stable) * x + ws[W_STABLEGRAD] / u.norm);
84 }

86 inline void update_stable_feature(float& g_scalar, float x, float& w)
87 {
88  float* ws = &w;
89  ws[W_STABLEGRAD] += g_scalar * x;
90 }

92 void update_inner(const svrg& s, example& ec)
93 {
94  update u;
95  // |ec| already has prediction according to inner weights.
96  u.g_scalar_inner = gradient_scalar(s, ec, ec.pred.scalar);
97  u.g_scalar_stable = gradient_scalar(s, ec, predict_stable(s, ec));
98  u.eta = s.all->eta;
99  u.norm = (float)s.stable_grad_count;
100  GD::foreach_feature<update, update_inner_feature>(*s.all, ec, u);
101 }

109 void learn(svrg& s, single_learner& base, example& ec)
110 {
111  predict(s, base, ec);
112 
113  const int pass = (int)s.all->passes_complete;
114 
115  if (pass % (s.stage_size + 1) == 0) // Compute exact gradient
116  {
117  if (s.prev_pass != pass && !s.all->logger.quiet)
118  {
119  std::cout << "svrg pass " << pass << ": committing stable point" << std::endl;
120  for (uint32_t j = 0; j < VW::num_weights(*s.all); j++)
121  {
122  float w = VW::get_weight(*s.all, j, W_INNER);
123  VW::set_weight(*s.all, j, W_STABLE, w);
124  VW::set_weight(*s.all, j, W_STABLEGRAD, 0.f);
125  }
126  s.stable_grad_count = 0;
127  std::cout << "svrg pass " << pass << ": computing exact gradient" << std::endl;
128  }
129  update_stable(s, ec);
130  s.stable_grad_count++;
131  }
132  else // Perform updates
133  {
134  if (s.prev_pass != pass && !s.all->logger.quiet)
135  {
136  std::cout << "svrg pass " << pass << ": taking steps" << std::endl;
137  }
138  update_inner(s, ec);
139  }
140 
141  s.prev_pass = pass;
142 }
