623 void addgrams(vw& all, size_t ngram, size_t skip_gram, features& fs, size_t initial_length, v_array<size_t>& gram_mask,
624  size_t skips)
625 {
626  if (ngram == 0 && gram_mask.last() < initial_length)
627  {
628  size_t last = initial_length - gram_mask.last();
629  for (size_t i = 0; i < last; i++)
630  {
631  uint64_t new_index = fs.indicies[i];
632  for (size_t n = 1; n < gram_mask.size(); n++)
633  new_index = new_index * quadratic_constant + fs.indicies[i + gram_mask[n]];
634

636  if (fs.space_names.size() > 0)
637  {

639  for (size_t n = 1; n < gram_mask.size(); n++)

642  feature_name += std::string(fs.space_names[i + gram_mask[n]].get()->second);

645  }
646  }
647  }
648  if (ngram > 0)
649  {
650  gram_mask.push_back(gram_mask.last() + 1 + skips);
651  addgrams(all, ngram - 1, skip_gram, fs, initial_length, gram_mask, 0);
652  gram_mask.pop();
653  }
654  if (skip_gram > 0 && ngram > 0)
655  addgrams(all, ngram, skip_gram - 1, fs, initial_length, gram_mask, skips + 1);
