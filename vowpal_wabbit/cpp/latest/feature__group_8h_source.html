395  bool sort(uint64_t parse_mask)
396  {
397  if (indicies.empty())
398  return false;
399 
400  if (!space_names.empty())
401  {
402  std::vector<feature_slice> slice;
403  slice.reserve(indicies.size());
404  for (size_t i = 0; i < indicies.size(); i++)
405  {
406  slice.push_back({values[i], indicies[i] & parse_mask, *space_names[i].get()});
407  }
408  // The comparator should return true if the first element is less than the second.
409  std::sort(slice.begin(), slice.end(), [](const feature_slice& first, const feature_slice& second) {
410  return (first.weight_index < second.weight_index) ||
411  ((first.weight_index == second.weight_index) && (first.x < second.x));
412  });
413 
414  for (size_t i = 0; i < slice.size(); i++)
415  {
416  values[i] = slice[i].x;
417  indicies[i] = slice[i].weight_index;
418  *space_names[i].get() = slice[i].space_name;
419  }
420  }
421  else
422  {
423  std::vector<feature> slice;
424  slice.reserve(indicies.size());
425 
426  for (size_t i = 0; i < indicies.size(); i++)
427  {
428  slice.push_back({values[i], indicies[i] & parse_mask});
429  }
430  // The comparator should return true if the first element is less than the second.
431  std::sort(slice.begin(), slice.end(), [](const feature& first, const feature& second) {
