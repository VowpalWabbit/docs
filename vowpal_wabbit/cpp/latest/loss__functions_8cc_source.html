18  float getLoss(shared_data* sd, float prediction, float label)
19  {
20  if (prediction <= sd->max_label && prediction >= sd->min_label)
21  {
22  float example_loss = (prediction - label) * (prediction - label);
23  return example_loss;
24  }
25  else if (prediction < sd->min_label)
26  if (label == sd->min_label)
27  return 0.;
28  else
29  return (float)((label - sd->min_label) * (label - sd->min_label) +
30  2. * (label - sd->min_label) * (sd->min_label - prediction));
31  else if (label == sd->max_label)
32  return 0.;
33  else
34  return float((sd->max_label - label) * (sd->max_label - label) +
35  2. * (sd->max_label - label) * (prediction - sd->max_label));
36  }
37 
38  float getUpdate(float prediction, float label, float update_scale, float pred_per_update)
39  {
40  if (update_scale * pred_per_update < 1e-6)
41  {
42  /* When exp(-eta_t)~= 1 we replace 1-exp(-eta_t)
43  * with its first order Taylor expansion around 0
44  * to avoid catastrophic cancellation.
45  */
46  return 2.f * (label - prediction) * update_scale;
47  }
48  return (label - prediction) * (1.f - correctedExp(-2.f * update_scale * pred_per_update)) / pred_per_update;
49  }

51  float getUnsafeUpdate(float prediction, float label, float update_scale)
52  {
53  return 2.f * (label - prediction) * update_scale;
54  }

155 class logloss : public loss_function
156 {
157  public:
158  std::string getType() { return "logistic"; }
159 
160  float getLoss(shared_data*, float prediction, float label)
161  {
162  if (label != -1.f && label != 1.f)
163  std::cout << "You are using label " << label << " not -1 or 1 as loss function expects!" << std::endl;
164  return log(1 + correctedExp(-label * prediction));
165  }
166 
167  float getUpdate(float prediction, float label, float update_scale, float pred_per_update)
168  {
169  float w, x;
170  float d = correctedExp(label * prediction);
171  if (update_scale * pred_per_update < 1e-6)
172  {
173  /* As with squared loss, for small eta_t we replace the update
174  * with its first order Taylor expansion to avoid numerical problems
175  */
176  return label * update_scale / (1 + d);
177  }
178  x = update_scale * pred_per_update + label * prediction + d;
179  w = wexpmx(x);
180  return -(label * w + prediction) / pred_per_update;
181  }
182 
183  float getUnsafeUpdate(float prediction, float label, float update_scale)
184  {
185  float d = correctedExp(label * prediction);
186  return label * update_scale / (1 + d);
187  }
188 
189  inline float wexpmx(float x)
190  {
191  /* This piece of code is approximating W(exp(x))-x.
192  * W is the Lambert W function: W(z)*exp(W(z))=z.
193  * The absolute error of this approximation is less than 9e-5.
194  * Faster/better approximations can be substituted here.
195  */
196  double w = x >= 1. ? 0.86 * x + 0.01 : correctedExp(0.8 * x - 0.65); // initial guess
197  double r = x >= 1. ? x - log(w) - w : 0.2 * x + 0.65 - w; // residual
198  double t = 1. + w;
199  double u = 2. * t * (t + 2. * r / 3.); // magic
200  return (float)(w * (1. + r / t * (u - r) / (u - 2. * r)) - x); // more magic
201  }
202 
203  float getRevertingWeight(shared_data*, float prediction, float eta_t)
204  {
205  float z = -fabs(prediction);
206  return (1 - z - correctedExp(z)) / eta_t;
207  }
208

210  {

236  float getLoss(shared_data*, float prediction, float label)
237  {
238  float e = label - prediction;
239  if (e > 0)
240  return tau * e;
241  else
242  return -(1 - tau) * e;
243  }
244 
245  float getUpdate(float prediction, float label, float update_scale, float pred_per_update)
246  {
247  float err = label - prediction;
248  if (err == 0)
249  return 0;
250  float normal = update_scale * pred_per_update; // base update size
251  if (err > 0)
252  {
253  normal = tau * normal;
254  return (normal < err ? tau * update_scale : err / pred_per_update);
255  }
256  else
257  {
258  normal = -(1 - tau) * normal;
259  return (normal > err ? (tau - 1) * update_scale : err / pred_per_update);
260  }
261  }
262 
263  float getUnsafeUpdate(float prediction, float label, float update_scale)
264  {
265  float err = label - prediction;
266  if (err == 0)
267  return 0;
268  if (err > 0)
269  return tau * update_scale;
270  return -(1 - tau) * update_scale;
271  }

274  {
275  float v, t;
276  t = 0.5f * (sd->min_label + sd->max_label);
277  if (prediction > t)

308  float getLoss(shared_data*, float prediction, float label)
309  {
310  if (label < 0.f)
311  std::cout << "You are using label " << label << " but loss function expects label >= 0!" << std::endl;
312  float exp_prediction = expf(prediction);
313  // deviance is used instead of log-likelihood
314  return 2 * (label * (logf(label + 1e-6f) - prediction) - (label - exp_prediction));
315  }

318  {
319  float exp_prediction = expf(prediction);
320  if (label > 0)

327  return -log1p(exp_prediction * update_scale * pred_per_update) / pred_per_update;

329  }
