390 int random_policy(search_private& priv, bool allow_current, bool allow_optimal, bool advance_prng = true)
391 {
392  if (priv.beta >= 1)
393  {
394  if (allow_current)
395  return (int)priv.current_policy;
396  if (priv.current_policy > 0)
397  return (((int)priv.current_policy) - 1);
398  if (allow_optimal)
399  return -1;
400  std::cerr << "internal error (bug): no valid policies to choose from! defaulting to current" << endl;
401  return (int)priv.current_policy;
402  }
403 
404  int num_valid_policies = (int)priv.current_policy + allow_optimal + allow_current;
405  int pid = -1;
406 
407  if (num_valid_policies == 0)
408  {
409  std::cerr << "internal error (bug): no valid policies to choose from! defaulting to current" << endl;
410  return (int)priv.current_policy;
411  }
412  else if (num_valid_policies == 1)
413  pid = 0;
414  else if (num_valid_policies == 2)
415  pid = (advance_prng ? priv._random_state->get_and_update_random() : priv._random_state->get_random()) >= priv.beta;
416  else
417  {
418  // SPEEDUP this up in the case that beta is small!
419  float r = (advance_prng ? priv._random_state->get_and_update_random() : priv._random_state->get_random());
420  pid = 0;
421 
422  if (r > priv.beta)
423  {
424  r -= priv.beta;
425  while ((r > 0) && (pid < num_valid_policies - 1))
426  {
427  pid++;
428  r -= priv.beta * powf(1.f - priv.beta, (float)pid);
429  }
430  }
431  }
432  // figure out which policy pid refers to
433  if (allow_optimal && (pid == num_valid_policies - 1))
434  return -1; // this is the optimal policy
435 
436  pid = (int)priv.current_policy - pid;
437  if (!allow_current)
438  pid--;
439 
440  return pid;
441 }

1959 void hoopla_permute(size_t* B, size_t* end)
1960 {
1961  // from Curtis IPL 2004, "Darts and hoopla board design"
1962  // first sort
1963  size_t N = end - B;
1964  std::sort(B, end, cmp_size_t);
1965  // make some temporary space
1966  size_t* A = calloc_or_throw<size_t>((N + 1) * 2);
1967  A[N] = B[0]; // arbitrarily choose the maximum in the middle
1968  A[N + 1] = B[N - 1]; // so the maximum goes next to it
1969  size_t lo = N, hi = N + 1; // which parts of A have we filled in? [lo,hi]
1970  size_t i = 0, j = N - 1; // which parts of B have we already covered? [0,i] and [j,N-1]
1971  while (i + 1 < j)
1972  {
1973  // there are four options depending on where things get placed
1974  size_t d1 = absdiff(A[lo], B[i + 1]); // put B[i+1] at the bottom
1975  size_t d2 = absdiff(A[lo], B[j - 1]); // put B[j-1] at the bottom
1976  size_t d3 = absdiff(A[hi], B[i + 1]); // put B[i+1] at the top
1977  size_t d4 = absdiff(A[hi], B[j - 1]); // put B[j-1] at the top
1978  size_t mx = std::max(std::max(d1, d2), std::max(d3, d4));
1979  if (d1 >= mx)
1980  A[--lo] = B[++i];
1981  else if (d2 >= mx)
1982  A[--lo] = B[--j];
1983  else if (d3 >= mx)
1984  A[++hi] = B[++i];
1985  else
1986  A[++hi] = B[--j];
1987  }
1988  // copy it back to B
1989  memcpy(B, A + lo, N * sizeof(size_t));
1990  // clean up
1991  free(A);
1992 }
