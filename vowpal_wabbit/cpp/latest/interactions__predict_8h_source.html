73 template <class R, class S, void (*T)(R&, float, S), bool audit, void (*audit_func)(R&, const audit_strings*), class W>
74 inline void inner_kernel(R& dat, features::iterator_all& begin, features::iterator_all& end, const uint64_t offset,
75  W& weights, feature_value ft_value, feature_index halfhash)
76 {
77  if (audit)
78  {
79  for (; begin != end; ++begin)
80  {
81  audit_func(dat, begin.audit() == nullptr ? &EMPTY_AUDIT_STRINGS : begin.audit()->get());
82  call_T<R, T>(dat, weights, INTERACTION_VALUE(ft_value, begin.value()), (begin.index() ^ halfhash) + offset);
83  audit_func(dat, nullptr);
84  }
85  }
86  else
87  {
88  for (; begin != end; ++begin)
89  call_T<R, T>(dat, weights, INTERACTION_VALUE(ft_value, begin.value()), (begin.index() ^ halfhash) + offset);
90  }
91 }

111  feature_gen_data empty_ns_data; // micro-optimization. don't want to call its constructor each time in loop.
112  empty_ns_data.loop_idx = 0;
113  empty_ns_data.x = 1.;
114  empty_ns_data.loop_end = 0;

117  // loop throw the set of possible interactions
118  for (auto& ns : interactions)
119  { // current list of namespaces to interact.
120 
121 #ifndef GEN_INTER_LOOP
122 
123  // unless GEN_INTER_LOOP is defined we use nested 'for' loops for interactions length 2 (pairs) and 3 (triples)
124  // and generic non-recursive algorythm for all other cases.
125  // nested 'for' loops approach is faster, but can't be used for interation of any length.
126 
127  const size_t len = ns.size();
128 
129  if (len == 2) // special case of pairs
130  {
131  features& first = features_data[(uint8_t)ns[0]];
132  if (first.nonempty())
133  {
134  features& second = features_data[(uint8_t)ns[1]];
135  if (second.nonempty())
136  {
137  const bool same_namespace = (!permutations && (ns[0] == ns[1]));
138 
139  for (size_t i = 0; i < first.indicies.size(); ++i)
140  {
141  feature_index halfhash = FNV_prime * (uint64_t)first.indicies[i];
142  if (audit)
143  {
144  audit_func(dat, i < first.space_names.size() ? first.space_names[i].get() : &EMPTY_AUDIT_STRINGS);
145  }
146  // next index differs for permutations and simple combinations
147  feature_value ft_value = first.values[i];
148  features::features_value_index_audit_range range = second.values_indices_audit();
149  features::iterator_all begin = range.begin();
150  if (same_namespace)

152

154  inner_kernel<R, S, T, audit, audit_func>(dat, begin, end, offset, weights, ft_value, halfhash);
155 
156  if (audit)
157  audit_func(dat, nullptr);
158  } // end for(fst)
159  } // end if (data[snd] size > 0)
160  } // end if (data[fst] size > 0)
161  }
162  else if (len == 3) // special case for triples
163  {
164  features& first = features_data[(uint8_t)ns[0]];
165  if (first.nonempty())
166  {
167  features& second = features_data[(uint8_t)ns[1]];
168  if (second.nonempty())
169  {
170  features& third = features_data[(uint8_t)ns[2]];
171  if (third.nonempty())
172  { // don't compare 1 and 3 as interaction is sorted
173  const bool same_namespace1 = (!permutations && (ns[0] == ns[1]));
174  const bool same_namespace2 = (!permutations && (ns[1] == ns[2]));
175 
176  for (size_t i = 0; i < first.indicies.size(); ++i)
177  {
178  if (audit)
179  {
180  audit_func(dat, i < first.space_names.size() ? first.space_names[i].get() : &EMPTY_AUDIT_STRINGS);
181  }
182  const uint64_t halfhash1 = FNV_prime * (uint64_t)first.indicies[i];
183  const float& first_ft_value = first.values[i];
184  size_t j = 0;
185  if (same_namespace1) // next index differs for permutations and simple combinations
186  j = (PROCESS_SELF_INTERACTIONS(first_ft_value)) ? i : i + 1;
187 
188  for (; j < second.indicies.size(); ++j)
189  { // f3 x k*(f2 x k*f1)
190  if (audit)
191  {
192  audit_func(dat, i < second.space_names.size() ? second.space_names[i].get() : &EMPTY_AUDIT_STRINGS);
193  }
194  feature_index halfhash = FNV_prime * (halfhash1 ^ (uint64_t)second.indicies[j]);
195  feature_value ft_value = INTERACTION_VALUE(first_ft_value, second.values[j]);
196 
197  features::features_value_index_audit_range range = third.values_indices_audit();
198  features::iterator_all begin = range.begin();
199  if (same_namespace2) // next index differs for permutations and simple combinations
200  begin += (PROCESS_SELF_INTERACTIONS(ft_value)) ? j : j + 1;
201 
202  features::iterator_all end = range.end();
203  inner_kernel<R, S, T, audit, audit_func>(dat, begin, end, offset, weights, ft_value, halfhash);
204  if (audit)
205  audit_func(dat, nullptr);
206  } // end for (snd)
207  if (audit)
208  audit_func(dat, nullptr);
209  } // end for (fst)
210 
211  } // end if (data[thr] size > 0)
212  } // end if (data[snd] size > 0)
213  } // end if (data[fst] size > 0)
214  }
215  else // generic case: quatriples, etc.
216 
217 #endif
218  {
219  bool must_skip_interaction = false;
220  // preparing state data
221  feature_gen_data* fgd = state_data.begin();
222  feature_gen_data* fgd2; // for further use
223  for (namespace_index n : ns)
224  {
225  features& ft = features_data[(int32_t)n];
226  const size_t ft_cnt = ft.indicies.size();
227 
228  if (ft_cnt == 0)
229  {
230  must_skip_interaction = true;
231  break;
232  }
233 
234  if (fgd == state_data.end())
235  {
236  state_data.push_back(empty_ns_data);
237  fgd = state_data.end() - 1; // reassign as memory could be realloced
238  }
239 
240  fgd->loop_end = ft_cnt - 1; // saving number of features for each namespace
241  fgd->ft_arr = &ft;
242  ++fgd;
243  }
244 
245  // if any of interacting namespace has 0 features - whole interaction is skipped
246  if (must_skip_interaction)
247  continue; // no_data_to_interact
248 
249  if (!permutations) // adjust state_data for simple combinations
250  { // if permutations mode is disabeled then namespaces in ns are already sorted and thus grouped
251  // (in fact, currently they are sorted even for enabled permutations mode)
252  // let's go throw the list and calculate number of features to skip in namespaces which
253  // repeated more than once to generate only simple combinations of features
254 
255  size_t margin = 0; // number of features to ignore if namespace has been seen before
256 
257  // iterate list backward as margin grows in this order
258 
259  for (fgd = state_data.end() - 1; fgd > state_data.begin(); --fgd)
260  {
261  fgd2 = fgd - 1;
262  fgd->self_interaction = (fgd->ft_arr == fgd2->ft_arr); // state_data.begin().self_interaction is always false
263  if (fgd->self_interaction)
264  {
265  size_t& loop_end = fgd2->loop_end;
266 
267  if (!PROCESS_SELF_INTERACTIONS((*fgd2->ft_arr).values[loop_end - margin]))
268  {
269  ++margin; // otherwise margin can't be increased
270  must_skip_interaction = loop_end < margin;
271  if (must_skip_interaction)
272  break;
273  }
274 
275  if (margin != 0)
276  loop_end -= margin; // skip some features and increase margin
277  }
278  else if (margin != 0)
279  margin = 0;
280  }
281 
282  // if impossible_without_permutations == true then we faced with case like interaction 'aaaa'
283  // where namespace 'a' contains less than 4 unique features. It's impossible to make simple
284  // combination of length 4 without repetitions from 3 or less elements.
285  if (must_skip_interaction)
286  continue; // impossible_without_permutations
287  } // end of state_data adjustment
288

290  fgd2 = state_data.end() - 1; // always equal to last ns
291  fgd->loop_idx = 0; // loop_idx contains current feature id for curently processed namespace.
292 
293  // beware: micro-optimization.
294  /* start & end are always point to features in last namespace of interaction.
295  for 'all.permutations == true' they are constant.*/
296  size_t start_i = 0;
297 
298  feature_gen_data* cur_data = fgd;
299  // end of micro-optimization block
300 
301  // generic feature generation cycle for interactions of any length
302  bool do_it = true;
303  while (do_it)
304  {
305  if (cur_data < fgd2) // can go further threw the list of namespaces in interaction
306  {
307  feature_gen_data* next_data = cur_data + 1;
308  size_t feature = cur_data->loop_idx;
309  features& fs = *(cur_data->ft_arr);
310 
311  if (next_data->self_interaction)
312  { // if next namespace is same, we should start with loop_idx + 1 to avoid feature interaction with itself
313  // unless feature has value x and x != x*x. E.g. x != 0 and x != 1. Features with x == 0 are already
314  // filtered out in parce_args.cc::maybeFeature().
315 
316  next_data->loop_idx =
317  (PROCESS_SELF_INTERACTIONS(fs.values[feature])) ? cur_data->loop_idx : cur_data->loop_idx + 1;
318  }
319  else
320  next_data->loop_idx = 0;
321 
322  if (audit)
323  audit_func(dat, fs.space_names[feature].get());
324 
325  if (cur_data == fgd) // first namespace
326  {
327  next_data->hash = FNV_prime * (uint64_t)fs.indicies[feature];
328  next_data->x = fs.values[feature]; // data->x == 1.
329  }
330  else
331  { // feature2 xor (16777619*feature1)
332  next_data->hash = FNV_prime * (cur_data->hash ^ (uint64_t)fs.indicies[feature]);
333  next_data->x = INTERACTION_VALUE(fs.values[feature], cur_data->x);
334  }
335 
336  ++cur_data;
337  }
338  else
339  { // last namespace - iterate its features and go back
340  if (!permutations) // start value is not a constant in this case
341  start_i = fgd2->loop_idx;
342 
343  features& fs = *(fgd2->ft_arr);
344 
345  feature_value ft_value = fgd2->x;
346  feature_index halfhash = fgd2->hash;
347 
348  features::features_value_index_audit_range range = fs.values_indices_audit();
349  features::iterator_all begin = range.begin();
350  begin += start_i;
351  features::iterator_all end = range.begin();
352  end += fgd2->loop_end + 1;
353  inner_kernel<R, S, T, audit, audit_func, W>(dat, begin, end, offset, weights, ft_value, halfhash);
354 
355  // trying to go back increasing loop_idx of each namespace by the way
356 
357  bool go_further = true;
358 
359  do
360  {
361  --cur_data;
362  go_further = (++cur_data->loop_idx > cur_data->loop_end); // increment loop_idx
363  if (audit)
364  audit_func(dat, nullptr);
365  } while (go_further && cur_data != fgd);
366 
367  do_it = !(cur_data == fgd && go_further);
368  // if do_it==false - we've reached 0 namespace but its 'cur_data.loop_idx > cur_data.loop_end' -> exit the
369  // while loop
370  } // if last namespace
371  } // while do_it
372  }
373  } // foreach interaction in all.interactions
374 
375  state_data.delete_v();
