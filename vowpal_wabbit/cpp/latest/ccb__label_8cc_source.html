226 {

241  }

247 CCB::conditional_contextual_bandit_outcome* parse_outcome(VW::string_view& outcome)
248 {
249  auto& ccb_outcome = *(new CCB::conditional_contextual_bandit_outcome());
250 
251  auto split_commas = v_init<VW::string_view>();
252  tokenize(',', outcome, split_commas);
253 
254  auto split_colons = v_init<VW::string_view>();
255  tokenize(':', split_commas[0], split_colons);
256 
257  if (split_colons.size() != 3)
258  THROW("Malformed ccb label");
259 
260  ccb_outcome.probabilities = v_init<ACTION_SCORE::action_score>();
261  ccb_outcome.probabilities.push_back(convert_to_score(split_colons[0], split_colons[2]));
262 
263  ccb_outcome.cost = float_of_string(split_colons[1]);
264  if (std::isnan(ccb_outcome.cost))
265  THROW("error NaN cost: " << split_colons[1]);
266 
267  split_colons.clear();
268 
269  for (size_t i = 1; i < split_commas.size(); i++)
270  {
271  tokenize(':', split_commas[i], split_colons);
272  if (split_colons.size() != 2)
273  THROW("Must be action probability pairs");
274  ccb_outcome.probabilities.push_back(convert_to_score(split_colons[0], split_colons[1]));
275  }
276 
277  split_colons.delete_v();
278  split_commas.delete_v();
279 
280  return &ccb_outcome;
281 }

291 void parse_label(parser* p, shared_data*, void* v, v_array<VW::string_view>& words)
292 {
293  CCB::label* ld = static_cast<CCB::label*>(v);
294  ld->weight = 1.0;
295 
296  if (words.size() < 2)
297  THROW("ccb labels may not be empty");
298  if (!(words[0] == CCB_LABEL))
299  {
300  THROW("ccb labels require the first word to be ccb");
301  }
302 
303  auto type = words[1];
304  if (type == SHARED_TYPE)
305  {
306  if (words.size() > 2)
307  THROW("shared labels may not have a cost");
308  ld->type = CCB::example_type::shared;
309  }
310  else if (type == ACTION_TYPE)
311  {
312  if (words.size() > 2)
313  THROW("action labels may not have a cost");
314  ld->type = CCB::example_type::action;
315  }
316  else if (type == SLOT_TYPE)
317  {
318  if (words.size() > 4)
319  THROW("ccb slot label can only have a type cost and exclude list");
320  ld->type = CCB::example_type::slot;
321 
322  // Skip the first two words "ccb <type>"
323  for (size_t i = 2; i < words.size(); i++)
324  {
325  auto is_outcome = words[i].find(':');
326  if (is_outcome != VW::string_view::npos)
327  {
328  if (ld->outcome != nullptr)
329  {
330  THROW("There may be only 1 outcome associated with a slot.")
331  }
332 
333  ld->outcome = parse_outcome(words[i]);
334  }
335  else
336  {
337  tokenize(',', words[i], p->parse_name);
338  parse_explicit_inclusions(ld, p->parse_name);
339  }
340  }
341 
342  // If a full distribution has been given, check if it sums to 1, otherwise throw.
343  if (ld->outcome && ld->outcome->probabilities.size() > 1)
344  {
345  float total_pred = std::accumulate(ld->outcome->probabilities.begin(), ld->outcome->probabilities.end(), 0.f,
346  [](float result_so_far, ACTION_SCORE::action_score action_pred) {
347  return result_so_far + action_pred.score;
348  });
349 
350  // TODO do a proper comparison here.
351  if (!VW::math::are_same(total_pred, 1.f))
352  {
353  THROW("When providing all prediction probabilities they must add up to 1.f, instead summed to " << total_pred);
354  }
355  }
356  }
357  else
358  {
359  THROW("unknown label type: " << type);
360  }
361 }
