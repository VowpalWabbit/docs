15 inline char* run_len_decode(char* p, uint64_t& i)
16 {
17  // read an int 7 bits at a time.
18  size_t count = 0;
19  while (*p & 128) i = i | ((uint64_t)(*(p++) & 127) << 7 * count++);
20  i = i | ((uint64_t)(*(p++)) << 7 * count);
21  return p;
22 }

24 inline char* run_len_encode(char* p, uint64_t i)
25 {
26  // store an int 7 bits at a time.
