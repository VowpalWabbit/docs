134 void cb_adf::learn_SM(multi_learner& base, multi_ex& examples)
135 {
136  gen_cs_test_example(examples, _cs_labels); // create test labels.
137  call_cs_ldf<false>(base, examples, _cb_labels, _cs_labels, _prepped_cs_labels, _offset);
138 
139  // Can probably do this more efficiently than 6 loops over the examples...
140  //[1: initialize temporary storage;
141  // 2: find chosen action;
142  // 3: create cs_labels (gen_cs_example_sm);
143  // 4: get probability of chosen action;
144  // 5: backup example wts;
145  // 6: restore example wts]
146  _a_s.clear();
147  _prob_s.clear();
148  // TODO: Check that predicted scores are always stored with the first example
149  for (uint32_t i = 0; i < examples[0]->pred.a_s.size(); i++)
150  {
151  _a_s.push_back({examples[0]->pred.a_s[i].action, examples[0]->pred.a_s[i].score});
152  _prob_s.push_back({examples[0]->pred.a_s[i].action, 0.0});
153  }
154 
155  float sign_offset = 1.0; // To account for negative rewards/costs
156  uint32_t chosen_action = 0;
157  float example_weight = 1.0;
158 
159  for (uint32_t i = 0; i < examples.size(); i++)
160  {
161  CB::label ld = examples[i]->l.cb;
162  if (ld.costs.size() == 1 && ld.costs[0].cost != FLT_MAX)
163  {
164  chosen_action = i;
165  example_weight = ld.costs[0].cost / safe_probability(ld.costs[0].probability);
166 
167  // Importance weights of examples cannot be negative.
168  // So we use a trick: set |w| as weight, and use sign(w) as an offset in the regression target.
169  if (ld.costs[0].cost < 0.0)
170  {
171  sign_offset = -1.0;
172  example_weight = -example_weight;
173  }
174  break;
175  }
176  }
177 
178  gen_cs_example_sm(examples, chosen_action, sign_offset, _a_s, _cs_labels);
179 
180  // Lambda is -1 in the call to generate_softmax because in vw, lower score is better; for softmax higher score is
181  // better.
182  generate_softmax(-1.0, begin_scores(_a_s), end_scores(_a_s), begin_scores(_prob_s), end_scores(_prob_s));
183 
184  // TODO: Check Marco's example that causes VW to report prob > 1.
185 
186  for (auto const& action_score : _prob_s) // Scale example_wt by prob of chosen action
187  {
188  if (action_score.action == chosen_action)
189  {
190  example_weight *= action_score.score;
191  break;
192  }
193  }
194 
195  _backup_weights.clear();
196  _backup_nf.clear();
197  for (auto const& action_score : _prob_s)
198  {
199  uint32_t current_action = action_score.action;
200  _backup_weights.push_back(examples[current_action]->weight);
201  _backup_nf.push_back((uint32_t)examples[current_action]->num_features);
202 
203  if (current_action == chosen_action)
204  examples[current_action]->weight *= example_weight * (1.0f - action_score.score);
205  else
206  examples[current_action]->weight *= example_weight * action_score.score;
207 
208  if (examples[current_action]->weight <= 1e-15)
209  examples[current_action]->weight = 0;
210  }
211 
212  // Do actual training
213  call_cs_ldf<true>(base, examples, _cb_labels, _cs_labels, _prepped_cs_labels, _offset);
214 
215  // Restore example weights and numFeatures
216  for (size_t i = 0; i < _prob_s.size(); i++)
217  {
218  uint32_t current_action = _prob_s[i].action;
219  examples[current_action]->weight = _backup_weights[i];
220  examples[current_action]->num_features = _backup_nf[i];
221  }
222 }

236 template <bool predict>
237 void cb_adf::learn_MTR(multi_learner& base, multi_ex& examples)
238 {
239  // uint32_t action = 0;
240  if (predict) // first get the prediction to return
241  {
242  gen_cs_example_ips(examples, _cs_labels);
243  call_cs_ldf<false>(base, examples, _cb_labels, _cs_labels, _prepped_cs_labels, _offset);
244  std::swap(examples[0]->pred.a_s, _a_s);
245  }
246  // second train on _one_ action (which requires up to 3 examples).
247  // We must go through the cost sensitive classifier layer to get
248  // proper feature handling.
249  gen_cs_example_mtr(_gen_cs, examples, _cs_labels);
250  uint32_t nf = (uint32_t)examples[_gen_cs.mtr_example]->num_features;
251  float old_weight = examples[_gen_cs.mtr_example]->weight;
252  const float clipped_p = std::max(examples[_gen_cs.mtr_example]->l.cb.costs[0].probability, _clip_p);
253  examples[_gen_cs.mtr_example]->weight *= 1.f / clipped_p * ((float)_gen_cs.event_sum / (float)_gen_cs.action_sum);
254 
255  std::swap(_gen_cs.mtr_ec_seq[0]->pred.a_s, _a_s_mtr_cs);
256  // TODO!!! cb_labels are not getting properly restored (empty costs are dropped)
257  GEN_CS::call_cs_ldf<true>(base, _gen_cs.mtr_ec_seq, _cb_labels, _cs_labels, _prepped_cs_labels, _offset);
258  examples[_gen_cs.mtr_example]->num_features = nf;
259  examples[_gen_cs.mtr_example]->weight = old_weight;
260  std::swap(_gen_cs.mtr_ec_seq[0]->pred.a_s, _a_s_mtr_cs);
261  std::swap(examples[0]->pred.a_s, _a_s);
262 }

265 example* test_adf_sequence(multi_ex& ec_seq)
266 {
267  if (ec_seq.empty())
268  THROW("cb_adf: At least one action must be provided for an example to be valid.");
269 
270  uint32_t count = 0;
271  example* ret = nullptr;
272  for (auto* ec : ec_seq)
273  {
274  // Check if there is more than one cost for this example.
275  if (ec->l.cb.costs.size() > 1)
276  THROW("cb_adf: badly formatted example, only one cost can be known.");
277 
278  // Check whether the cost was initialized to a value.
279  if (ec->l.cb.costs.size() == 1 && ec->l.cb.costs[0].cost != FLT_MAX)
280  {
281  ret = ec;
282  count += 1;
283  if (count > 1)
284  THROW("cb_adf: badly formatted example, only one line can have a cost");
285  }
286  }
287 
288  return ret;
