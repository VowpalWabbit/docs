41 void bs_predict_vote(example& ec, std::vector<double>& pred_vec)
42 {
43  // majority vote in linear time
44  unsigned int counter = 0;
45  int current_label = 1, init_label = 1;
46  // float sum_labels = 0; // uncomment for: "avg on votes" and getLoss()
47  bool majority_found = false;
48  bool multivote_detected = false; // distinct(votes)>2: used to skip part of the algorithm
49  auto pred_vec_sz = pred_vec.size();
50  int* pred_vec_int = new int[pred_vec_sz];
51 
52  for (unsigned int i = 0; i < pred_vec_sz; i++)
53  {
54  pred_vec_int[i] = (int)floor(
55  pred_vec[i] + 0.5); // could be added: link(), min_label/max_label, cutoff between true/false for binary
56 
57  if (!multivote_detected) // distinct(votes)>2 detection bloc
58  {
59  if (i == 0)
60  {
61  init_label = pred_vec_int[i];
62  current_label = pred_vec_int[i];
63  }
64  else if (init_label != current_label && pred_vec_int[i] != current_label && pred_vec_int[i] != init_label)
65  multivote_detected = true; // more than 2 distinct votes detected
66  }
67 
68  if (counter == 0)
69  {
70  counter = 1;
71  current_label = pred_vec_int[i];
72  }
73  else
74  {
75  if (pred_vec_int[i] == current_label)
76  counter++;
77  else
78  {
79  counter--;
80  }
81  }
82  }
83 
84  if (counter > 0 && multivote_detected) // remove this condition for: "avg on votes" and getLoss()
85  {
86  counter = 0;
87  for (unsigned int i = 0; i < pred_vec.size(); i++)
88  if (pred_vec_int[i] == current_label)
89  {
90  counter++;
91  // sum_labels += pred_vec[i]; // uncomment for: "avg on votes" and getLoss()
92  }
93  if (counter * 2 > pred_vec.size())
94  majority_found = true;
95  }
96 
97  if (multivote_detected && !majority_found) // then find most frequent element - if tie: smallest tie label
98  {
99  std::sort(pred_vec_int, pred_vec_int + pred_vec.size());
100  int tmp_label = pred_vec_int[0];
101  counter = 1;
102  for (unsigned int i = 1, temp_count = 1; i < pred_vec.size(); i++)
103  {
104  if (tmp_label == pred_vec_int[i])
105  temp_count++;
106  else
107  {
108  if (temp_count > counter)
109  {
110  current_label = tmp_label;
111  counter = temp_count;
112  }
113  tmp_label = pred_vec_int[i];
114  temp_count = 1;
115  }
116  }
117  /* uncomment for: "avg on votes" and getLoss()
118  sum_labels = 0;
119  for(unsigned int i=0; i<pred_vec.size(); i++)
120  if(pred_vec_int[i] == current_label)
121  sum_labels += pred_vec[i]; */
122  }
123  // TODO: unique_ptr would also handle exception case
124  delete[] pred_vec_int;
125 
126  // ld.prediction = sum_labels/(float)counter; //replace line below for: "avg on votes" and getLoss()
127  ec.pred.scalar = (float)current_label;
128 
129  // ec.loss = all.loss->getLoss(all.sd, ld.prediction, ld.label) * ec.weight; //replace line below for: "avg on votes"
130  // and getLoss()

132 }

176 void predict_or_learn(bs& d, single_learner& base, example& ec)
177 {
178  vw& all = *d.all;
179  bool shouldOutput = all.raw_prediction > 0;
180 
181  float weight_temp = ec.weight;
182 
183  std::stringstream outputStringStream;
184  d.pred_vec->clear();
185 
186  for (size_t i = 1; i <= d.B; i++)
187  {
188  ec.weight = weight_temp * (float)BS::weight_gen(d._random_state);
189 
190  if (is_learn)
191  base.learn(ec, i - 1);
192  else
193  base.predict(ec, i - 1);
194 
195  d.pred_vec->push_back(ec.pred.scalar);
196 
197  if (shouldOutput)
198  {
199  if (i > 1)
200  outputStringStream << ' ';
201  outputStringStream << i << ':' << ec.partial_prediction;
202  }
203  }
204 
205  ec.weight = weight_temp;
206 
207  switch (d.bs_type)
208  {
209  case BS_TYPE_MEAN:
210  bs_predict_mean(all, ec, *d.pred_vec);
211  break;
212  case BS_TYPE_VOTE:
213  bs_predict_vote(ec, *d.pred_vec);
214  break;
215  default:
216  THROW("Unknown bs_type specified: " << d.bs_type);
217  }
218 
219  if (shouldOutput)
220  all.print_text_by_ref(all.raw_prediction, outputStringStream.str(), ec.tag);
221 }
