70 float cb_explore_adf_regcb::binary_search(float fhat, float delta, float sens, float tol)
71 {
72  const float maxw = (std::min)(fhat / sens, FLT_MAX);
73 
74  if (maxw * fhat * fhat <= delta)
75  return maxw;
76 
77  float l = 0;
78  float u = maxw;
79  float w, v;
80 
81  for (int iter = 0; iter < B_SEARCH_MAX_ITER; iter++)
82  {
83  w = (u + l) / 2.f;
84  v = w * (fhat * fhat - (fhat - sens * w) * (fhat - sens * w)) - delta;
85  if (v > 0)
86  u = w;
87  else
88  l = w;
89  if (fabs(v) <= tol || u - l <= tol)
90  break;
91  }
92 
93  return l;
94 }

96 void cb_explore_adf_regcb::get_cost_ranges(float delta, LEARNER::multi_learner& base, multi_ex& examples, bool min_only)
97 {
98  const size_t num_actions = examples[0]->pred.a_s.size();
99  _min_costs.resize(num_actions);
100  _max_costs.resize(num_actions);
101 
102  _ex_as.clear();
103  _ex_costs.clear();
104 
105  // backup cb example data
106  for (const auto& ex : examples)
107  {
108  _ex_as.push_back(ex->pred.a_s);
109  _ex_costs.push_back(ex->l.cb.costs);
110  }
111 
112  // set regressor predictions
113  for (const auto& as : _ex_as[0])
114  {
115  examples[as.action]->pred.scalar = as.score;
116  }
117 
118  const float cmin = _min_cb_cost;
119  const float cmax = _max_cb_cost;
120 
121  for (size_t a = 0; a < num_actions; ++a)
122  {
123  example* ec = examples[a];
124  ec->l.simple.label = cmin - 1;
125  float sens = base.sensitivity(*ec);
126  float w = 0; // importance weight
127 
128  if (ec->pred.scalar < cmin || std::isnan(sens) || std::isinf(sens))
129  _min_costs[a] = cmin;
130  else
131  {
132  w = binary_search(ec->pred.scalar - cmin + 1, delta, sens);
133  _min_costs[a] = (std::max)(ec->pred.scalar - sens * w, cmin);
134  if (_min_costs[a] > cmax)
135  _min_costs[a] = cmax;
136  }
137 
138  if (!min_only)
139  {
140  ec->l.simple.label = cmax + 1;
141  sens = base.sensitivity(*ec);
142  if (ec->pred.scalar > cmax || std::isnan(sens) || std::isinf(sens))
143  {
144  _max_costs[a] = cmax;
145  }
146  else
147  {
148  w = binary_search(cmax + 1 - ec->pred.scalar, delta, sens);
149  _max_costs[a] = (std::min)(ec->pred.scalar + sens * w, cmax);
150  if (_max_costs[a] < cmin)
151  _max_costs[a] = cmin;
152  }
153  }
154  }
155 
156  // reset cb example data
157  for (size_t i = 0; i < examples.size(); ++i)
158  {
159  examples[i]->pred.a_s = _ex_as[i];
160  examples[i]->l.cb.costs = _ex_costs[i];
161  }
162 }

164 template <bool is_learn>
165 void cb_explore_adf_regcb::predict_or_learn_impl(LEARNER::multi_learner& base, multi_ex& examples)
166 {
167  if (is_learn)
168  {
169  for (size_t i = 0; i < examples.size() - 1; ++i)
170  {
171  CB::label& ld = examples[i]->l.cb;
172  if (ld.costs.size() == 1)
173  ld.costs[0].probability = 1.f; // no importance weighting
174  }
175 
176  LEARNER::multiline_learn_or_predict<true>(base, examples, examples[0]->ft_offset);
177  ++_counter;
178  }
179  else
180  LEARNER::multiline_learn_or_predict<false>(base, examples, examples[0]->ft_offset);
181 
182  v_array<ACTION_SCORE::action_score>& preds = examples[0]->pred.a_s;
183  uint32_t num_actions = (uint32_t)preds.size();
184 
185  const float max_range = _max_cb_cost - _min_cb_cost;
186  // threshold on empirical loss difference
187  const float delta = _c0 * log((float)(num_actions * _counter)) * pow(max_range, 2);
188 
189  if (!is_learn)
190  {
191  get_cost_ranges(delta, base, examples, /*min_only=*/_regcbopt);
192 
193  if (_regcbopt) // optimistic variant
194  {
195  float min_cost = FLT_MAX;
196  size_t a_opt = 0; // optimistic action
197  for (size_t a = 0; a < num_actions; ++a)
198  {
199  if (_min_costs[a] < min_cost)
200  {
201  min_cost = _min_costs[a];
202  a_opt = a;
203  }
204  }
205  for (size_t i = 0; i < preds.size(); ++i)
206  {
207  if (preds[i].action == a_opt || (!_first_only && _min_costs[preds[i].action] == min_cost))
208  preds[i].score = 1;
209  else
210  preds[i].score = 0;
211  }
212  }
213  else // elimination variant
214  {
215  float min_max_cost = FLT_MAX;
216  for (size_t a = 0; a < num_actions; ++a)
217  if (_max_costs[a] < min_max_cost)
218  min_max_cost = _max_costs[a];
219  for (size_t i = 0; i < preds.size(); ++i)
220  {
221  if (_min_costs[preds[i].action] <= min_max_cost)
222  preds[i].score = 1;
223  else
224  preds[i].score = 0;
225  // explore uniformly on support
226  exploration::enforce_minimum_probability(
227  1.0, /*update_zero_elements=*/false, begin_scores(preds), end_scores(preds));
228  }
229  }
230  }
