32 bool know_all_cost_example(CB::label& ld)
33 {
34  if (ld.costs.size() <= 1) // this means we specified an example where all actions are possible but only specified the
35  // cost for the observed action
36  return false;
37 
38  // if we specified more than 1 action for this example, i.e. either we have a limited set of possible actions, or all
39  // actions are specified than check if all actions have a specified cost
40  for (auto& cl : ld.costs)
41  if (cl.cost == FLT_MAX)
42  return false;
43 
44  return true;
45 }

48 void predict_or_learn(cb& data, single_learner& base, example& ec)
49 {
50  CB::label ld = ec.l.cb;
51  cb_to_cs& c = data.cbcs;
52  c.known_cost = get_observed_cost(ld);
53  if (c.known_cost != nullptr && (c.known_cost->action < 1 || c.known_cost->action > c.num_actions))
54  std::cerr << "invalid action: " << c.known_cost->action << std::endl;
55 
56  // generate a cost-sensitive example to update classifiers
57  gen_cs_example<is_learn>(c, ec, ld, data.cb_cs_ld);
58 
59  if (c.cb_type != CB_TYPE_DM)
60  {
61  ec.l.cs = data.cb_cs_ld;
62 
63  if (is_learn)
64  base.learn(ec);
65  else
66  base.predict(ec);
67 
68  for (size_t i = 0; i < ld.costs.size(); i++)
69  ld.costs[i].partial_prediction = data.cb_cs_ld.costs[i].partial_prediction;
70  ec.l.cb = ld;
71  }

177
