73 void parse_label(parser*, shared_data* sd, void* v, v_array<VW::string_view>& words)
74 {
75  label_t* ld = (label_t*)v;
76 
77  switch (words.size())
78  {
79  case 0:
80  break;
81  case 1:
82  ld->label = sd->ldict ? (uint32_t)sd->ldict->get(words[0]) : int_of_string(words[0]);
83  ld->weight = 1.0;
84  break;
85  case 2:
86  ld->label = sd->ldict ? (uint32_t)sd->ldict->get(words[0]) : int_of_string(words[0]);
87  ld->weight = float_of_string(words[1]);
88  break;
89  default:
90  std::cerr << "malformed example!\n";
91  std::cerr << "words.size() = " << words.size() << std::endl;
92  }
93  if (ld->label == 0)
94  THROW("label 0 is not allowed for multiclass. Valid labels are {1,k}"
95  << (sd->ldict ? "\nthis likely happened because you specified an invalid label with named labels" : ""));
96 }
