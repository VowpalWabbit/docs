9 float collision_cleanup(features& fs)
10 {
11  uint64_t last_index = (uint64_t)-1;
12  float sum_sq = 0.f;
13  features::iterator pos = fs.begin();
14  for (features::iterator& f : fs)
15  {
16  if (last_index == f.index())
17  pos.value() += f.value();
18  else
19  {
20  sum_sq += pos.value() * pos.value();
21  ++pos;
22  pos.value() = f.value();
23  pos.index() = f.index();
24  last_index = f.index();
25  }
26  }
27 
28  sum_sq += pos.value() * pos.value();
29  fs.sum_feat_sq = sum_sq;
30  ++pos;
31  fs.truncate_to(pos);
32 
33  return sum_sq;
34 }

89 void move_feature_namespace(example* dst, example* src, namespace_index c)
90 {
91  if (std::find(src->indices.begin(), src->indices.end(), c) == src->indices.end())
92  return; // index not present in src
