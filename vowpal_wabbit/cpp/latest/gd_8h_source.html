36 inline void vec_add_multipredict(multipredict_info<T>& mp, const float fx, uint64_t fi)
37 {
38  if ((-1e-10 < fx) && (fx < 1e-10))
39  return;
40  uint64_t mask = mp.weights.mask();
41  polyprediction* p = mp.pred;
42  fi &= mask;
43  uint64_t top = fi + (uint64_t)((mp.count - 1) * mp.step);
44  uint64_t i = 0;
45  if (top <= mask)
46  {
47  i += fi;
48  for (; i <= top; i += mp.step, ++p)
49  p->scalar +=
50  fx * mp.weights[i]; // TODO: figure out how to use weight_parameters::iterator (not using change_begin())
51  }
52  else // TODO: this could be faster by unrolling into two loops
53  for (size_t c = 0; c < mp.count; ++c, fi += (uint64_t)mp.step, ++p)
54  {
55  fi &= mask;
56  p->scalar += fx * mp.weights[fi];
57  }
58 }
