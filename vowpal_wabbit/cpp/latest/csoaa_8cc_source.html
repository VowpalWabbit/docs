53 template <bool is_learn>
54 void predict_or_learn(csoaa& c, single_learner& base, example& ec)
55 {
56  // std::cerr << "------------- passthrough" << std::endl;
57  COST_SENSITIVE::label ld = ec.l.cs;
58  uint32_t prediction = 1;
59  float score = FLT_MAX;
60  size_t pt_start = ec.passthrough ? ec.passthrough->size() : 0;
61  ec.l.simple = {0., 0., 0.};
62  if (!ld.costs.empty())
63  {
64  for (auto& cl : ld.costs)
65  inner_loop<is_learn>(base, ec, cl.class_index, cl.x, prediction, score, cl.partial_prediction);
66  ec.partial_prediction = score;
67  }
68  else if (DO_MULTIPREDICT && !is_learn)
69  {
70  ec.l.simple = {FLT_MAX, 0.f, 0.f};
71  base.multipredict(ec, 0, c.num_classes, c.pred, false);
72  for (uint32_t i = 1; i <= c.num_classes; i++)
73  {
74  add_passthrough_feature(ec, i, c.pred[i - 1].scalar);
75  if (c.pred[i - 1].scalar < c.pred[prediction - 1].scalar)
76  prediction = i;
77  }
78  ec.partial_prediction = c.pred[prediction - 1].scalar;
79  }
80  else
81  {
82  float temp;
83  for (uint32_t i = 1; i <= c.num_classes; i++) inner_loop<false>(base, ec, i, FLT_MAX, prediction, score, temp);
84  }
85  if (ec.passthrough)
86  {
87  uint64_t second_best = 0;
88  float second_best_cost = FLT_MAX;
89  for (size_t i = 0; i < ec.passthrough->size() - pt_start; i++)
90  {
91  float val = ec.passthrough->values[pt_start + i];
92  if ((val > ec.partial_prediction) && (val < second_best_cost))
93  {
94  second_best_cost = val;
95  second_best = ec.passthrough->indicies[pt_start + i];
96  }
97  }
98  if (second_best_cost < FLT_MAX)
99  {
100  float margin = second_best_cost - ec.partial_prediction;
101  add_passthrough_feature(ec, constant * 2, margin);
102  add_passthrough_feature(ec, constant * 2 + 1 + second_best, 1.);
103  }
104  else
105  add_passthrough_feature(ec, constant * 3, 1.);
106  }
107 
108  ec.pred.multiclass = prediction;
109  ec.l.cs = ld;
110 }

288 void do_actual_learning_wap(ldf& data, single_learner& base, multi_ex& ec_seq)
289 {
290  size_t K = ec_seq.size();
291  std::vector<COST_SENSITIVE::wclass*> all_costs;
292  for (const auto& example : ec_seq) all_costs.push_back(&example->l.cs.costs[0]);
293  compute_wap_values(all_costs);
294 
295  for (size_t k1 = 0; k1 < K; k1++)
296  {
297  example* ec1 = ec_seq[k1];
298 
299  // save original variables
300  COST_SENSITIVE::label save_cs_label = ec1->l.cs;
301  label_data& simple_label = ec1->l.simple;
302 
303  v_array<COST_SENSITIVE::wclass> costs1 = save_cs_label.costs;
304  if (costs1[0].class_index == (uint32_t)-1)
305  continue;
306 
307  LabelDict::add_example_namespace_from_memory(data.label_features, *ec1, costs1[0].class_index);
308 
309  for (size_t k2 = k1 + 1; k2 < K; k2++)
310  {
311  example* ec2 = ec_seq[k2];
312  v_array<COST_SENSITIVE::wclass> costs2 = ec2->l.cs.costs;
313 
314  if (costs2[0].class_index == (uint32_t)-1)
315  continue;
316  float value_diff = fabs(costs2[0].wap_value - costs1[0].wap_value);
317  // float value_diff = fabs(costs2[0].x - costs1[0].x);
318  if (value_diff < 1e-6)
319  continue;
320 
321  LabelDict::add_example_namespace_from_memory(data.label_features, *ec2, costs2[0].class_index);
322 
323  // learn
324  simple_label.initial = 0.;
325  simple_label.label = (costs1[0].x < costs2[0].x) ? -1.0f : 1.0f;
326  float old_weight = ec1->weight;
327  ec1->weight = value_diff;
328  ec1->partial_prediction = 0.;
329  subtract_example(*data.all, ec1, ec2);
330  uint64_t old_offset = ec1->ft_offset;
331  ec1->ft_offset = data.ft_offset;
332  base.learn(*ec1);
333  ec1->ft_offset = old_offset;
334  ec1->weight = old_weight;
335  unsubtract_example(ec1);
336 
337  LabelDict::del_example_namespace_from_memory(data.label_features, *ec2, costs2[0].class_index);
338  }
339  LabelDict::del_example_namespace_from_memory(data.label_features, *ec1, costs1[0].class_index);
340 
341  // restore original cost-sensitive label, sum of importance weights
342  ec1->l.cs = save_cs_label;
343  // TODO: What about partial_prediction? See do_actual_learning_oaa.
344  }

347 void do_actual_learning_oaa(ldf& data, single_learner& base, multi_ex& ec_seq)
348 {
349  float min_cost = FLT_MAX;
350  float max_cost = -FLT_MAX;
351 
352  for (const auto& example : ec_seq)
353  {
354  float ec_cost = example->l.cs.costs[0].x;
355  if (ec_cost < min_cost)
356  min_cost = ec_cost;
357  if (ec_cost > max_cost)
358  max_cost = ec_cost;
359  }
360 
361  for (const auto& ec : ec_seq)
362  {
363  // save original variables
364  label save_cs_label = ec->l.cs;
365  const auto& costs = save_cs_label.costs;
366 
367  // build example for the base learner
368  label_data simple_label;
369 
370  simple_label.initial = 0.;
371  float old_weight = ec->weight;
372  if (!data.treat_as_classifier) // treat like regression
373  simple_label.label = costs[0].x;
374  else // treat like classification
375  {
376  if (costs[0].x <= min_cost)
377  {
378  simple_label.label = -1.;
379  ec->weight = old_weight * (max_cost - min_cost);
380  }
381  else
382  {
383  simple_label.label = 1.;
384  ec->weight = old_weight * (costs[0].x - min_cost);
385  }
386  }
387  ec->l.simple = simple_label;
388 
389  // learn
390  LabelDict::add_example_namespace_from_memory(data.label_features, *ec, costs[0].class_index);
391  uint64_t old_offset = ec->ft_offset;
392  ec->ft_offset = data.ft_offset;
393  base.learn(*ec);
394  ec->ft_offset = old_offset;
395  LabelDict::del_example_namespace_from_memory(data.label_features, *ec, costs[0].class_index);
396  ec->weight = old_weight;
397 
398  // restore original cost-sensitive label, sum of importance weights and partial_prediction
399  ec->l.cs = save_cs_label;
400  ec->partial_prediction = costs[0].partial_prediction;
401  }
402 }

410 /*
411  * 1) process all labels at first
412  * 2) verify no labels in the middle of data
413  * 3) learn_or_predict(data) with rest
414  */
415 template <bool is_learn>
416 void do_actual_learning(ldf& data, single_learner& base, multi_ex& ec_seq_all)
417 {
418  if (ec_seq_all.empty())
419  return; // nothing to do
420 
421  data.ft_offset = ec_seq_all[0]->ft_offset;
422 
423  // handle label definitions
424  auto ec_seq = process_labels(data, ec_seq_all);
425  if (ec_seq.empty())
426  return; // nothing more to do
427 
428  // Ensure there are no more labels
429  // (can be done in existing loops later but as a side effect learning
430  // will happen with bad example)
431  if (ec_seq_has_label_definition(ec_seq))
432  {
433  THROW("error: label definition encountered in data block");
434  }
435 
437  uint32_t K = (uint32_t)ec_seq.size();
438 
439  bool isTest = test_ldf_sequence(data, ec_seq);
441  uint32_t predicted_K = 0;
442  if (data.rank)
443  {
444  data.a_s.clear();
445  data.stored_preds.clear();
446  for (uint32_t k = 0; k < K; k++)
447  {
448  example* ec = ec_seq[k];
449  data.stored_preds.push_back(ec->pred.a_s);
450  make_single_prediction(data, base, *ec);
451  action_score s;
452  s.score = ec->partial_prediction;
453  s.action = k;
454  data.a_s.push_back(s);
455  }
456 
457  qsort((void*)data.a_s.begin(), data.a_s.size(), sizeof(action_score), score_comp);
458  }
459  else
460  {
461  float min_score = FLT_MAX;
462  for (uint32_t k = 0; k < K; k++)
463  {
464  example* ec = ec_seq[k];
465  make_single_prediction(data, base, *ec);
466  if (ec->partial_prediction < min_score)
467  {
468  min_score = ec->partial_prediction;
469  predicted_K = k;
470  }
471  }
472  }
473 
475  if (is_learn && !isTest)
476  {
477  if (data.is_wap)
478  do_actual_learning_wap(data, base, ec_seq);
479  else
480  do_actual_learning_oaa(data, base, ec_seq);
481  }
482 
483  if (data.rank)
484  {
485  data.stored_preds[0].clear();
486  for (size_t k = 0; k < K; k++)
487  {
488  ec_seq[k]->pred.a_s = data.stored_preds[k];
489  ec_seq[0]->pred.a_s.push_back(data.a_s[k]);
490  }
491  }
492  else
493  {
494  // Mark the predicted subexample with its class_index, all other with 0
495  for (size_t k = 0; k < K; k++)
496  {
497  if (k == predicted_K)
498  ec_seq[k]->pred.multiclass = ec_seq[k]->l.cs.costs[0].class_index;
499  else
500  ec_seq[k]->pred.multiclass = 0;
501  }
502  }
503 
505  if (data.is_probabilities)
506  {
507  float sum_prob = 0;
508  for (const auto& example : ec_seq)
509  {
510  // probability(correct_class) = 1 / (1+exp(-score)), where score is higher for better classes,
511  // but partial_prediction is lower for better classes (we are predicting the cost),
512  // so we need to take score = -partial_prediction,
513  // thus probability(correct_class) = 1 / (1+exp(-(-partial_prediction)))
514  float prob = 1.f / (1.f + correctedExp(example->partial_prediction));
515  example->pred.prob = prob;
516  sum_prob += prob;
517  }
518  // make sure that the probabilities sum up (exactly) to one
519  for (const auto& example : ec_seq)
520  {
521  example->pred.prob /= sum_prob;
522  }
523  }
524 }

540 {
541  label& ld = ec.l.cs;
542  v_array<COST_SENSITIVE::wclass> costs = ld.costs;
543 
544  if (example_is_newline(ec))
545  return;
546  if (ec_is_label_definition(ec))
547  return;

551  float loss = 0.;
552

578  {

610 }

664 void output_example_seq(vw& all, ldf& data, multi_ex& ec_seq)
665 {
666  size_t K = ec_seq.size();
667  if ((K > 0) && !ec_seq_is_label_definition(ec_seq))
668  {
669  if (test_ldf_sequence(data, ec_seq))
670  all.sd->weighted_unlabeled_examples += ec_seq[0]->weight;
671  else
672  all.sd->weighted_labeled_examples += ec_seq[0]->weight;
673  all.sd->example_number++;
674

677  output_rank_example(all, **(ec_seq.begin()), hit_loss, &(ec_seq));

684  all.print_text_by_ref(all.raw_prediction, "", empty);
685  }
686 
687  if (data.is_probabilities)
688  {
689  float min_cost = FLT_MAX;
690  size_t correct_class_k = 0;
691 
692  for (size_t k = 0; k < K; k++)
693  {
694  float ec_cost = ec_seq[k]->l.cs.costs[0].x;
695  if (ec_cost < min_cost)

697  min_cost = ec_cost;
698  correct_class_k = k;
699  }
700  }

703  float correct_class_prob = ec_seq[correct_class_k]->pred.prob;
704  if (correct_class_prob > 0)
705  multiclass_log_loss = -log(correct_class_prob);
706 
707  // TODO: How to detect if we should update holdout or normal loss?
708  // (ec.test_only) OR (COST_SENSITIVE::example_is_test(ec))
709  // What should be the "ec"? data.ec_seq[0]?
710  // Based on parse_args.cc (where "average multiclass log loss") is printed,
711  // I decided to try yet another way: (!all.holdout_set_off).
712  if (!all.holdout_set_off)
713  all.sd->holdout_multiclass_log_loss += multiclass_log_loss;
714  else
715  all.sd->multiclass_log_loss += multiclass_log_loss;
716  }
