148  void on_example(example* ec)
149  {
150  if (ec->indices.size() > 1) // 1+ nonconstant feature. (most common case first)
151  _context.template process<example, learn_ex>(*ec);

164 template <typename context_type>
165 class multi_example_handler
166 {
167  private:
168  bool complete_multi_ex(example* ec)
169  {
170  auto& master = _context.get_master();
171  const bool is_test_ec = master.p->lp.test_label(&ec->l);
172  const bool is_newline = (example_is_newline_not_header(*ec, master) && is_test_ec);
173  if (!is_newline)
174  {
175  ec_seq.push_back(ec);
176  }
177  else
178  {
179  VW::finish_example(master, *ec);
180  }
181  return is_newline;
182  }
183 
184  bool try_complete_multi_ex(example* ec)
185  {
186  if (ec->indices.size() > 1) // 1+ nonconstant feature. (most common case first)
187  return complete_multi_ex(ec);
188  else if (ec->end_pass)
189  _context.template process<example, end_pass>(*ec);
190  else if (is_save_cmd(ec))
191  _context.template process<example, save>(*ec);
192  else
193  return complete_multi_ex(ec);
194  return false;
195  }
196 
197  public:
198  multi_example_handler(const context_type context) : _context(context) {}
199 
200  ~multi_example_handler()
201  {
202  if (!ec_seq.empty())

204  _context.template process<multi_ex, learn_multi_ex>(ec_seq);

206  }
