56 void output_example(vw& all, explore_eval& c, example& ec, multi_ex* ec_seq)
57 {
58  if (example_is_newline_not_header(ec))
59  return;
60 
61  size_t num_features = 0;
62 
63  float loss = 0.;
64  ACTION_SCORE::action_scores preds = (*ec_seq)[0]->pred.a_s;
65 
66  for (size_t i = 0; i < (*ec_seq).size(); i++)
67  if (!CB::ec_is_example_header(*(*ec_seq)[i]))
68  num_features += (*ec_seq)[i]->num_features;
69 
70  bool labeled_example = true;
71  if (c.known_cost.probability > 0)
72  {
73  for (uint32_t i = 0; i < preds.size(); i++)
74  {
75  float l = get_cost_estimate(&c.known_cost, preds[i].action);
76  loss += l * preds[i].score;
77  }
78  }
79  else
80  labeled_example = false;
81 
82  bool holdout_example = labeled_example;
83  for (size_t i = 0; i < ec_seq->size(); i++) holdout_example &= (*ec_seq)[i]->test_only;
84 
85  all.sd->update(holdout_example, labeled_example, loss, ec.weight, num_features);
86 
87  for (int sink : all.final_prediction_sink) print_action_score(sink, ec.pred.a_s, ec.tag);
88 
89  if (all.raw_prediction > 0)
90  {
91  std::string outputString;
92  std::stringstream outputStringStream(outputString);
93  const auto& costs = ec.l.cb.costs;
94 
95  for (size_t i = 0; i < costs.size(); i++)
96  {
97  if (i > 0)
98  outputStringStream << ' ';
99  outputStringStream << costs[i].action << ':' << costs[i].partial_prediction;
100  }
101  all.print_text_by_ref(all.raw_prediction, outputStringStream.str(), ec.tag);
102  }
103 
104  CB::print_update(all, !labeled_example, ec, ec_seq, true);
105 }

127 template <bool is_learn>
128 void do_actual_learning(explore_eval& data, multi_learner& base, multi_ex& ec_seq)
129 {
130  example* label_example = CB_ADF::test_adf_sequence(ec_seq);
131 
132  if (label_example != nullptr) // extract label
133  {
134  data.action_label = label_example->l.cb;
135  label_example->l.cb = data.empty_label;
136  }
137  multiline_learn_or_predict<false>(base, ec_seq, data.offset);
138 
139  if (label_example != nullptr) // restore label
140  label_example->l.cb = data.action_label;
141 
142  data.known_cost = CB_ADF::get_observed_cost(ec_seq);
143  if (label_example != nullptr && is_learn)
144  {
145  ACTION_SCORE::action_scores& a_s = ec_seq[0]->pred.a_s;
146 
147  float action_probability = 0;
148  for (size_t i = 0; i < a_s.size(); i++)
149  if (data.known_cost.action == a_s[i].action)
150  action_probability = a_s[i].score;
151 
152  float threshold = action_probability / data.known_cost.probability;
153 
154  if (!data.fixed_multiplier)
155  data.multiplier = std::min(data.multiplier, 1 / threshold);
156  else
157  threshold *= data.multiplier;
158 
159  if (threshold > 1. + 1e-6)
160  data.violations++;
161 
162  if (data._random_state->get_and_update_random() < threshold)
163  {
164  example* ec_found = nullptr;
165  for (example*& ec : ec_seq)
166  {
167  if (ec->l.cb.costs.size() == 1 && ec->l.cb.costs[0].cost != FLT_MAX && ec->l.cb.costs[0].probability > 0)
168  ec_found = ec;
169  if (threshold > 1)
170  ec->weight *= threshold;
171  }
172  ec_found->l.cb.costs[0].probability = action_probability;
173 
174  multiline_learn_or_predict<true>(base, ec_seq, data.offset);
175 
176  if (threshold > 1)
177  {
178  float inv_threshold = 1.f / threshold;
179  for (auto& ec : ec_seq) ec->weight *= inv_threshold;
180  }
181  ec_found->l.cb.costs[0].probability = data.known_cost.probability;
182  data.update_count++;
183  }
184  }
185 }
