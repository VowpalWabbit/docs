216 void inject_slot_id(ccb& data, example* shared, size_t id)
217 {
218  // id is zero based, so the vector must be of size id + 1
219  if (id + 1 > data.slot_id_hashes.size())
220  {
221  data.slot_id_hashes.resize(id + 1, 0);
222  }
223 
224  uint64_t index;
225  if (data.slot_id_hashes[id] == 0)
226  {

228  index = VW::hash_feature(*data.all, current_index_str, data.id_namespace_hash);
229 
230  // To maintain indicies consistent with what the parser does we must scale.
231  index *= static_cast<uint64_t>(data.all->wpp) << data.base_learner_stride_shift;
232  data.slot_id_hashes[id] = index;
233  }
234  else
235  {
236  index = data.slot_id_hashes[id];
237  }
238 
239  shared->feature_space[ccb_id_namespace].push_back(1., index);
240  shared->indices.push_back(ccb_id_namespace);
241 
242  if (audit)
243  {
244  auto current_index_str = "index" + std::to_string(id);
245  shared->feature_space[ccb_id_namespace].space_names.push_back(
246  std::make_shared<audit_strings>(data.id_namespace_str, current_index_str));

286 void calculate_and_insert_interactions(
287  example* shared, std::vector<example*> actions, std::vector<std::string>& generated_interactions)
288 {
289  std::bitset<INTERACTIONS::printable_ns_size> found_namespaces;
290 
291  const auto original_size = generated_interactions.size();
292  for (size_t i = 0; i < original_size; i++)
293  {
294  auto interaction_copy = generated_interactions[i];
295  interaction_copy.push_back((char)ccb_id_namespace);
296  generated_interactions.push_back(interaction_copy);
297  }
298 
299  for (const auto& action : actions)
300  {
301  for (const auto& action_index : action->indices)
302  {
303  if (INTERACTIONS::is_printable_namespace(action_index) &&
304  !found_namespaces[action_index - INTERACTIONS::printable_start])
305  {
306  found_namespaces[action_index - INTERACTIONS::printable_start] = true;
307  generated_interactions.push_back({(char)action_index, (char)ccb_id_namespace});
308  }
309  }
310  }
311 
312  for (const auto& shared_index : shared->indices)
313  {
314  if (INTERACTIONS::is_printable_namespace(shared_index) &&
315  !found_namespaces[shared_index - INTERACTIONS::printable_start])
316  {
317  found_namespaces[shared_index - INTERACTIONS::printable_start] = true;
318  generated_interactions.push_back({(char)shared_index, (char)ccb_id_namespace});
319  }
320  }

323 // build a cb example from the ccb example
324 template <bool is_learn>
325 void build_cb_example(multi_ex& cb_ex, example* slot, ccb& data)
326 {
327  bool slot_has_label = slot->l.conditional_contextual_bandit.outcome != nullptr;
328 
329  // Merge the slot features with the shared example and set it in the cb multi-example
330  // TODO is it imporant for total_sum_feat_sq and num_features to be correct at this point?
331  inject_slot_features(data.shared, slot);
332  cb_ex.push_back(data.shared);
333 
334  // Retrieve the action index whitelist (if the list is empty, then all actions are white-listed)
335  auto& explicit_includes = slot->l.conditional_contextual_bandit.explicit_included_actions;
336  if (!explicit_includes.empty())
337  {
338  // First time seeing this, initialize the vector with falses so we can start setting each included action.
339  if (data.include_list.empty())
340  {
341  data.include_list.assign(data.actions.size(), false);
342  }
343 
344  for (uint32_t included_action_id : explicit_includes)
345  {
346  data.include_list[included_action_id] = true;
347  }
348  }
349 
350  // set the available actions in the cb multi-example
351  uint32_t index = 0;
352  data.origin_index.clear();
353  data.origin_index.resize(data.actions.size(), 0);
354  for (size_t i = 0; i < data.actions.size(); i++)
355  {
356  // Filter actions that are not explicitly included. If the list is empty though, everything is included.
357  if (!data.include_list.empty() && !data.include_list[i])
358  {
359  continue;
360  }
361 
362  // Filter actions chosen by previous slots
363  if (data.exclude_list[i])
364  {
365  continue;
366  }
367 
368  // Select the action
369  cb_ex.push_back(data.actions[i]);
370 
371  // Save the original index from the root multi-example
372  data.origin_index[index++] = (uint32_t)i;
373 
374  // Remember the index of the chosen action
375  if (is_learn && slot_has_label && i == slot->l.conditional_contextual_bandit.outcome->probabilities[0].action)
376  {
377  // This is used to remove the label later.
378  data.action_with_label = (uint32_t)i;
379  attach_label_to_example(index, data.actions[i], slot->l.conditional_contextual_bandit.outcome, data);
380  }
381  }
382 
383  // Must provide a prediction that cb can write into, this will be saved into the decision scores object later.
384  data.shared->pred.a_s = data.action_score_pool.get_object();
385 
386  // Tag can be used for specifying the sampling seed per slot. For it to be used it must be inserted into the shared
387  // example.
388  std::swap(data.shared->tag, slot->tag);
389 }

391 // iterate over slots contained in the multi-example, and for each slot, build a cb example and perform a
392 // cb_explore_adf call.
393 template <bool is_learn>
394 void learn_or_predict(ccb& data, multi_learner& base, multi_ex& examples)
395 {
396  clear_all(data);
397  // split shared, actions and slots
398  if (!split_multi_example_and_stash_labels(examples, data))
399  {
400  return;
401  }
402 
403 #ifndef NDEBUG
404  if (!sanity_checks<is_learn>(data))
405  {
406  return;
407  }
408 #endif
409 
410  // This will overwrite the labels with CB.
411  create_cb_labels(data);
412 
413  // Reset exclusion list for this example.
414  data.exclude_list.assign(data.actions.size(), false);
415 
416  auto decision_scores = examples[0]->pred.decision_scores;
417 
418  // for each slot, re-build the cb example and call cb_explore_adf
419  size_t slot_id = 0;
420  for (example* slot : data.slots)
421  {
422  // Namespace crossing for slot features.
423  data.generated_interactions.clear();
424  std::copy(data.original_interactions->begin(), data.original_interactions->end(),
425  std::back_inserter(data.generated_interactions));
426  calculate_and_insert_interactions(data.shared, data.actions, data.generated_interactions);
427  data.shared->interactions = &data.generated_interactions;
428  for (auto ex : data.actions)
429  {
430  ex->interactions = &data.generated_interactions;
431  }
432 
433  data.include_list.clear();
434  build_cb_example<is_learn>(data.cb_ex, slot, data);
435 
436  if (data.all->audit)
437  {
438  inject_slot_id<true>(data, data.shared, slot_id);
439  }
440  else
441  {
442  inject_slot_id<false>(data, data.shared, slot_id);
443  }
444 
445  if (has_action(data.cb_ex))
446  {
447  // the cb example contains at least 1 action
448  multiline_learn_or_predict<is_learn>(base, data.cb_ex, examples[0]->ft_offset);
449  save_action_scores(data, decision_scores);
450  clear_pred_and_label(data);
451  }
452  else
453  {
454  // the cb example contains no action => cannot decide
455  decision_scores.push_back(data.action_score_pool.get_object());
456  }
457 
458  data.shared->interactions = data.original_interactions;
459  for (auto ex : data.actions)
460  {
461  ex->interactions = data.original_interactions;
462  }
463  remove_slot_features(data.shared, slot);
464 
465  if (data.all->audit)
466  {
467  remove_slot_id<true>(data.shared);
468  }
469  else
470  {
471  remove_slot_id<false>(data.shared);
472  }
473 
474  // Put back the original shared example tag.
475  std::swap(data.shared->tag, slot->tag);
476  slot_id++;
477  data.cb_ex.clear();
478  }
479 
480  delete_cb_labels(data);
481 
482  // Restore ccb labels to the example objects.
483  for (size_t i = 0; i < examples.size(); i++)
484  {
485  examples[i]->l.conditional_contextual_bandit = {
486  data.stored_labels[i].type, data.stored_labels[i].outcome, data.stored_labels[i].explicit_included_actions, 0.};
487  }
488 
489  // Save the predictions
490  examples[0]->pred.decision_scores = decision_scores;
491 }

581 void output_example(vw& all, ccb& /*c*/, multi_ex& ec_seq)
582 {
583  if (ec_seq.empty())
584  {
585  return;
586  }
587 
588  std::vector<example*> slots;
589  size_t num_features = 0;
590  float loss = 0.;
591 
592  // Should this be done for shared, action and slot?
593  for (auto ec : ec_seq)
594  {
595  num_features += ec->num_features;
596 
597  if (ec->l.conditional_contextual_bandit.type == CCB::example_type::slot)
598  {
599  slots.push_back(ec);
600  }
601  }
602 
603  // Is it hold out?
604  size_t num_labelled = 0;
605  auto preds = ec_seq[0]->pred.decision_scores;
606  for (size_t i = 0; i < slots.size(); i++)
607  {
608  auto outcome = slots[i]->l.conditional_contextual_bandit.outcome;
609  if (outcome != nullptr)
610  {
611  num_labelled++;
612  float l = CB_ALGS::get_cost_estimate(
613  outcome->probabilities[TOP_ACTION_INDEX], outcome->cost, preds[i][TOP_ACTION_INDEX].action);
614  loss += l * preds[i][TOP_ACTION_INDEX].score;
615  }
616  }
617 
618  if (num_labelled > 0 && num_labelled < slots.size())
619  {
620  std::cerr << "Warning: Unlabeled example in train set, was this intentional?\n";
621  }
622 
623  bool holdout_example = num_labelled > 0;
624  for (const auto& example : ec_seq)
625  {
626  holdout_example &= example->test_only;
627  }
628 
629  // TODO what does weight mean here?
630  all.sd->update(holdout_example, num_labelled > 0, loss, ec_seq[SHARED_EX_INDEX]->weight, num_features);
631 
632  for (auto sink : all.final_prediction_sink)
633  {
634  print_decision_scores(sink, ec_seq[SHARED_EX_INDEX]->pred.decision_scores);
635  }
636 
637  CCB::print_update(all, slots, preds, num_features);
638 }
