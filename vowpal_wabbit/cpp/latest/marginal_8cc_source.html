47 float get_adanormalhedge_weights(float R, float C)
48 {
49  float Rpos = R > 0 ? R : 0.f;
50  if (C == 0. || Rpos == 0.)
51  return 0;
52  return 2 * Rpos * correctedExp(Rpos * Rpos / (3 * C)) / (3 * C);

55 template <bool is_learn>
56 void make_marginal(data& sm, example& ec)
57 {
58  uint64_t mask = sm.all->weights.mask();
59  float label = ec.l.simple.label;
60  vw& all = *sm.all;
61  sm.alg_loss = 0.;
62  sm.net_weight = 0.;
63  sm.net_feature_weight = 0.;
64  sm.average_pred = 0.;
65 
66  for (example::iterator i = ec.begin(); i != ec.end(); ++i)
67  {
68  namespace_index n = i.index();
69  if (sm.id_features[n])
70  {
71  std::swap(sm.temp[n], *i);
72  features& f = *i;
73  f.clear();
74  for (features::iterator j = sm.temp[n].begin(); j != sm.temp[n].end(); ++j)
75  {
76  float first_value = j.value();
77  uint64_t first_index = j.index() & mask;
78  if (++j == sm.temp[n].end())
79  {
80  std::cout << "warning: id feature namespace has " << sm.temp[n].size()
81  << " features. Should be a multiple of 2" << std::endl;
82  break;
83  }
84  float second_value = j.value();
85  uint64_t second_index = j.index() & mask;
86  if (first_value != 1. || second_value != 1.)
87  {
88  std::cout << "warning: bad id features, must have value 1." << std::endl;
89  continue;
90  }
91  uint64_t key = second_index + ec.ft_offset;
92  if (sm.marginals.find(key) == sm.marginals.end()) // need to initialize things.
93  {
94  sm.marginals.insert(std::make_pair(key, std::make_pair(sm.initial_numerator, sm.initial_denominator)));
95  if (sm.compete)
96  {
97  expert e = {0, 0, 1.};
98  sm.expert_state.insert(std::make_pair(key, std::make_pair(e, e)));
99  }
100  }
101  float marginal_pred = (float)(sm.marginals[key].first / sm.marginals[key].second);
102  f.push_back(marginal_pred, first_index);
103  if (!sm.temp[n].space_names.empty())
104  f.space_names.push_back(sm.temp[n].space_names[2 * (f.size() - 1)]);
105 
106  if (sm.compete) // compute the prediction from the marginals using the weights
107  {
108  float weight = sm.expert_state[key].first.weight;
109  sm.average_pred += weight * marginal_pred;
110  sm.net_weight += weight;
111  sm.net_feature_weight += sm.expert_state[key].second.weight;
112  if (is_learn)
113  sm.alg_loss += weight * all.loss->getLoss(all.sd, marginal_pred, label);
114  }
115  }
116  }
117  }
118 }

130 template <bool is_learn>
131 void compute_expert_loss(data& sm, example& ec)
132 {
133  vw& all = *sm.all;
134  // add in the feature-based expert and normalize,
135  float label = ec.l.simple.label;
136 
137  if (sm.net_weight + sm.net_feature_weight > 0.)
138  sm.average_pred += sm.net_feature_weight * sm.feature_pred;
139  else
140  {
141  sm.net_feature_weight = 1.;
142  sm.average_pred = sm.feature_pred;
143  }
144  float inv_weight = 1.0f / (sm.net_weight + sm.net_feature_weight);
145  sm.average_pred *= inv_weight;
146  ec.pred.scalar = sm.average_pred;
147  ec.partial_prediction = sm.average_pred;
148 
149  if (is_learn)
150  {
151  sm.alg_loss += sm.net_feature_weight * all.loss->getLoss(all.sd, sm.feature_pred, label);
152  sm.alg_loss *= inv_weight;
153  }
154 }

156 void update_marginal(data& sm, example& ec)
157 {
158  vw& all = *sm.all;
159  uint64_t mask = sm.all->weights.mask();
160  float label = ec.l.simple.label;
161  float weight = ec.weight;
162  if (sm.unweighted_marginals)
163  weight = 1.;
164 
165  for (example::iterator i = ec.begin(); i != ec.end(); ++i)
166  {
167  namespace_index n = i.index();
168  if (sm.id_features[n])
169  for (features::iterator j = sm.temp[n].begin(); j != sm.temp[n].end(); ++j)
170  {
171  if (++j == sm.temp[n].end())
172  break;
173 
174  uint64_t second_index = j.index() & mask;
175  uint64_t key = second_index + ec.ft_offset;
176  marginal& m = sm.marginals[key];
177 
178  if (sm.compete) // now update weights, before updating marginals
179  {
180  expert_pair& e = sm.expert_state[key];
181  float regret1 = sm.alg_loss - all.loss->getLoss(all.sd, (float)(m.first / m.second), label);
182  float regret2 = sm.alg_loss - all.loss->getLoss(all.sd, sm.feature_pred, label);
183 
184  e.first.regret += regret1 * weight;
185  e.first.abs_regret += regret1 * regret1 * weight; // fabs(regret1);
186  e.first.weight = get_adanormalhedge_weights(e.first.regret, e.first.abs_regret);
187  e.second.regret += regret2 * weight;
188  e.second.abs_regret += regret2 * regret2 * weight; // fabs(regret2);
189  e.second.weight = get_adanormalhedge_weights(e.second.regret, e.second.abs_regret);
190  }
191 
192  m.first = m.first * (1. - sm.decay) + ec.l.simple.label * weight;
193  m.second = m.second * (1. - sm.decay) + weight;
194  }
195  }

198 template <bool is_learn>
199 void predict_or_learn(data& sm, LEARNER::single_learner& base, example& ec)
200 {
201  make_marginal<is_learn>(sm, ec);
202  if (is_learn)
203  if (sm.update_before_learn)
204  {
205  base.predict(ec);
206  float pred = ec.pred.scalar;
207  if (sm.compete)
208  {
209  sm.feature_pred = pred;
210  compute_expert_loss<is_learn>(sm, ec);
211  }
212  undo_marginal(sm, ec);
213  update_marginal(sm, ec); // update features before learning.
214  make_marginal<is_learn>(sm, ec);
215  base.learn(ec);
216  ec.pred.scalar = pred;
217  }
218  else
219  {
220  base.learn(ec);
221  if (sm.compete)
222  {
223  sm.feature_pred = ec.pred.scalar;
224  compute_expert_loss<is_learn>(sm, ec);
225  }
226  update_marginal(sm, ec);
227  }
228  else
229  {
230  base.predict(ec);
231  float pred = ec.pred.scalar;
232  if (sm.compete)
233  {
234  sm.feature_pred = pred;
235  compute_expert_loss<is_learn>(sm, ec);
236  }
237  }
238 
239  // undo marginalization
240  undo_marginal(sm, ec);
241 }
