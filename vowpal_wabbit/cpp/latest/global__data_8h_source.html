70 
71  public:
72  namedlabels(const std::string& label_list) : m_label_list(label_list)
73  {
74  tokenize(',', m_label_list, m_id2name);
75 
76  m_K = static_cast<uint32_t>(m_id2name.size());
77  m_name2id.max_load_factor(0.25);
78  m_name2id.reserve(m_K);
79 
80  for (uint32_t k = 0; k < m_K; k++)
81  {
82  const VW::string_view& l = m_id2name[static_cast<size_t>(k)];
83  auto iter = m_name2id.find(l);
84  if (iter != m_name2id.end())
85  THROW("error: label dictionary initialized with multiple occurances of: " << l);
86  m_name2id.emplace(l, k + 1);
87  }
88  }
89 
90  uint32_t getK() { return m_K; }
91 
92  uint32_t get(VW::string_view s) const
93  {
94  auto iter = m_name2id.find(s);
95  if (iter == m_name2id.end())
96  {
97  std::cerr << "warning: missing named label '" << s << '\'' << std::endl;
98  return 0;
99  }
100  return iter->second;
101  }
102 
103  VW::string_view get(uint32_t v) const
104  {
105  static_assert(sizeof(size_t) >= sizeof(uint32_t), "size_t is smaller than 32-bits. Potential overflow issues.");
106  if ((v == 0) || (v > m_K))
107  {

109  }
