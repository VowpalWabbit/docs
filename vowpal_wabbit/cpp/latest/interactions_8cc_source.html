17 // recursive function used internally in this module
18 void expand_namespaces_with_recursion(
19  std::string const& ns, std::vector<std::string>& res, std::string& val, size_t pos)
20 {
21  assert(pos <= ns.size());
22 
23  if (pos == ns.size())
24  {
25  // we're at the end of interaction
26 
27  // and store it in res
28  res.push_back(val);
29  // don't free s memory as it's data will be used later
30  }
31  else
32  {
33  // we're at the middle of interaction
34  if (ns[pos] != ':')
35  {
36  // not a wildcard
37  val.push_back(ns[pos]);
38  expand_namespaces_with_recursion(ns, res, val, pos + 1);
39  val.pop_back(); // i don't need value itself
40  }
41  else
42  {
43  for (unsigned char j = printable_start; j <= printable_end; ++j)
44  {
45  if (valid_ns(j))
46  {
47  val.push_back(j);
48  expand_namespaces_with_recursion(ns, res, val, pos + 1);
49  val.pop_back(); // i don't need value itself
50  }
51  }
52  }
53  }
54 }

87 // returns true if iteraction contains one or more duplicated namespaces
88 // with one exeption - returns false if interaction made of one namespace
89 // like 'aaa' as it has no sense to sort such things.

91 inline bool must_be_left_sorted(const std::string& oi)
92 {
93  if (oi.size() <= 1)
94  return true; // one letter in std::string - no need to sort
95 
96  bool diff_ns_found = false;
97  bool pair_found = false;
98 
99  for (auto i = std::begin(oi); i != std::end(oi) - 1; ++i)
100  if (*i == *(i + 1)) // pair found
101  {
102  if (diff_ns_found)
103  return true; // case 'abb'
104  pair_found = true;
105  }
106  else
107  {
108  if (pair_found)
109  return true; // case 'aab'
110  diff_ns_found = true;
111  }
112 
113  return false; // 'aaa' or 'abc'
114 }

120 void sort_and_filter_duplicate_interactions(
121  std::vector<std::string>& vec, bool filter_duplicates, size_t& removed_cnt, size_t& sorted_cnt)
122 {
123  // 2 out parameters
124  removed_cnt = 0;
125  sorted_cnt = 0;
126 
127  // interaction value sort + original position
128  std::vector<std::pair<std::string, size_t>> vec_sorted;
129  for (size_t i = 0; i < vec.size(); ++i)
130  {
131  std::string sorted_i(vec[i]);
132  std::stable_sort(std::begin(sorted_i), std::end(sorted_i));
133  vec_sorted.push_back(std::make_pair(sorted_i, i));
134  }
135 
136  if (filter_duplicates)
137  {
138  // remove duplicates
139  std::stable_sort(vec_sorted.begin(), vec_sorted.end(),
140  [](std::pair<std::string, size_t> const& a, std::pair<std::string, size_t> const& b) {
141  return a.first < b.first;
142  });
143  auto last = unique(vec_sorted.begin(), vec_sorted.end(),
144  [](std::pair<std::string, size_t> const& a, std::pair<std::string, size_t> const& b) {
145  return a.first == b.first;
146  });
147  vec_sorted.erase(last, vec_sorted.end());
148 
149  // report number of removed interactions
150  removed_cnt = vec.size() - vec_sorted.size();
151 
152  // restore original order
153  std::stable_sort(vec_sorted.begin(), vec_sorted.end(),
154  [](std::pair<std::string, size_t> const& a, std::pair<std::string, size_t> const& b) {
155  return a.second < b.second;
156  });
157  }
158 
159  // we have original vector and vector with duplicates removed + corresponding indexes in original vector
160  // plus second vector's data is sorted. We can reuse it if we need interaction to be left sorted.
161  // let's make a new vector from these two sources - without dulicates and with sorted data whenever it's needed.
162  std::vector<std::string> res;
163  for (auto& i : vec_sorted)
164  {
165  if (must_be_left_sorted(i.first))
166  {
167  // if so - copy sorted data to result
168  res.push_back(i.first);
169  ++sorted_cnt;
170  }
171  else // else - move unsorted data to result
172  res.push_back(vec[i.second]);
173  }
174 
175  vec = res;
176 }

207 constexpr int64_t fast_factorial[] = {1, 1, 2, 6, 24, 120, 720, 5040, 40320, 362880, 3628800, 39916800, 479001600,
208  6227020800, 87178291200, 1307674368000, 20922789888000, 355687428096000, 6402373705728000, 121645100408832000,

210 constexpr size_t size_fast_factorial = sizeof(fast_factorial) / sizeof(*fast_factorial);

212 // helper factorial function that allows to perform:
213 // n!/(n-k)! = (n-k+1)*(n-k+2)..*(n-1)*n
214 // by specifying n-k as second argument
215 // that helps to avoid size_t overflow for big n
216 // leave second argument = 1 to get regular factorial function
217 
218 inline size_t factor(const size_t n, const size_t start_from = 1)
219 {
220  if (n <= 0)
221  return 1;
222  if (start_from == 1 && n < size_fast_factorial)
223  return (size_t)fast_factorial[n];
224 
225  size_t res = 1;
226  for (size_t i = start_from + 1; i <= n; ++i) res *= i;
227  return res;
228 }

232 void eval_count_of_generated_ft(vw& all, example& ec, size_t& new_features_cnt, float& new_features_value)
233 {
234  new_features_cnt = 0;
235  new_features_value = 0.;
236 
237  v_array<float> results = v_init<float>();
238 
239  if (all.permutations)
240  {
241  // just multiply precomputed values for all namespaces
242  for (std::string& inter : *ec.interactions)
243  {
244  size_t num_features_in_inter = 1;
245  float sum_feat_sq_in_inter = 1.;
246 
247  for (namespace_index ns : inter)
248  {
249  num_features_in_inter *= ec.feature_space[ns].size();
250  sum_feat_sq_in_inter *= ec.feature_space[ns].sum_feat_sq;
251  if (num_features_in_inter == 0)
252  break;
253  }
254 
255  if (num_features_in_inter == 0)
256  continue;
257 
258  new_features_cnt += num_features_in_inter;
259  new_features_value += sum_feat_sq_in_inter;
260  }
261  }
262  else // case of simple combinations
263  {
264 #ifdef DEBUG_EVAL_COUNT_OF_GEN_FT
265  size_t correct_features_cnt = 0;
266  float correct_features_value = 0.;
267  eval_gen_data dat(correct_features_cnt, correct_features_value);
268  generate_interactions<eval_gen_data, uint64_t, ft_cnt>(all, ec, dat);
269 #endif
270 
271  for (std::string& inter : *ec.interactions)
272  {
273  size_t num_features_in_inter = 1;
274  float sum_feat_sq_in_inter = 1.;
275 
276  for (auto ns = inter.begin(); ns != inter.end(); ++ns)
277  {
278  if ((ns == inter.end() - 1) || (*ns != *(ns + 1))) // neighbour namespaces are different
279  {
280  // just multiply precomputed values
281  const int nsc = *ns;
282  num_features_in_inter *= ec.feature_space[nsc].size();
283  sum_feat_sq_in_inter *= ec.feature_space[nsc].sum_feat_sq;
284  if (num_features_in_inter == 0)
285  break; // one of namespaces has no features - go to next interaction
286  }
287  else // we are at beginning of a block made of same namespace (interaction is preliminary sorted)
288  {
289  // let's find out real length of this block
290  size_t order_of_inter = 2; // alredy compared ns == ns+1
291 
292  for (auto ns_end = ns + 2; ns_end < inter.end(); ++ns_end)
293  if (*ns == *ns_end)
294  ++order_of_inter;
295 
296  // namespace is same for whole block
297  features& fs = ec.feature_space[(const int)*ns];
298 
299  // count number of features with value != 1.;
300  size_t cnt_ft_value_non_1 = 0;
301 
302  // in this block we shall calculate number of generated features and sum of their values
303  // keeping in mind rules applicable for simple combinations instead of permutations
304 
305  // let's calculate sum of their squared value for whole block
306 
307  // ensure results as big as order_of_inter and empty.
308  for (size_t i = 0; i < results.size(); ++i) results[i] = 0.;
309  while (results.size() < order_of_inter) results.push_back(0.);
310 
311  // recurrent value calculations
312  for (size_t i = 0; i < fs.size(); ++i)
313  {
314  const float x = fs.values[i] * fs.values[i];
315 
316  if (!PROCESS_SELF_INTERACTIONS(fs.values[i]))
317  {
318  for (size_t i = order_of_inter - 1; i > 0; --i) results[i] += results[i - 1] * x;
319 
320  results[0] += x;
321  }
322  else
323  {
324  results[0] += x;
325 
326  for (size_t i = 1; i < order_of_inter; ++i) results[i] += results[i - 1] * x;
327 
328  ++cnt_ft_value_non_1;
329  }
330  }
331 
332  sum_feat_sq_in_inter *= results[order_of_inter - 1]; // will be explained in http://bit.ly/1Hk9JX1
333 
334  // let's calculate the number of a new features
335 
336  // if number of features is less than order of interaction then go to the next interaction
337  // as you can't make simple combination of interaction 'aaa' if a contains < 3 features.
338  // unless one of them has value != 1. and we are counting them.
339  const size_t ft_size = fs.size();
340  if (cnt_ft_value_non_1 == 0 && ft_size < order_of_inter)
341  {
342  num_features_in_inter = 0;
343  break;
344  }
345 
346  size_t n;
347  if (cnt_ft_value_non_1 == 0) // number of generated simple combinations is C(n,k)
348  {
349  n = (size_t)choose((int64_t)ft_size, (int64_t)order_of_inter);
350  }
351  else
352  {
353  n = 0;
354  for (size_t l = 0; l <= order_of_inter; ++l)
355  {
356  // C(l+m-1, l) * C(n-m, k-l)
357  size_t num = (l == 0) ? 1 : (size_t)choose(l + cnt_ft_value_non_1 - 1, l);
358 
359  if (ft_size - cnt_ft_value_non_1 >= order_of_inter - l)
360  num *= (size_t)choose(ft_size - cnt_ft_value_non_1, order_of_inter - l);
361  else
362  num = 0;
363 
364  n += num;
365  }
366 
367  } // details on http://bit.ly/1Hk9JX1
368 
369  num_features_in_inter *= n;
370 
371  ns += order_of_inter - 1; // jump over whole block
372  }
373  }
374 
375  if (num_features_in_inter == 0)
376  continue; // signal that values should be ignored (as default value is 1)
377 
378  new_features_cnt += num_features_in_inter;
379  new_features_value += sum_feat_sq_in_inter;
380  }
381 
382 #ifdef DEBUG_EVAL_COUNT_OF_GEN_FT
383  if (correct_features_cnt != new_features_cnt)
384  all.trace_message << "Incorrect new features count " << new_features_cnt << " must be " << correct_features_cnt
385  << std::endl;
386  if (fabs(correct_features_value - new_features_value) > 1e-5)
387  all.trace_message << "Incorrect new features value " << new_features_value << " must be "
388  << correct_features_value << std::endl;
389 #endif
390  }
391 
392  results.delete_v();
393 }
