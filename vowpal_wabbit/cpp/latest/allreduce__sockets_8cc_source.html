84 socket_t AllReduceSockets::getsock()
85 {
86  socket_t sock = socket(PF_INET, SOCK_STREAM, 0);
87  if (sock < 0)
88  THROWERRNO("socket");
89 
90  // SO_REUSEADDR will allow port rebinding on Windows, causing multiple instances
91  // of VW on the same machine to potentially contact the wrong tree node.
92 #ifndef _WIN32
93  int on = 1;
94  if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, (char*)&on, sizeof(on)) < 0)
95  {
96  if (!quiet)
97  cerr << "setsockopt SO_REUSEADDR: " << strerror(errno) << endl;
98  }
99 #endif
100 
101  // Enable TCP Keep Alive to prevent socket leaks
102  int enableTKA = 1;
103  if (setsockopt(sock, SOL_SOCKET, SO_KEEPALIVE, (char*)&enableTKA, sizeof(enableTKA)) < 0)
104  {
105  if (!quiet)
106  cerr << "setsockopt SO_KEEPALIVE: " << strerror(errno) << endl;
107  }
108 
109  return sock;

291 {
292  size_t my_bufsize = std::min(ar_buf_size, (parent_read_pos - children_sent_pos));
293 
294  if (my_bufsize > 0)
295  {
296  // going to pass up this chunk of data to the children
297  if (socks.children[0] != -1 &&
298  send(socks.children[0], buffer + children_sent_pos, (int)my_bufsize, 0) < (int)my_bufsize)
299  {
300  THROW("Write to left child failed");
301  }
302  if (socks.children[1] != -1 &&
303  send(socks.children[1], buffer + children_sent_pos, (int)my_bufsize, 0) < (int)my_bufsize)
304  {
305  THROW("Write to right child failed");
306  }
307 
308  children_sent_pos += my_bufsize;
309  }

312 void AllReduceSockets::broadcast(char* buffer, const size_t n)
313 {
314  size_t parent_read_pos = 0; // First unread float from parent
315  size_t children_sent_pos = 0; // First unsent float to children
316  // parent_sent_pos <= left_read_pos
317  // parent_sent_pos <= right_read_pos
318 
319  if (socks.parent == -1)
320  {
321  parent_read_pos = n;
322  }
323  if (socks.children[0] == -1 && socks.children[1] == -1)
324  children_sent_pos = n;
325 
326  while (parent_read_pos < n || children_sent_pos < n)
327  {
328  pass_down(buffer, parent_read_pos, children_sent_pos);
329  if (parent_read_pos >= n && children_sent_pos >= n)
330  break;
331 
332  if (socks.parent != -1)
333  {
334  // there is data to be read from the parent
335  if (parent_read_pos == n)
336  THROW("I think parent has no data to send but he thinks he has");
337 
338  size_t count = std::min(ar_buf_size, n - parent_read_pos);
339  int read_size = recv(socks.parent, buffer + parent_read_pos, (int)count, 0);
340  if (read_size == -1)
341  {
342  THROW(" recv from parent: " << strerror(errno));
343  }
344  parent_read_pos += read_size;
345  }
346  }
