41 int build_tree(int* parent, uint16_t* kid_count, size_t source_count, int offset)
42 {
43  if (source_count == 1)
44  {
45  kid_count[offset] = 0;
46  return offset;
47  }
48 
49  int height = (int)floor(log((double)source_count) / log(2.0));
50  int root = (1 << height) - 1;
51  int left_count = root;
52  int left_offset = offset;
53  int left_child = build_tree(parent, kid_count, left_count, left_offset);
54  int oroot = root + offset;
55  parent[left_child] = oroot;
56 
57  size_t right_count = source_count - left_count - 1;
58  if (right_count > 0)
59  {
60  int right_offset = oroot + 1;
61 
62  int right_child = build_tree(parent, kid_count, right_count, right_offset);
63  parent[right_child] = oroot;
64  kid_count[oroot] = 2;
65  }
66  else
67  kid_count[oroot] = 1;
68 
69  return oroot;
70 }

173  }
174

177  THROWERRNO("inet_ntop: ");
178

180  char servInfo[NI_MAXSERV];

182  THROWERRNO("getnameinfo: ");

184  if (!m_quiet)
185  std::cerr << "inbound connection from " << dotted_quad << "(" << hostname << ':' << ntohs(m_port)
186  << ") serv=" << servInfo << std::endl;

188  size_t nonce = 0;
189  if (recv(f, (char*)&nonce, sizeof(nonce), 0) != sizeof(nonce))
190  {
191  THROW(dotted_quad << "(" << hostname << ':' << ntohs(m_port) << "): nonce read failed, exiting");

197  }

200  {

202  }

204  {

226  }

233  partial_nodeset.filled = 0;

239  }
