16 char* bufread_label(CB::label* ld, char* c, io_buf& cache)
17 {
18  size_t num = *(size_t*)c;
19  ld->costs.clear();
20  c += sizeof(size_t);
21  size_t total = sizeof(cb_class) * num + sizeof(ld->weight);
22  if (cache.buf_read(c, total) < total)
23  {
24  std::cout << "error in demarshal of cost data" << std::endl;
25  return c;
26  }
27  for (size_t i = 0; i < num; i++)
28  {
29  cb_class temp = *(cb_class*)c;
30  c += sizeof(cb_class);
31  ld->costs.push_back(temp);
32  }
33  memcpy(&ld->weight, c, sizeof(ld->weight));
34  c += sizeof(ld->weight);

36 }

111 void parse_label(parser* p, shared_data*, void* v, v_array<VW::string_view>& words)
112 {
113  CB::label* ld = (CB::label*)v;
114  ld->costs.clear();
115  ld->weight = 1.0;
116 
117  for (auto const& word : words)
118  {
119  cb_class f;
120  tokenize(':', word, p->parse_name);
121 
122  if (p->parse_name.empty() || p->parse_name.size() > 3)
123  THROW("malformed cost specification: " << p->parse_name);
124 
125  f.partial_prediction = 0.;
126  f.action = (uint32_t)hashstring(p->parse_name[0].begin(), p->parse_name[0].length(), 0);
127  f.cost = FLT_MAX;
128 
129  if (p->parse_name.size() > 1)
130  f.cost = float_of_string(p->parse_name[1]);
131 
132  if (std::isnan(f.cost))
133  THROW("error NaN cost (" << p->parse_name[1] << " for action: " << p->parse_name[0]);
134 
135  f.probability = .0;
136  if (p->parse_name.size() > 2)
137  f.probability = float_of_string(p->parse_name[2]);
138 
139  if (std::isnan(f.probability))
140  THROW("error NaN probability (" << p->parse_name[2] << " for action: " << p->parse_name[0]);
141 
142  if (f.probability > 1.0)
143  {
144  std::cerr << "invalid probability > 1 specified for an action, resetting to 1." << std::endl;
145  f.probability = 1.0;
146  }
147  if (f.probability < 0.0)
148  {
149  std::cerr << "invalid probability < 0 specified for an action, resetting to 0." << std::endl;
150  f.probability = .0;
151  }
152  if (p->parse_name[0] == "shared")
153  {
154  if (p->parse_name.size() == 1)
155  {
156  f.probability = -1.f;
157  }
158  else
159  std::cerr << "shared feature vectors should not have costs" << std::endl;
160  }
161 
162  ld->costs.push_back(f);
163  }
164 }
