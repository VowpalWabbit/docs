38 void predict_or_learn(sfm_data&, LEARNER::multi_learner& base, multi_ex& ec_seq)
39 {
40  if (ec_seq.size() == 0)
41  THROW("cb_adf: At least one action must be provided for an example to be valid.");
42 
43  multi_ex::value_type shared_example = nullptr;
44 
45  const bool has_example_header = CB::ec_is_example_header(*ec_seq[0]);
46  if (has_example_header)
47  {
48  shared_example = ec_seq[0];
49  ec_seq.erase(ec_seq.begin());
50  // merge sequences
51  for (auto& example : ec_seq) LabelDict::add_example_namespaces_from_example(*example, *shared_example);
52  std::swap(ec_seq[0]->pred, shared_example->pred);
53  }
54  if (ec_seq.size() == 0)
55  return;
56  if (is_learn)
57  base.learn(ec_seq);
58  else
59  base.predict(ec_seq);
60 
61  if (has_example_header)
62  {
63  for (auto& example : ec_seq) LabelDict::del_example_namespaces_from_example(*example, *shared_example);
64  std::swap(shared_example->pred, ec_seq[0]->pred);
65  ec_seq.insert(ec_seq.begin(), shared_example);
66  }
67 }
