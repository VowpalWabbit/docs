50 template <bool is_learn>
51 void gen_cs_example_dm(cb_to_cs& c, example& ec, COST_SENSITIVE::label& cs_ld)
52 { // this implements the direct estimation method, where costs are directly specified by the learned regressor.
53  CB::label ld = ec.l.cb;
54 
55  float min = FLT_MAX;
56  uint32_t argmin = 1;
57  // generate cost sensitive example
58  cs_ld.costs.clear();
59  c.pred_scores.costs.clear();
60 
61  if (ld.costs.size() == 0 ||
62  (ld.costs.size() == 1 &&
63  ld.costs[0].cost != FLT_MAX)) // this is a typical example where we can perform all actions
64  { // in this case generate cost-sensitive example with all actions
65  for (uint32_t i = 1; i <= c.num_actions; i++)
66  {
67  COST_SENSITIVE::wclass wc = {0., i, 0., 0.};
68  // get cost prediction for this action
69  wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, c.known_cost, ec, i, 0);
70  if (wc.x < min)
71  {
72  min = wc.x;
73  argmin = i;
74  }
75 
76  c.pred_scores.costs.push_back(wc);
77 
78  if (c.known_cost != nullptr && c.known_cost->action == i)
79  {
80  c.nb_ex_regressors++;
81  c.avg_loss_regressors += (1.0f / c.nb_ex_regressors) *
82  ((c.known_cost->cost - wc.x) * (c.known_cost->cost - wc.x) - c.avg_loss_regressors);
83  c.last_pred_reg = wc.x;
84  c.last_correct_cost = c.known_cost->cost;
85  }
86 
87  cs_ld.costs.push_back(wc);
88  }
89  }
90  else // this is an example where we can only perform a subset of the actions
91  { // in this case generate cost-sensitive example with only allowed actions
92  for (auto& cl : ld.costs)
93  {
94  COST_SENSITIVE::wclass wc = {0., cl.action, 0., 0.};
95 
96  // get cost prediction for this action
97  wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, c.known_cost, ec, cl.action, 0);
98  if (wc.x < min || (wc.x == min && cl.action < argmin))
99  {
100  min = wc.x;
101  argmin = cl.action;
102  }
103  c.pred_scores.costs.push_back(wc);
104 
105  if (c.known_cost != nullptr && c.known_cost->action == cl.action)
106  {
107  c.nb_ex_regressors++;
108  c.avg_loss_regressors += (1.0f / c.nb_ex_regressors) *
109  ((c.known_cost->cost - wc.x) * (c.known_cost->cost - wc.x) - c.avg_loss_regressors);
110  c.last_pred_reg = wc.x;
111  c.last_correct_cost = c.known_cost->cost;
112  }
113 
114  cs_ld.costs.push_back(wc);
115  }
116  }
117 
118  ec.pred.multiclass = argmin;
119 }

121 template <bool is_learn>
122 void gen_cs_label(cb_to_cs& c, example& ec, COST_SENSITIVE::label& cs_ld, uint32_t action, float clip_p = 0.f)
123 {
124  COST_SENSITIVE::wclass wc = {0., action, 0., 0.};
125 
126  // get cost prediction for this action
127  wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, c.known_cost, ec, action, c.num_actions);
128 
129  c.pred_scores.costs.push_back(wc);
130  // add correction if we observed cost for this action and regressor is wrong
131  if (c.known_cost != nullptr && c.known_cost->action == action)
132  {
133  c.nb_ex_regressors++;
134  c.avg_loss_regressors += (1.0f / c.nb_ex_regressors) *
135  ((c.known_cost->cost - wc.x) * (c.known_cost->cost - wc.x) - c.avg_loss_regressors);
136  c.last_pred_reg = wc.x;
137  c.last_correct_cost = c.known_cost->cost;
138  wc.x += (c.known_cost->cost - wc.x) / std::max(c.known_cost->probability, clip_p);
139  }
140 
141  cs_ld.costs.push_back(wc);
142 }

194 template <bool is_learn>
195 void gen_cs_example_dr(cb_to_cs_adf& c, multi_ex& examples, COST_SENSITIVE::label& cs_labels, float clip_p = 0.f)
196 { // size_t mysize = examples.size();
197  c.pred_scores.costs.clear();
198 
199  cs_labels.costs.clear();
200  for (size_t i = 0; i < examples.size(); i++)
201  {
202  if (CB_ALGS::example_is_newline_not_header(*examples[i]))
203  continue;
204 
205  COST_SENSITIVE::wclass wc = {0., (uint32_t)i, 0., 0.};
206 
207  if (c.known_cost.action == i)
208  {
209  int known_index = c.known_cost.action;
210  c.known_cost.action = 0;
211  // get cost prediction for this label
212  // num_actions should be 1 effectively.
213  // my get_cost_pred function will use 1 for 'index-1+base'
214  wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, &(c.known_cost), *(examples[i]), 0, 2);
215  c.known_cost.action = known_index;
216  }
217  else
218  wc.x = CB_ALGS::get_cost_pred<is_learn>(c.scorer, nullptr, *(examples[i]), 0, 2);
219 
220  c.pred_scores.costs.push_back(wc); // done
221 
222  // add correction if we observed cost for this action and regressor is wrong
223  if (c.known_cost.probability != -1 && c.known_cost.action == i)
224  wc.x += (c.known_cost.cost - wc.x) / std::max(c.known_cost.probability, clip_p);
225  cs_labels.costs.push_back(wc);
226  }
227 }

253  if (prepped_cs_labels.size() < cs_labels.costs.size() + 1)
254  {
255  prepped_cs_labels.resize(cs_labels.costs.size() + 1);
256  prepped_cs_labels.end() = prepped_cs_labels.end_array;
257  }
258 
259  // 1st: save cb_label (into mydata) and store cs_label for each example, which will be passed into base.learn.
260  // also save offsets
261  uint64_t saved_offset = examples[0]->ft_offset;
262  size_t index = 0;
263  for (auto ec : examples)
264  {
265  cb_labels.push_back(ec->l.cb);
266  prepped_cs_labels[index].costs.clear();
267  prepped_cs_labels[index].costs.push_back(cs_labels.costs[index]);
268  ec->l.cs = prepped_cs_labels[index++];
269  ec->ft_offset = offset;
270  }
271 
272  // 2nd: predict for each ex
273  // // call base.predict for all examples
274  if (is_learn)
275  base.learn(examples, (int32_t)id);
276  else
277  base.predict(examples, (int32_t)id);
278 
279  // 3rd: restore cb_label for each example
280  // (**ec).l.cb = array.element.
281  // and restore offsets
282  for (size_t i = 0; i < examples.size(); ++i)
283  {
284  examples[i]->l.cb = cb_labels[i];
285  examples[i]->ft_offset = saved_offset;
286  }
