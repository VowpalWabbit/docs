32 size_t really_read(int sock, void* in, size_t count)
33 {
34  char* buf = (char*)in;
35  size_t done = 0;
36  int r = 0;
37  while (done < count)
38  {
39  if ((r =
40 #ifdef _WIN32
41  recv(sock, buf, (unsigned int)(count - done), 0)
42 #else
43  read(sock, buf, (unsigned int)(count - done))
44 #endif
45  ) == 0)
46  return 0;
47  else if (r < 0)
48  {
49  THROWERRNO("read(" << sock << "," << count << "-" << done << ")");
50  }
51  else
52  {
53  done += r;
54  buf += r;
55  }
56  }
57  return done;
58 }
