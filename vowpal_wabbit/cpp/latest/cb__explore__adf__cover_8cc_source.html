67 void cb_explore_adf_cover::predict_or_learn_impl(LEARNER::multi_learner& base, multi_ex& examples)
68 {
69  // Redundant with the call in cb_explore_adf_base, but encapsulation means we need to do this again here
70  _gen_cs.known_cost = CB_ADF::get_observed_cost(examples);
71 
72  // Randomize over predictions from a base set of predictors
73  // Use cost sensitive oracle to cover actions to form distribution.
74  const bool is_mtr = _gen_cs.cb_type == CB_TYPE_MTR;
75  if (is_learn)
76  {
77  if (is_mtr) // use DR estimates for non-ERM policies in MTR
78  GEN_CS::gen_cs_example_dr<true>(_gen_cs, examples, _cs_labels);
79  else
80  GEN_CS::gen_cs_example<false>(_gen_cs, examples, _cs_labels);
81  LEARNER::multiline_learn_or_predict<true>(base, examples, examples[0]->ft_offset);
82  }
83  else
84  {
85  GEN_CS::gen_cs_example_ips(examples, _cs_labels);
86  LEARNER::multiline_learn_or_predict<false>(base, examples, examples[0]->ft_offset);
87  }
88  v_array<ACTION_SCORE::action_score>& preds = examples[0]->pred.a_s;
89  const uint32_t num_actions = (uint32_t)preds.size();
90 
91  float additive_probability = 1.f / (float)_cover_size;
92  const float min_prob = (std::min)(1.f / num_actions, 1.f / (float)std::sqrt(_counter * num_actions));
93  _action_probs.clear();
94  for (uint32_t i = 0; i < num_actions; i++) _action_probs.push_back({i, 0.});
95  _scores.clear();
96  for (uint32_t i = 0; i < num_actions; i++) _scores.push_back(preds[i].score);
97 
98  if (!_first_only)
99  {
100  size_t tied_actions = fill_tied(preds);
101  for (size_t i = 0; i < tied_actions; ++i)
102  _action_probs[preds[i].action].score += additive_probability / tied_actions;
103  }
104  else
105  _action_probs[preds[0].action].score += additive_probability;
106 
107  float norm = min_prob * num_actions + (additive_probability - min_prob);
108  for (size_t i = 1; i < _cover_size; i++)
109  {
110  // Create costs of each action based on online cover
111  if (is_learn)
112  {
113  _cs_labels_2.costs.clear();
114  /* Cover's learn policy is similar to bag in that we have multiple learners
115  * The main difference here is that Cover's learners interact with each other.
116  * The following code generates cost = cost + penalty where a penalty is applied
117  * to any action a previous policy has already selected
118  */
119  for (uint32_t j = 0; j < num_actions; j++)
120  {
121  float pseudo_cost =
122  _cs_labels.costs[j].x - _psi * min_prob / ((std::max)(_action_probs[j].score, min_prob) / norm);
123  _cs_labels_2.costs.push_back({pseudo_cost, j, 0., 0.});
124  }
125  GEN_CS::call_cs_ldf<true>(
126  *(_cs_ldf_learner), examples, _cb_labels, _cs_labels_2, _prepped_cs_labels, examples[0]->ft_offset, i + 1);
127  }
128  else
129  GEN_CS::call_cs_ldf<false>(
130  *(_cs_ldf_learner), examples, _cb_labels, _cs_labels, _prepped_cs_labels, examples[0]->ft_offset, i + 1);
131 
132  for (uint32_t i = 0; i < num_actions; i++) _scores[i] += preds[i].score;
133  if (!_first_only)
134  {
135  size_t tied_actions = fill_tied(preds);
136  const float add_prob = additive_probability / tied_actions;
137  for (size_t i = 0; i < tied_actions; ++i)
138  {
139  if (_action_probs[preds[i].action].score < min_prob)
140  norm += (std::max)(0.f, add_prob - (min_prob - _action_probs[preds[i].action].score));
141  else
142  norm += add_prob;
143  _action_probs[preds[i].action].score += add_prob;
144  }
145  }
146  else
147  {
148  uint32_t action = preds[0].action;
149  if (_action_probs[action].score < min_prob)
150  norm += (std::max)(0.f, additive_probability - (min_prob - _action_probs[action].score));
151  else
152  norm += additive_probability;
153  _action_probs[action].score += additive_probability;
154  }
155  }
156 
157  exploration::enforce_minimum_probability(
158  min_prob * num_actions, !_nounif, begin_scores(_action_probs), end_scores(_action_probs));
159 
160  sort_action_probs(_action_probs, _scores);
161  for (size_t i = 0; i < num_actions; i++) preds[i] = _action_probs[i];
162 
163  if (is_learn)
164  ++_counter;
165 }

178 LEARNER::base_learner* setup(config::options_i& options, vw& all)
179 {
180  using config::make_option;
181 
182  bool cb_explore_adf_option = false;
183  std::string type_string = "mtr";
184  size_t cover_size = 0;
185  float psi = 0.;
186  bool nounif = false;
187  bool first_only = false;
188 
189  config::option_group_definition new_options("Contextual Bandit Exploration with Action Dependent Features");
190  new_options
191  .add(make_option("cb_explore_adf", cb_explore_adf_option)
192  .keep()
193  .help("Online explore-exploit for a contextual bandit problem with multiline action dependent features"))
194  .add(make_option("cover", cover_size).keep().help("Online cover based exploration"))
195  .add(make_option("psi", psi).keep().default_value(1.0f).help("disagreement parameter for cover"))
196  .add(make_option("nounif", nounif).keep().help("do not explore uniformly on zero-probability actions in cover"))
197  .add(make_option("first_only", first_only).keep().help("Only explore the first action in a tie-breaking event"))
198  .add(make_option("cb_type", type_string)
199  .keep()
200  .help("contextual bandit method to use in {ips,dr,mtr}. Default: mtr"));
201  options.add_and_parse(new_options);
202 
203  if (!cb_explore_adf_option || !options.was_supplied("cover"))
204  return nullptr;
205 
206  // Ensure serialization of cb_type in all cases.
207  if (!options.was_supplied("cb_type"))
208  {
209  options.insert("cb_type", type_string);
210  options.add_and_parse(new_options);
211  }
212 
213  // Ensure serialization of cb_adf in all cases.
214  if (!options.was_supplied("cb_adf"))
215  {
216  options.insert("cb_adf", "");
217  }
218 
219  all.delete_prediction = ACTION_SCORE::delete_action_scores;
220 
221  // Set cb_type
222  size_t cb_type_enum;
223  if (type_string.compare("dr") == 0)
224  cb_type_enum = CB_TYPE_DR;
225  else if (type_string.compare("ips") == 0)
226  cb_type_enum = CB_TYPE_IPS;
227  else if (type_string.compare("mtr") == 0)
228  {
229  all.trace_message << "warning: currently, mtr is only used for the first policy in cover, other policies use dr"
230  << std::endl;
231  cb_type_enum = CB_TYPE_MTR;
232  }
233  else
234  {
235  all.trace_message << "warning: cb_type must be in {'ips','dr','mtr'}; resetting to mtr." << std::endl;
236  options.replace("cb_type", "mtr");
237  cb_type_enum = CB_TYPE_MTR;
238  }
239 
240  // Set explore_type
241  size_t problem_multiplier = cover_size + 1;
242 
243  LEARNER::multi_learner* base = LEARNER::as_multiline(setup_base(options, all));
244  all.p->lp = CB::cb_label;
245  all.label_type = label_type_t::cb;
246 
247  using explore_type = cb_explore_adf_base<cb_explore_adf_cover>;
248  auto data = scoped_calloc_or_throw<explore_type>(
249  cover_size, psi, nounif, first_only, as_multiline(all.cost_sensitive), all.scorer, cb_type_enum);
250 
251  LEARNER::learner<explore_type, multi_ex>& l = init_learner(
252  data, base, explore_type::learn, explore_type::predict, problem_multiplier, prediction_type_t::action_probs);
253 
254  l.set_finish_example(explore_type::finish_multiline_example);
255  return make_base(l);
256 }
