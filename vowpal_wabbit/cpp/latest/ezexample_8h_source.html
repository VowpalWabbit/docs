160  void remns()
161  {
162  if (ec->indices.empty())
163  {
164  current_seed = 0;
165  current_ns = 0;
166  }
167  else
168  {
169  if (ns_exists[(int)current_ns])
170  {
171  ec->total_sum_feat_sq -= ec->feature_space[(int)current_ns].sum_feat_sq;
172  ec->feature_space[(int)current_ns].clear();
173  ec->num_features -= ec->feature_space[(int)current_ns].size();
174 
175  ns_exists[(int)current_ns] = false;
176  }
177 
178  current_seed = past_seeds.back();
179  past_seeds.pop_back();
180  ec->indices.pop();
181  example_changed_since_prediction = true;
182  }
