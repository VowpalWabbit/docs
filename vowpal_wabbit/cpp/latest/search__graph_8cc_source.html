11 /*
12 example format:
13 
14 ALL NODES
15 ALL EDGES
16 <blank>
17 ALL NODES
18 ALL EDGES
19 <blank>
20 
21 and so on
22 
23 node lines look like normal vw examples with unary features:
24 
25 label:weight |n features
26 label:weight |n features
27 ...
28 
29 they are *implicitly* labeled starting at 1. (note the namespace
30 needn't be called n.) if weight is omitted it is assumed to be 1.0.
31 
32 edge lines look like:
33 
34 n1 n2 n3 ... |e features
35 n1 n2 n3 ... |e features
36 ...
37 
38 here, n1 n2 n3 are integer node ids, starting at one. technically
39 these are hyperedges, since they can touch more than two nodes. in the
40 canonical representation, there will just be n1 and n2.
41 
42 the only thing that differentiates edges from nodes is that edges have
43 >1 input.
44 
45 edges can either be directed or undirected. the default is undirected,
46 in which case the order of nodes listed in the edge is irrelevant. if
47 you add --search_graph_directed then the edges will be interpreted as
48 directed edges. for simple edges that connect two nodes, such as "n1
49 n2 | ...", the direction will be n1-->n2. for hyperedges of size K
50 such as "n1 n2 n3 n4 | ...", the default assumption is that the edge
51 has K-1 sources and 1 sink, so {n1,n2,n3}-->n4 in this case. if you
52 desire hyper-edges with more than one sink, you can specify them by
53 separating the source nodes from the sink nodes with a "0". for
54 instance "n1 n2 n3 0 n4 | ..." is the same as "n1 n2 n3 n4 | ...", but
55 "n1 n2 0 n3 n4 | ..." means {n1,n2}-->{n3,n4}. in undirected mode, all
56 0s are ignored; in directed mode, all but the first are ignored.
57 */

145 void run_bfs(task_data& D, multi_ex& ec)
146 {
147  D.bfs.clear();
148  std::vector<bool> touched;
149  for (size_t n = 0; n < D.N; n++) touched.push_back(false);
150 
151  touched[0] = true;
152  D.bfs.push_back(0);
153 
154  size_t i = 0;
155  while (D.bfs.size() < D.N)
156  {
157  while (i < D.bfs.size())
158  {
159  uint32_t n = D.bfs[i];
160  for (size_t id : D.adj[n])
161  for (size_t j = 0; j < ec[id]->l.cs.costs.size(); j++)
162  {
163  uint32_t m = ec[id]->l.cs.costs[j].class_index;
164  if ((m > 0) && (!touched[m - 1]))
165  {
166  D.bfs.push_back(m - 1);
167  touched[m - 1] = true;
168  }
169  }
170  i++;
171  }
172 
173  if (D.bfs.size() < D.N)
174  // we finished a SCC, need to find another
175  for (uint32_t n = 0; n < D.N; n++)
176  if (!touched[n])
177  {
178  touched[n] = true;
179  D.bfs.push_back(n);
180  break;
181  }
182  }
183 }

185 void setup(Search::search& sch, multi_ex& ec)
186 {
187  task_data& D = *sch.get_task_data<task_data>();
188  D.multiplier = D.wpp << D.ss;
189  D.wpp = sch.get_vw_pointer_unsafe().wpp;
190  D.mask = sch.get_vw_pointer_unsafe().weights.mask();
191  D.ss = sch.get_vw_pointer_unsafe().weights.stride_shift();
192  D.N = 0;
193  D.E = 0;
194  for (size_t i = 0; i < ec.size(); i++)
195  if (example_is_edge(ec[i]))
196  D.E++;
197  else // it's a node!
198  {
199  if (D.E > 0)
200  THROW("error: got a node after getting edges!");
201 
202  D.N++;
203  if (ec[i]->l.cs.costs.size() > 0)
204  {
205  D.true_counts[ec[i]->l.cs.costs[0].class_index] += 1.;
206  D.true_counts_total += 1.;
207  }
208  }
209 
210  if ((D.N == 0) && (D.E > 0))
211  THROW("error: got edges without any nodes (perhaps ring_size is too small?)!");
212 
213  D.adj = std::vector<std::vector<size_t>>(D.N, std::vector<size_t>(0));
214 
215  for (size_t i = D.N; i < ec.size(); i++)
216  {
217  for (size_t n = 0; n < ec[i]->l.cs.costs.size(); n++)
218  {
219  if (ec[i]->l.cs.costs[n].class_index > D.N)
220  THROW("error: edge source points to too large of a node id: " << (ec[i]->l.cs.costs[n].class_index) << " > "
221  << D.N);
222  }
223  for (size_t n = 0; n < ec[i]->l.cs.costs.size(); n++)
224  {
225  size_t nn = ec[i]->l.cs.costs[n].class_index;
226  if ((nn > 0) &&
227  (((D.adj[nn - 1].size() == 0) || (D.adj[nn - 1][D.adj[nn - 1].size() - 1] != i)))) // don't allow dups
228  D.adj[nn - 1].push_back(i);
229  }
230  }
231 
232  run_bfs(D, ec);
233 
234  D.pred.clear();
235  for (size_t n = 0; n < D.N; n++) D.pred.push_back(D.K + 1);
236 }

269 void add_edge_features(Search::search& sch, task_data& D, size_t n, multi_ex& ec)
270 {
271  D.cur_node = ec[n];
272 
273  for (size_t i : D.adj[n])
274  {
275  for (size_t k = 0; k < D.numN; k++) D.neighbor_predictions[k] = 0.;
276 
277  float pred_total = 0.;
278  uint32_t last_pred = 0;
279  if (D.use_structure)
280  {
281  bool n_in_sink = true;
282  if (D.directed)
283  for (size_t j = 0; j < ec[i]->l.cs.costs.size() - 1; j++)
284  {
285  size_t m = ec[i]->l.cs.costs[j].class_index;
286  if (m == 0)
287  break;
288  if (m - 1 == n)
289  {
290  n_in_sink = false;
291  break;
292  }
293  }
294 
295  bool m_in_sink = false;
296  for (size_t j = 0; j < ec[i]->l.cs.costs.size(); j++)
297  {
298  size_t m = ec[i]->l.cs.costs[j].class_index;
299  if (m == 0)
300  {
301  m_in_sink = true;
302  continue;
303  }
304  if (j == ec[i]->l.cs.costs.size() - 1)
305  m_in_sink = true;
306  m--;
307  if (m == n)
308  continue;
309  size_t other_side = (D.directed && (n_in_sink != m_in_sink)) ? (D.K + 1) : 0;
310  D.neighbor_predictions[D.pred[m] - 1 + other_side] += 1.;
311  pred_total += 1.;
312  last_pred = (uint32_t)D.pred[m] - 1 + (uint32_t)other_side;
313  }
314  }
315  else
316  {
317  D.neighbor_predictions[0] += 1.;
318  pred_total += 1.;
319  last_pred = 0;
320  }
321 
322  if (pred_total == 0.)
323  continue;
324  // std::cerr << n << ':' << i << " -> ["; for (size_t k=0; k<D.numN; k++) std::cerr << ' ' <<
325  // D.neighbor_predictions[k]; std::cerr
326  // << " ]" << std::endl;
327  for (size_t k = 0; k < D.numN; k++) D.neighbor_predictions[k] /= pred_total;
328  example& edge = *ec[i];
329 
330  if (pred_total <= 1.) // single edge
331  {
332  D.neighbor_predictions[0] = (float)last_pred;
333  GD::foreach_feature<task_data, uint64_t, add_edge_features_single_fn>(sch.get_vw_pointer_unsafe(), edge, D);
334  }
335  else // lots of edges
336  GD::foreach_feature<task_data, uint64_t, add_edge_features_group_fn>(sch.get_vw_pointer_unsafe(), edge, D);
337  }
338  ec[n]->indices.push_back(neighbor_namespace);
339  ec[n]->total_sum_feat_sq += ec[n]->feature_space[neighbor_namespace].sum_feat_sq;
340  ec[n]->num_features += ec[n]->feature_space[neighbor_namespace].size();
341 
342  vw& all = sch.get_vw_pointer_unsafe();
343  for (std::string& i : all.pairs)
344  {
345  int i0 = (int)i[0];
346  int i1 = (int)i[1];
347  if ((i0 == (int)neighbor_namespace) || (i1 == (int)neighbor_namespace))
348  {
349  ec[n]->num_features += ec[n]->feature_space[i0].size() * ec[n]->feature_space[i1].size();
350  ec[n]->total_sum_feat_sq += ec[n]->feature_space[i0].sum_feat_sq * ec[n]->feature_space[i1].sum_feat_sq;
351  }
352  }
353 }

366 float macro_f(task_data& D)
367 {
368  float total_f1 = 0.;
369  float count_f1 = 0.;
370  for (size_t k = 1; k <= D.K; k++)
371  {
372  float trueC = 0.;
373  float predC = 0.;
374  for (size_t j = 1; j <= D.K; j++)
375  {
376  trueC += (float)D.confusion_matrix[IDX(k, j)];
377  predC += (float)D.confusion_matrix[IDX(j, k)];
378  }
379  if (trueC == 0)
380  continue;
381  float correctC = (float)D.confusion_matrix[IDX(k, k)];
382  count_f1++;
383  if (correctC > 0)
384  {
385  float pre = correctC / predC;
386  float rec = correctC / trueC;
387  total_f1 += 2 * pre * rec / (pre + rec);
388  }
389  }
390  return total_f1 / count_f1;
391 }

393 void run(Search::search& sch, multi_ex& ec)
394 {
395  task_data& D = *sch.get_task_data<task_data>();
396  float loss_val = 0.5f / (float)D.num_loops;
397  for (size_t n = 0; n < D.N; n++) D.pred[n] = D.K + 1;
398 
399  for (size_t loop = 0; loop < D.num_loops; loop++)
400  {
401  bool last_loop = loop == (D.num_loops - 1);
402  int start = 0;
403  int end = D.N;
404  int step = 1;
405  if (loop % 2 == 1)
406  {
407  start = D.N - 1;
408  end = -1;
409  step = -1;
410  } // go inward on odd loops
411  for (int n_id = start; n_id != end; n_id += step)
412  {
413  uint32_t n = D.bfs[n_id];
414  uint32_t k = (ec[n]->l.cs.costs.size() > 0) ? ec[n]->l.cs.costs[0].class_index : 0;
415 
416  bool add_features = /* D.use_structure && */ sch.predictNeedsExample();
417  // add_features = false;
418 
419  if (add_features)
420  add_edge_features(sch, D, n, ec);
421  Search::predictor P = Search::predictor(sch, n + 1);
422  P.set_input(*ec[n]);
423  if (false && (k > 0))
424  {
425  float min_count = 1e12f;
426  for (size_t k2 = 1; k2 <= D.K; k2++) min_count = std::min(min_count, D.true_counts[k2]);
427  float w = min_count / D.true_counts[k];
428  // float w = D.true_counts_total / D.true_counts[k] / (float)(D.K);
429  P.set_weight(w);
430  // std::cerr << "w = " << D.true_counts_total / D.true_counts[k] / (float)(D.K) << std::endl;
431  // P.set_weight( D.true_counts_total / D.true_counts[k] / (float)(D.K) );
432  }
433  if (D.separate_learners)
434  P.set_learner_id(loop);
435  if (k > 0) // for test examples
436  P.set_oracle(k);
437  // add all the conditioning
438  for (size_t i = 0; i < D.adj[n].size(); i++)
439  {
440  for (size_t j = 0; j < ec[i]->l.cs.costs.size(); j++)
441  {
442  uint32_t m = ec[i]->l.cs.costs[j].class_index;
443  if (m == 0)
444  continue;
445  m--;
446  if (m == n)
447  continue;
448  P.add_condition(m + 1, 'e');
449  }
450  }
451 
452  // make the prediction
453  D.pred[n] = P.predict();
454  if (ec[n]->l.cs.costs.size() > 0) // for test examples
455  sch.loss((ec[n]->l.cs.costs[0].class_index == D.pred[n]) ? 0.f : (last_loop ? 0.5f : loss_val));
456 
457  if (add_features)
458  del_edge_features(D, n, ec);
459  }
460  }
461 
462  for (uint32_t n = 0; n < D.N; n++) D.confusion_matrix[IDX(ec[n]->l.cs.costs[0].class_index, D.pred[n])]++;
463  sch.loss(1.f - macro_f(D));
464 
465  if (sch.output().good())
466  for (uint32_t n = 0; n < D.N; n++) sch.output() << D.pred[n] << ' ';
467 }
