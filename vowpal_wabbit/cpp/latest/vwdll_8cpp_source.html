16 // This interface now provides "wide" functions for compatibility with .NET interop
17 // The default functions assume a wide (16 bit char pointer) that is converted to a utf8-string and passed to
18 // a function which takes a narrow (8 bit char pointer) function. Both are exposed in the c/c++ API
19 // so that programs using 8 bit wide characters can use the direct call without conversion and
20 // programs using 16 bit characters can use the default wide versions of the functions.
21 // "Ansi versions (FcnA instead of Fcn) have only been written for functions which handle strings.
22 
23 // a future optimization would be to write an inner version of hash feature which either hashed the
24 // wide string directly (and live with the different hash values) or incorporate the UTF-16 to UTF-8 conversion
25 // in the hashing to avoid allocating an intermediate string.

353 class memory_io_buf : public io_buf
354 {
355 public:
356  memory_io_buf() : readOffset(0) {
357  files.push_back(-1); // this is a hack because buf will do nothing if files is empty
358  }
359 
360  virtual ssize_t write_file(int file, const void* buf, size_t nbytes) {
361  auto byteBuf = reinterpret_cast<const char*>(buf);
362  data.insert(data.end(), &byteBuf[0], &byteBuf[nbytes]);
363  return nbytes;
364  }
365 
366  virtual ssize_t read_file(int f, void* buf, size_t nbytes) {
367  nbytes = min(nbytes, data.size()-readOffset);
368  copy(data.data()+readOffset, data.data()+readOffset+nbytes, reinterpret_cast<char *>(buf));
369  readOffset += nbytes;
370  return nbytes;
371  }
372 
373  virtual bool close_file() {
374  if (files.size() > 0) {
375  files.pop();
376  return true;
377  }
378  return false;
379  }
380 
381  char* GetDataPointer() {
382  return data.data();
383  }
384 
385  size_t GetDataSize() const {
386  return data.size();
387  }
388 
389 private:
390  vector<char> data;
391  size_t readOffset;
392 };
