62 void value_policy(mwt& c, float val, uint64_t index) // estimate the value of a single feature.
63 {
64  if (val < 0 || floor(val) != val)
65  std::cout << "error " << val << " is not a valid action " << std::endl;
66 
67  uint32_t value = (uint32_t)val;
68  uint64_t new_index = (index & c.all->weights.mask()) >> c.all->weights.stride_shift();
69 
70  if (!c.evals[new_index].seen)
71  {
72  c.evals[new_index].seen = true;
73  c.policies.push_back(new_index);
74  }
75 
76  c.evals[new_index].action = value;
77 }

79 template <bool learn, bool exclude, bool is_learn>
80 void predict_or_learn(mwt& c, single_learner& base, example& ec)
81 {
82  c.observation = get_observed_cost(ec.l.cb);
83 
84  if (c.observation != nullptr)
85  {
86  c.total++;
87  // For each nonzero feature in observed namespaces, check it's value.
88  for (unsigned char ns : ec.indices)
89  if (c.namespaces[ns])
90  GD::foreach_feature<mwt, value_policy>(c.all, ec.feature_space[ns], c);
91  for (uint64_t policy : c.policies)
92  {
93  c.evals[policy].cost += get_cost_estimate(c.observation, c.evals[policy].action);
94  c.evals[policy].action = 0;
95  }
96  }
97  if (exclude || learn)
98  {
99  c.indices.clear();
100  uint32_t stride_shift = c.all->weights.stride_shift();
101  uint64_t weight_mask = c.all->weights.mask();
102  for (unsigned char ns : ec.indices)
103  if (c.namespaces[ns])
104  {
105  c.indices.push_back(ns);
106  if (learn)
107  {
108  c.feature_space[ns].clear();
109  for (features::iterator& f : ec.feature_space[ns])
110  {
111  uint64_t new_index = ((f.index() & weight_mask) >> stride_shift) * c.num_classes + (uint64_t)f.value();
112  c.feature_space[ns].push_back(1, new_index << stride_shift);
113  }
114  }
115  std::swap(c.feature_space[ns], ec.feature_space[ns]);
116  }
117  }
118 
119  // modify the predictions to use a vector with a score for each evaluated feature.
120  v_array<float> preds = ec.pred.scalars;
121 
122  if (learn)
123  {
124  if (is_learn)
125  base.learn(ec);
126  else
127  base.predict(ec);
128  }
129 
130  if (exclude || learn)
131  while (!c.indices.empty())
132  {
133  unsigned char ns = c.indices.pop();
134  std::swap(c.feature_space[ns], ec.feature_space[ns]);
135  }
136 
137  // modify the predictions to use a vector with a score for each evaluated feature.
138  preds.clear();
139  if (learn)
140  preds.push_back((float)ec.pred.multiclass);
141  for (uint64_t index : c.policies) preds.push_back((float)c.evals[index].cost / (float)c.total);
142 
143  ec.pred.scalars = preds;
144 }

172 void finish_example(vw& all, mwt& c, example& ec)

177  loss = get_cost_estimate(c.observation, (uint32_t)ec.pred.scalars[0]);
178  all.sd->update(ec.test_only, c.observation != nullptr, loss, 1.f, ec.num_features);
179 
180  for (int sink : all.final_prediction_sink) print_scalars(sink, ec.pred.scalars, ec.tag);
181 
182  if (c.learn)

184  v_array<float> temp = ec.pred.scalars;
185  ec.pred.multiclass = (uint32_t)temp[0];

187  ec.pred.scalars = temp;
188  }
189  VW::finish_example(all, ec);
