61 inline void predict_with_confidence(uncertainty& d, const float fx, float& fw)
62 {
63  float* w = &fw;
64  d.pred += w[W_XT] * fx;
65  float sqrtf_ng2 = sqrtf(w[W_G2]);
66  float uncertain = ((d.b.data.ftrl_beta + sqrtf_ng2) / d.b.data.ftrl_alpha + d.b.data.l2_lambda);
67  d.score += (1 / uncertain) * sign(fx);
68 }

118 void inner_update_proximal(update_data& d, float x, float& wref)
119 {
120  float* w = &wref;
121  float gradient = d.update * x;
122  float ng2 = w[W_G2] + gradient * gradient;
123  float sqrt_ng2 = sqrtf(ng2);
124  float sqrt_wW_G2 = sqrtf(w[W_G2]);
125  float sigma = (sqrt_ng2 - sqrt_wW_G2) / d.ftrl_alpha;
126  w[W_ZT] += gradient - sigma * w[W_XT];
127  w[W_G2] = ng2;
128  sqrt_wW_G2 = sqrt_ng2;
129  float flag = sign(w[W_ZT]);
130  float fabs_zt = w[W_ZT] * flag;
131  if (fabs_zt <= d.l1_lambda)
132  w[W_XT] = 0.;
133  else
134  {
135  float step = 1 / (d.l2_lambda + (d.ftrl_beta + sqrt_wW_G2) / d.ftrl_alpha);
136  w[W_XT] = step * flag * (d.l1_lambda - fabs_zt);
137  }
138 }

140 void inner_update_pistol_state_and_predict(update_data& d, float x, float& wref)
141 {
142  float* w = &wref;
143 
144  float fabs_x = fabs(x);
145  if (fabs_x > w[W_MX])
146  w[W_MX] = fabs_x;
147 
148  float squared_theta = w[W_ZT] * w[W_ZT];
149  float tmp = 1.f / (d.ftrl_alpha * w[W_MX] * (w[W_G2] + w[W_MX]));
150  w[W_XT] = std::sqrt(w[W_G2]) * d.ftrl_beta * w[W_ZT] * correctedExp(squared_theta / 2.f * tmp) * tmp;
151 
152  d.predict += w[W_XT] * x;
153 }

155 void inner_update_pistol_post(update_data& d, float x, float& wref)
156 {
157  float* w = &wref;
158  float gradient = d.update * x;
159 
160  w[W_ZT] += -gradient;
161  w[W_G2] += fabs(gradient);
162 }

164 // Coin betting vectors
165 // W_XT 0 current parameter
166 // W_ZT 1 sum negative gradients
167 // W_G2 2 sum of absolute value of gradients
168 // W_MX 3 maximum absolute value
169 // W_WE 4 Wealth
170 // W_MG 5 Maximum Lipschitz constant
171 void inner_update_cb_state_and_predict(update_data& d, float x, float& wref)
172 {
173  float* w = &wref;
174  float w_mx = w[W_MX];
175  float w_xt = 0.0;
176 
177  float fabs_x = fabs(x);
178  if (fabs_x > w_mx)
179  {
180  w_mx = fabs_x;
181  }
182 
183  // COCOB update without sigmoid
184  if (w[W_MG] * w_mx > 0)
185  w_xt = ((d.ftrl_alpha + w[W_WE]) / (w[W_MG] * w_mx * (w[W_MG] * w_mx + w[W_G2]))) * w[W_ZT];
186 
187  d.predict += w_xt * x;
188  if (w_mx > 0)
189  d.normalized_squared_norm_x += x * x / (w_mx * w_mx);
190 }

192 void inner_update_cb_post(update_data& d, float x, float& wref)
193 {
194  float* w = &wref;
195  float fabs_x = fabs(x);
196  float gradient = d.update * x;
197 
198  if (fabs_x > w[W_MX])
199  {
200  w[W_MX] = fabs_x;
201  }
202 
203  float fabs_gradient = fabs(d.update);
204  if (fabs_gradient > w[W_MG])
205  w[W_MG] = fabs_gradient > d.ftrl_beta ? fabs_gradient : d.ftrl_beta;
206 
207  // COCOB update without sigmoid.
208  // If a new Lipschitz constant and/or magnitude of x is found, the w is
209  // recalculated and used in the update of the wealth below.
210  if (w[W_MG] * w[W_MX] > 0)
211  w[W_XT] = ((d.ftrl_alpha + w[W_WE]) / (w[W_MG] * w[W_MX] * (w[W_MG] * w[W_MX] + w[W_G2]))) * w[W_ZT];
212  else
213  w[W_XT] = 0;
214 
215  w[W_ZT] += -gradient;
216  w[W_G2] += fabs(gradient);
217  w[W_WE] += (-gradient * w[W_XT]);
218 }
