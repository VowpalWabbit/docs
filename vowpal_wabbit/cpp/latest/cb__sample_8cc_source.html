24  template <bool is_learn>
25  inline void learn_or_predict(multi_learner &base, multi_ex &examples)
26  {
27  multiline_learn_or_predict<is_learn>(base, examples, examples[0]->ft_offset);
28 
29  auto action_scores = examples[0]->pred.a_s;
30 
31  uint32_t chosen_action = 0;
32  int64_t maybe_labelled_action = -1;
33 
34  // Find that chosen action in the learning case, skip the shared example.
35  auto it = std::find_if(examples.begin(), examples.end(), [](example *item) { return !item->l.cb.costs.empty(); });
36  if (it != examples.end())
37  {
38  maybe_labelled_action = static_cast<int64_t>(std::distance(examples.begin(), it));
39  }
40 
41  // If we are learning and have a label, then take that action as the chosen action. Otherwise sample the
42  // distribution.
43  if (is_learn && maybe_labelled_action >= 0)
44  {
45  // Find where the labelled action is in the final prediction to determine if swapping needs to occur.
46  // This only matters if the prediction decided to explore, but the same output should happen for the learn case.
47  for (size_t i = 0; i < action_scores.size(); i++)
48  {
49  auto &a_s = action_scores[i];
50  if (a_s.action == static_cast<uint32_t>(maybe_labelled_action))
51  {
52  chosen_action = static_cast<uint32_t>(i);
53  break;
54  }
55  }
56  }
57  else
58  {
59  bool tag_provided_seed = false;
60  uint64_t seed = _random_state->get_current_state();
61  if (!examples[0]->tag.empty())
62  {
63  const std::string SEED_IDENTIFIER = "seed=";
64  if (strncmp(examples[0]->tag.begin(), SEED_IDENTIFIER.c_str(), SEED_IDENTIFIER.size()) == 0 &&
65  examples[0]->tag.size() > SEED_IDENTIFIER.size())
66  {
67  VW::string_view tag_seed(examples[0]->tag.begin() + 5, examples[0]->tag.size());
68  seed = uniform_hash(tag_seed.begin(), tag_seed.size(), 0);
69  tag_provided_seed = true;
70  }
71  }
72 
73  // Sampling is done after the base learner has generated a pdf.
74  auto result = exploration::sample_after_normalizing(
75  seed, ACTION_SCORE::begin_scores(action_scores), ACTION_SCORE::end_scores(action_scores), chosen_action);
76  assert(result == S_EXPLORATION_OK);
77  _UNUSED(result);
78 
79  // Update the seed state in place if it was used for this example.
80  if (!tag_provided_seed)
81  {
82  _random_state->get_and_update_random();
83  }
84  }
85 
86  auto result = exploration::swap_chosen(action_scores.begin(), action_scores.end(), chosen_action);
87  assert(result == S_EXPLORATION_OK);
88 
89  _UNUSED(result);
90  }
91 
92  private:
93  std::shared_ptr<rand_state> _random_state;
94 };
95 } // namespace VW
