38 template <bool is_learn, label_parser& lp>
39 void predict_or_learn(expreplay<lp>& er, LEARNER::single_learner& base, example& ec)
40 { // regardless of what happens, we must predict
41  base.predict(ec);
42  // if we're not learning, that's all that has to happen
43  if (!is_learn || lp.get_weight(&ec.l) == 0.)
44  return;
45 
46  for (size_t replay = 1; replay < er.replay_count; replay++)
47  {
48  size_t n = (size_t)(er._random_state->get_and_update_random() * (float)er.N);
49  if (er.filled[n])
50  base.learn(er.buf[n]);
51  }
52 
53  size_t n = (size_t)(er._random_state->get_and_update_random() * (float)er.N);
54  if (er.filled[n])
55  base.learn(er.buf[n]);
56 
57  er.filled[n] = true;
58  VW::copy_example_data(er.all->audit, &er.buf[n], &ec); // don't copy the label
59  if (lp.copy_label)
60  lp.copy_label(&er.buf[n].l, &ec.l);
61  else
62  er.buf[n].l = ec.l;
63 }
