177 bool children(log_multi& b, uint32_t& current, uint32_t& class_index, uint32_t label)
178 {
179  class_index = (uint32_t)b.nodes[current].preds.unique_add_sorted(node_pred(label));
180  b.nodes[current].preds[class_index].label_count++;
181 
182  if (b.nodes[current].preds[class_index].label_count > b.nodes[current].max_count)
183  {
184  b.nodes[current].max_count = b.nodes[current].preds[class_index].label_count;
185  b.nodes[current].max_count_label = b.nodes[current].preds[class_index].label;
186  }
187 
188  if (b.nodes[current].internal)
189  return true;
190  else if (b.nodes[current].preds.size() > 1 &&
191  (b.predictors_used < b.max_predictors ||
192  b.nodes[current].min_count - b.nodes[current].max_count > b.swap_resist * (b.nodes[0].min_count + 1)))
193  {
194  // need children and we can make them.
195  uint32_t left_child;
196  uint32_t right_child;
197  if (b.predictors_used < b.max_predictors)
198  {
199  left_child = (uint32_t)b.nodes.size();
200  b.nodes.push_back(init_node());
201  right_child = (uint32_t)b.nodes.size();
202  b.nodes.push_back(init_node());
203  b.nodes[current].base_predictor = (uint32_t)b.predictors_used++;
204  }
205  else
206  {
207  uint32_t swap_child = find_switch_node(b);
208  uint32_t swap_parent = b.nodes[swap_child].parent;
209  uint32_t swap_grandparent = b.nodes[swap_parent].parent;
210  if (b.nodes[swap_child].min_count != b.nodes[0].min_count)
211  std::cout << "glargh " << b.nodes[swap_child].min_count << " != " << b.nodes[0].min_count << std::endl;
212  b.nbofswaps++;
213 
214  uint32_t nonswap_child;
215  if (swap_child == b.nodes[swap_parent].right)
216  nonswap_child = b.nodes[swap_parent].left;
217  else
218  nonswap_child = b.nodes[swap_parent].right;
219 
220  if (swap_parent == b.nodes[swap_grandparent].left)
221  b.nodes[swap_grandparent].left = nonswap_child;
222  else
223  b.nodes[swap_grandparent].right = nonswap_child;
224  b.nodes[nonswap_child].parent = swap_grandparent;
225  update_min_count(b, nonswap_child);
226 
227  init_leaf(b.nodes[swap_child]);
228  left_child = swap_child;
229  b.nodes[current].base_predictor = b.nodes[swap_parent].base_predictor;
230  init_leaf(b.nodes[swap_parent]);
231  right_child = swap_parent;
232  }
233  b.nodes[current].left = left_child;
234  b.nodes[left_child].parent = current;
235  b.nodes[current].right = right_child;
236  b.nodes[right_child].parent = current;
237 
238  b.nodes[left_child].min_count = b.nodes[current].min_count / 2;
239  b.nodes[right_child].min_count = b.nodes[current].min_count - b.nodes[left_child].min_count;
240  update_min_count(b, left_child);
241 
242  b.nodes[left_child].max_count_label = b.nodes[current].max_count_label;
243  b.nodes[right_child].max_count_label = b.nodes[current].max_count_label;
244 
245  b.nodes[current].internal = true;
246  }
247  return b.nodes[current].internal;
