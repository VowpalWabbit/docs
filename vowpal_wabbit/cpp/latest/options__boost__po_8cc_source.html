51 void options_boost_po::add_and_parse(const option_group_definition& group)
52 {
53  po::options_description new_options(group.m_name);
54 
55  for (auto opt_ptr : group.m_options)
56  {
57  add_to_description(opt_ptr, new_options);
58  m_defined_options.insert(opt_ptr->m_name);
59  m_defined_options.insert(opt_ptr->m_short_name);
60  m_defined_options.insert("-" + opt_ptr->m_short_name);
61 
62  // The last definition is kept. There was a bug where using .insert at a later pointer changed the command line but
63  // the previously defined option's default value was serialized into the model. This resolves that state info.
64  m_options[opt_ptr->m_name] = opt_ptr;
65  }
66 
67  // Add the help for the given options.
68  new_options.print(m_help_stringstream);
69 
70  try
71  {
72  po::variables_map vm;
73  auto parsed_options = po::command_line_parser(m_command_line)
74  .options(new_options)
75  .style(po::command_line_style::default_style ^ po::command_line_style::allow_guessing)
76  .allow_unregistered()
77  .run();
78 
79  for (auto const& option : parsed_options.options)
80  {
81  // If the supplied option is interpreted as a number, then ignore it. There are no options like this and it is
82  // just a false positive.
83  if (is_number(option.string_key))
84  {
85  m_ignore_supplied.insert(option.string_key);
86  }
87 
88  m_supplied_options.insert(option.string_key);
89 
90  // If a std::string is later determined to be a value the erase it. This happens for negative numbers "-2"
91  for (auto& val : option.value)
92  {
93  m_ignore_supplied.insert(val);
94  }
95 
96  // Parsed options can contain short options in the form -k, we can only check these as the group definitions come
97  // in.
98  if (option.string_key.length() > 0 && option.string_key[0] == '-')
99  {
100  auto short_name = option.string_key.substr(1);
101  for (const auto& opt_ptr : group.m_options)
102  {
103  if (opt_ptr->m_short_name == short_name)
104  {
105  m_supplied_options.insert(short_name);
106  }
107  }
108  }
109  }
110 
111  po::store(parsed_options, vm);
112  po::notify(vm);
113  }
114  catch (boost::exception_detail::clone_impl<
115  boost::exception_detail::error_info_injector<boost::program_options::invalid_option_value>>& ex)
116  {
117  THROW_EX(VW::vw_argument_invalid_value_exception, ex.what());
118  }
119  catch (boost::exception_detail::clone_impl<boost::exception_detail::error_info_injector<boost::bad_lexical_cast>>& ex)
120  {
121  THROW_EX(VW::vw_argument_invalid_value_exception, ex.what());
122  }
123  catch (boost::exception_detail::clone_impl<
124  boost::exception_detail::error_info_injector<boost::program_options::ambiguous_option>>& ex)
125  {
126  THROW(ex.what());
127  }
128  catch (boost::program_options::ambiguous_option& ex)
129  {
130  THROW(ex.what());
131  }
132 }
