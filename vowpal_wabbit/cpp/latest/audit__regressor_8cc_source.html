24 inline void audit_regressor_interaction(audit_regressor_data& dat, const audit_strings* f)
25 {
26  // same as audit_interaction in gd.cc

28  {

30  return;
31  }
32 
33  std::string ns_pre;
34  if (!dat.ns_pre->empty())
35  ns_pre += '*';
36 
37  if (f->first != "" && ((f->first) != " "))
38  {

40  ns_pre += '^';

44  ns_pre.append(f->second);

49 inline void audit_regressor_feature(audit_regressor_data& dat, const float, const uint64_t ft_idx)
50 {
51  parameters& weights = dat.all->weights;
52  if (weights[ft_idx] != 0)
53  ++dat.values_audited;
54  else
55  return;
56 
57  std::string ns_pre;
58  for (std::vector<std::string>::const_iterator s = dat.ns_pre->begin(); s != dat.ns_pre->end(); ++s) ns_pre += *s;
59 
60  std::ostringstream tempstream;
61  tempstream << ':' << ((ft_idx & weights.mask()) >> weights.stride_shift()) << ':' << weights[ft_idx];
62 
63  std::string temp = ns_pre + tempstream.str() + '\n';
64  if (dat.total_class_cnt > 1) // add class prefix for multiclass problems
65  temp = std::to_string(dat.cur_class) + ':' + temp;
66 
67  dat.out_file->bin_write_fixed(temp.c_str(), (uint32_t)temp.size());
68 
69  weights[ft_idx] = 0.; // mark value audited
70 }

72 void audit_regressor_lda(audit_regressor_data& rd, LEARNER::single_learner& /* base */, example& ec)
73 {
74  vw& all = *rd.all;
75 
76  std::ostringstream tempstream;
77  parameters& weights = rd.all->weights;
78  for (unsigned char* i = ec.indices.begin(); i != ec.indices.end(); i++)
79  {
80  features& fs = ec.feature_space[*i];
81  for (size_t j = 0; j < fs.size(); ++j)
82  {
83  tempstream << '\t' << fs.space_names[j].get()->first << '^' << fs.space_names[j].get()->second << ':'
84  << ((fs.indicies[j] >> weights.stride_shift()) & all.parse_mask);
85  for (size_t k = 0; k < all.lda; k++)
86  {
87  weight& w = weights[(fs.indicies[j] + k)];
88  tempstream << ':' << w;
89  w = 0.;
90  }
91  tempstream << std::endl;
92  }
93  }
94 
95  rd.out_file->bin_write_fixed(tempstream.str().c_str(), (uint32_t)tempstream.str().size());
96 }

101 void audit_regressor(audit_regressor_data& rd, LEARNER::single_learner& base, example& ec)
102 {
103  vw& all = *rd.all;
104 
105  if (all.lda > 0)
106  audit_regressor_lda(rd, base, ec);
107  else
108  {
109  rd.cur_class = 0;
110  uint64_t old_offset = ec.ft_offset;
111 
112  while (rd.cur_class < rd.total_class_cnt)
113  {
114  for (unsigned char* i = ec.indices.begin(); i != ec.indices.end(); ++i)
115  {
116  features& fs = ec.feature_space[(size_t)*i];
117  if (fs.space_names.size() > 0)
118  for (size_t j = 0; j < fs.size(); ++j)
119  {
120  audit_regressor_interaction(rd, fs.space_names[j].get());
121  audit_regressor_feature(rd, fs.values[j], (uint32_t)fs.indicies[j] + ec.ft_offset);
122  audit_regressor_interaction(rd, NULL);
123  }
124  else
125  for (size_t j = 0; j < fs.size(); ++j)
126  audit_regressor_feature(rd, fs.values[j], (uint32_t)fs.indicies[j] + ec.ft_offset);
127  }
128 
129  if (rd.all->weights.sparse)
130  INTERACTIONS::generate_interactions<audit_regressor_data, const uint64_t, audit_regressor_feature, true,
131  audit_regressor_interaction, sparse_parameters>(
132  rd.all->interactions, rd.all->permutations, ec, rd, rd.all->weights.sparse_weights);
133  else
134  INTERACTIONS::generate_interactions<audit_regressor_data, const uint64_t, audit_regressor_feature, true,
135  audit_regressor_interaction, dense_parameters>(
136  rd.all->interactions, rd.all->permutations, ec, rd, rd.all->weights.dense_weights);
137 
138  ec.ft_offset += rd.increment;
139  ++rd.cur_class;
140  }
141 
142  ec.ft_offset = old_offset; // make sure example is not changed.
143  }
144 }

198 void init_driver(audit_regressor_data& dat)
199 {
200  // checks a few settings that might be applied after audit_regressor_setup() is called
201  if ((dat.all->options->was_supplied("cache_file") || dat.all->options->was_supplied("cache")) &&
202  !dat.all->options->was_supplied("kill_cache"))
203  {
204  THROW("audit_regressor is incompatible with a cache file. Use it in single pass mode only.");
205  }
206 
207  dat.all->sd->dump_interval = 1.; // regressor could initialize these if saved with --save_resume
208  dat.all->sd->example_number = 0;
209 
210  dat.increment = dat.all->l->increment / dat.all->l->weights;
211  dat.total_class_cnt = dat.all->l->weights;
212 
213  if (dat.all->options->was_supplied("csoaa"))
214  {
215  size_t n = dat.all->options->get_typed_option<uint32_t>("csoaa").value();
216  if (n != dat.total_class_cnt)
217  {
218  dat.total_class_cnt = n;
219  dat.increment = dat.all->l->increment / n;
220  }
221  }
222 
223  // count non-null feature values in regressor
224  if (dat.all->weights.sparse)
225  regressor_values(dat, dat.all->weights.sparse_weights);
226  else
227  regressor_values(dat, dat.all->weights.dense_weights);
228 
229  if (dat.loaded_regressor_values == 0)
230  THROW("regressor has no non-zero weights. Nothing to audit.");
231 
232  if (!dat.all->logger.quiet)
233  {
234  dat.all->trace_message << "Regressor contains " << dat.loaded_regressor_values << " values\n";
235  dat.all->trace_message << std::left << std::setw(shared_data::col_example_counter) << "example"
236  << " " << std::setw(shared_data::col_example_weight) << "values"
237  << " " << std::setw(shared_data::col_current_label) << "total" << std::endl;
238  dat.all->trace_message << std::left << std::setw(shared_data::col_example_counter) << "counter"
