32 /* The i/o buffer can be conceptualized as an array below:
33 ** _______________________________________________________________________________________
34 ** |__________|__________|__________|__________|__________|__________|__________|__________| **
35 ** space.begin space.head space.end space.endarray **
36 **
37 ** space.begin = the beginning of the loaded values in the buffer
38 ** space.head = the end of the last-read point in the buffer
39 ** space.end = the end of the loaded values from file
40 ** space.endarray = the end of the allocated space for the array
41 **
42 ** The values are ordered so that:
43 ** space.begin <= space.head <= space.end <= space.endarray
44 **
45 ** Initially space.begin == space.head since no values have been read.
46 **
47 ** The interval [space.head, space.end] may be shifted down to space.begin
48 ** if the requested number of bytes to be read is larger than the interval size.
49 ** This is done to avoid reallocating arrays as much as possible.
50 */

93  switch (flag)
94  {
95  case READ:

97  {

100  _sopen_s(&ret, name, _O_RDONLY | _O_BINARY | _O_SEQUENTIAL, _SH_DENYWR, 0);
101 #else
102  ret = open(name, O_RDONLY | O_LARGEFILE);
103 #endif
104  }
105  else if (!stdin_off)
106 #ifdef _WIN32
107  ret = _fileno(stdin);
108 #else
109  ret = fileno(stdin);
110 #endif
111  if (ret != -1)
112  files.push_back(ret);
113  break;
114 
115  case WRITE:
116 #ifdef _WIN32
117  _sopen_s(&ret, name, _O_CREAT | _O_WRONLY | _O_BINARY | _O_TRUNC, _SH_DENYWR, _S_IREAD | _S_IWRITE);
118 #else
119  ret = open(name, O_CREAT | O_WRONLY | O_LARGEFILE | O_TRUNC, 0666);
120 #endif
121  if (ret != -1)
122  files.push_back(ret);
123  break;
124 
125  default:
126  std::cerr << "Unknown file operation. Something other than READ/WRITE specified" << std::endl;
127  ret = -1;
128  }
129  if (ret == -1 && *name != '\0')
130  THROWERRNO("can't open: " << name);
131  return ret;
132  }

169  ssize_t fill(int f)
170  { // if the loaded values have reached the allocated space
171  if (space.end_array - space.end() == 0)
172  { // reallocate to twice as much space
173  size_t head_loc = head - space.begin();
174  space.resize(2 * (space.end_array - space.begin()));
175  head = space.begin() + head_loc;
176  }
177  // read more bytes from file up to the remaining allocated space
178  ssize_t num_read = read_file(f, space.end(), space.end_array - space.end());
179  if (num_read >= 0)
180  { // if some bytes were actually loaded, update the end of loaded values
181  space.end() = space.end() + num_read;
182  return num_read;
183  }
184  else
185  return 0;
186  }

227  // if read_message is null, just read it in. Otherwise do a comparison and barf on read_message.
228  size_t bin_read_fixed(char* data, size_t len, const char* read_message)
229  {
230  if (len > 0)
231  {
232  char* p;
233  // if the model is corrupt the number of bytes can be less then specified (as there isn't enought data available
234  // in the file)
235  len = buf_read(p, len);
236 
237  // compute hash for check-sum
238  if (_verify_hash)
239  _hash = (uint32_t)uniform_hash(p, len, _hash);
240 
241  if (*read_message == '\0')
242  memcpy(data, p, len);
243  else if (memcmp(data, p, len) != 0)
244  THROW(read_message);
245  return len;
246  }
247  return 0;
248  }
