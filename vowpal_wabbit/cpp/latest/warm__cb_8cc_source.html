172  const uint64_t mask = data.all->weights.mask();
173 
174  for (size_t a = 0; a < data.num_actions; ++a)
175  {
176  auto& eca = *data.ecs[a];

206 void setup_lambdas(warm_cb& data)
207 {
208  // The lambdas are arranged in ascending order
209  std::vector<float>& lambdas = data.lambdas;
210  for (uint32_t i = 0; i < data.choices_lambda; i++) lambdas.push_back(0.f);
211 
212  // interaction only: set all lambda's to be identically 1
213  if (!data.upd_ws && data.upd_inter)
214  {
215  for (uint32_t i = 0; i < data.choices_lambda; i++) lambdas[i] = 1.0;
216  return;
217  }
218 
219  // warm start only: set all lambda's to be identically 0
220  if (!data.upd_inter && data.upd_ws)
221  {
222  for (uint32_t i = 0; i < data.choices_lambda; i++) lambdas[i] = 0.0;
223  return;
224  }
225 
226  uint32_t mid = data.choices_lambda / 2;
227 
228  if (data.lambda_scheme == ABS_CENTRAL || data.lambda_scheme == ABS_CENTRAL_ZEROONE)
229  lambdas[mid] = 0.5;
230  else
231  lambdas[mid] = minimax_lambda(data.epsilon);
232 
233  for (uint32_t i = mid; i > 0; i--) lambdas[i - 1] = lambdas[i] / 2.0f;
234 
235  for (uint32_t i = mid + 1; i < data.choices_lambda; i++) lambdas[i] = 1.f - (1.f - lambdas[i - 1]) / 2.0f;
236 
237  if (data.lambda_scheme == MINIMAX_CENTRAL_ZEROONE || data.lambda_scheme == ABS_CENTRAL_ZEROONE)
238  {
239  lambdas[0] = 0.0;
240  lambdas[data.choices_lambda - 1] = 1.0;
241  }
242 }

256 uint32_t corrupt_action(warm_cb& data, uint32_t action, int ec_type)
257 {
258  float cor_prob = 0.;
259  uint32_t cor_type = UAR;
260  uint32_t cor_action;
261 
262  if (ec_type == WARM_START)
263  {
264  cor_prob = data.cor_prob_ws;
265  cor_type = data.cor_type_ws;

269  if (randf < cor_prob)
270  {
271  if (cor_type == UAR)

291 float compute_weight_multiplier(warm_cb& data, size_t i, int ec_type)
292 {
293  float weight_multiplier;
294  float ws_train_size = (float)data.ws_train_size;
295  float inter_train_size = (float)data.inter_period;
296  float total_train_size = ws_train_size + inter_train_size;
297  float total_weight = (1 - data.lambdas[i]) * ws_train_size + data.lambdas[i] * inter_train_size;
298 
299  if (ec_type == WARM_START)
300  weight_multiplier = (1 - data.lambdas[i]) * total_train_size / (total_weight + FLT_MIN);
301  else
302  weight_multiplier = data.lambdas[i] * total_train_size / (total_weight + FLT_MIN);
303 
304  return weight_multiplier;
305 }

348 void learn_sup_adf(warm_cb& data, example& ec, int ec_type)
349 {
350  copy_example_to_adf(data, ec);
351  // generate cost-sensitive label (for cost-sensitive learner's temporary use)
352  auto& csls = data.csls;
353  auto& cbls = data.cbls;
354  for (uint32_t a = 0; a < data.num_actions; ++a)
355  {
356  csls[a].costs[0].class_index = a + 1;
357  if (use_cs)
358  csls[a].costs[0].x = loss_cs(data, ec.l.cs.costs, a + 1);
359  else
360  csls[a].costs[0].x = loss(data, ec.l.multi.label, a + 1);
361  }
362  for (size_t a = 0; a < data.num_actions; ++a)
363  {
364  cbls[a] = data.ecs[a]->l.cb;
365  data.ecs[a]->l.cs = csls[a];
366  }
367 
368  std::vector<float> old_weights;
369  for (size_t a = 0; a < data.num_actions; ++a) old_weights.push_back(data.ecs[a]->weight);
370 
371  for (uint32_t i = 0; i < data.choices_lambda; i++)
372  {
373  float weight_multiplier = compute_weight_multiplier(data, i, ec_type);
374  for (size_t a = 0; a < data.num_actions; ++a) data.ecs[a]->weight = old_weights[a] * weight_multiplier;
375  multi_learner* cs_learner = as_multiline(data.all->cost_sensitive);
376  cs_learner->learn(data.ecs, i);
377  }
378 
379  for (size_t a = 0; a < data.num_actions; ++a) data.ecs[a]->weight = old_weights[a];
380 
381  for (size_t a = 0; a < data.num_actions; ++a) data.ecs[a]->l.cb = cbls[a];
382 }

476 void predict_or_learn_adf(warm_cb& data, multi_learner& base, example& ec)
477 {

479  if (use_cs)

481  else

496  ec.weight = 0;

499  // Interaction phase
500  else if (data.inter_iter < data.inter_period)

503  accumu_var_adf(data, base, ec);
504  data.a_s_adf.clear();
505  data.inter_iter++;
506  }
507  // Skipping the rest of the examples
508  else
509  {
510  ec.weight = 0;
511  ec.pred.multiclass = 1;
512  }
513 
514  // Restore the original labels
515  if (use_cs)
516  ec.l.cs = data.cs_label;
517  else
518  ec.l.multi = data.mc_label;
