16  void load_string(std::string const& source)
17  {
18  std::stringstream ss(source);
19  std::string item;
20  while (std::getline(ss, item, ','))
21  {
22  std::stringstream inner_ss(item);
23  std::string klass;
24  std::string weight;
25  std::getline(inner_ss, klass, ':');
26  std::getline(inner_ss, weight, ':');
27 
28  if (!klass.size() || !weight.size())
29  {
30  THROW("error: while parsing --classweight " << item);
31  }
32 
33  int klass_int = std::stoi(klass);
34  float weight_double = std::stof(weight);
35 
36  weights[klass_int] = weight_double;
37  }
38  }

40  float get_class_weight(uint32_t klass)
41  {
42  auto got = weights.find(klass);
43  if (got == weights.end())
44  return 1.0f;
45  else
46  return got->second;
47  }
