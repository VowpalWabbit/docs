64 // - much faster (around 50% but depends on the string to parse)
65 // - less error control, but utilised inside a very strict parser

67 inline float parseFloat(const char* p, size_t& end_idx, const char* endLine = nullptr)
68 {
69  const char* start = p;
70  bool endLine_is_null = endLine == nullptr;
71 
72  end_idx = 0;
73 
74  if (!p || !*p)
75  {
76  return 0;
77  }
78  int s = 1;
79  while ((*p == ' ') && (endLine_is_null || p < endLine)) p++;
80 
81  if (*p == '-')
82  {
83  s = -1;
84  p++;
85  }
86 
87  float acc = 0;
88  while (*p >= '0' && *p <= '9' && (endLine_is_null || p < endLine)) acc = acc * 10 + *p++ - '0';
89 
90  int num_dec = 0;
91  if (*p == '.')
92  {
93  while (*(++p) >= '0' && *p <= '9' && (endLine_is_null || p < endLine))
94  {
95  if (num_dec < 35)
96  {
97  acc = acc * 10 + (*p - '0');
98  num_dec++;
99  }
100  }
101  }
102 
103  int exp_acc = 0;
104  if ((*p == 'e' || *p == 'E') && (endLine_is_null || p < endLine))
105  {
106  p++;
107  int exp_s = 1;
108  if (*p == '-' && (endLine_is_null || p < endLine))
109  {
110  exp_s = -1;
111  p++;
112  }
113  while (*p >= '0' && *p <= '9' && (endLine_is_null || p < endLine)) exp_acc = exp_acc * 10 + *p++ - '0';
114  exp_acc *= exp_s;
115  }
116  if (*p == ' ' || *p == '\n' || *p == '\t' || p == endLine) // easy case succeeded.
117  {
118  acc *= powf(10, (float)(exp_acc - num_dec));
119  end_idx = p - start;
120  return s * acc;
121  }
122  else
123  {
124  // can't use stod because that throws an exception. Use strtod instead.
125  char* end = nullptr;
126  auto ret = strtof(start, &end);
127  if (end >= start)
128  {
129  end_idx = end - start;
130  }
131  return ret;
132  }
133 
134 }
