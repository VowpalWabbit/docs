58  void initialize_Z(parameters& weights)
59  {
60  uint32_t length = 1 << all->num_bits;
61  if (normalize) // initialize normalization part
62  {
63  for (uint32_t i = 0; i < length; i++) (&(weights.strided_index(i)))[NORM2] = 0.1f;
64  }
65  if (!random_init)
66  {
67  // simple initialization
68  for (int i = 1; i <= m; i++) (&(weights.strided_index(i)))[i] = 1.f;
69  }
70  else
71  {
72  // more complicated initialization: orthgonal basis of a random matrix
73 
74  const double PI2 = 2.f * 3.1415927f;
75 
76  for (uint32_t i = 0; i < length; i++)
77  {
78  weight& w = weights.strided_index(i);
79  float r1, r2;
80  for (int j = 1; j <= m; j++)
81  {
82  // box-muller tranform: https://en.wikipedia.org/wiki/Box%E2%80%93Muller_transform
83  // redraw until r1 should be strictly positive
84  do
85  {
86  r1 = _random_state->get_and_update_random();
87  r2 = _random_state->get_and_update_random();
88  } while (r1 == 0.f);
89 
90  (&w)[j] = std::sqrt(-2.f * log(r1)) * (float)cos(PI2 * r2);
91  }
92  }
93  }
94 
95  // Gram-Schmidt
96  for (int j = 1; j <= m; j++)
97  {
98  for (int k = 1; k <= j - 1; k++)
99  {
100  double tmp = 0;
101 
102  for (uint32_t i = 0; i < length; i++)
103  tmp += ((double)(&(weights.strided_index(i)))[j]) * (&(weights.strided_index(i)))[k];
104  for (uint32_t i = 0; i < length; i++)
105  (&(weights.strided_index(i)))[j] -= (float)tmp * (&(weights.strided_index(i)))[k];
106  }
107  double norm = 0;
108  for (uint32_t i = 0; i < length; i++)
109  norm += ((double)(&(weights.strided_index(i)))[j]) * (&(weights.strided_index(i)))[j];
110  norm = std::sqrt(norm);
111  for (uint32_t i = 0; i < length; i++) (&(weights.strided_index(i)))[j] /= (float)norm;
112  }
113  }

243  {
244  for (int j = 1; j <= m; j++)

262  void check()
263  {
264  double max_norm = 0;
265  for (int i = 1; i <= m; i++)
266  for (int j = i; j <= m; j++) max_norm = fmax(max_norm, fabs(K[i][j]));
267  // printf("|K| = %f\n", max_norm);
268  if (max_norm < 1e7)
269  return;
270 
271  // implicit -> explicit representation
272  // printf("begin conversion: t = %d, norm(K) = %f\n", t, max_norm);
273 
274  // first step: K <- AKA'
275 
276  // K <- AK
277  for (int j = 1; j <= m; j++)
278  {
279  memset(tmp, 0, sizeof(double) * (m + 1));
280 
281  for (int i = 1; i <= m; i++)
282  {
283  for (int h = 1; h <= m; h++)
284  {
285  tmp[i] += A[i][h] * K[h][j];
286  }
287  }
288 
289  for (int i = 1; i <= m; i++) K[i][j] = tmp[i];
290  }
291  // K <- KA'
292  for (int i = 1; i <= m; i++)
293  {
294  memset(tmp, 0, sizeof(double) * (m + 1));
295 
296  for (int j = 1; j <= m; j++)
297  for (int h = 1; h <= m; h++) tmp[j] += K[i][h] * A[j][h];
298 
299  for (int j = 1; j <= m; j++)
300  {
301  K[i][j] = tmp[j];
302  }
303  }
304 
305  // second step: w[0] <- w[0] + (DZ)'b, b <- 0.
306 
307  uint32_t length = 1 << all->num_bits;
308  for (uint32_t i = 0; i < length; i++)
309  {
310  weight& w = all->weights.strided_index(i);
311  for (int j = 1; j <= m; j++) w += (&w)[j] * b[j] * D[j];
312  }
313 
314  memset(b, 0, sizeof(double) * (m + 1));
315 
316  // third step: Z <- ADZ, A, D <- Identity
317 
318  // double norm = 0;
319  for (uint32_t i = 0; i < length; ++i)
320  {
321  memset(tmp, 0, sizeof(float) * (m + 1));
322  weight& w = all->weights.strided_index(i);
323  for (int j = 1; j <= m; j++)
324  {
325  for (int h = 1; h <= m; ++h) tmp[j] += A[j][h] * D[h] * (&w)[h];
326  }
327  for (int j = 1; j <= m; ++j)
328  {
329  // norm = std::max(norm, fabs(tmp[j]));
330  (&w)[j] = tmp[j];
331  }
332  }
333  // printf("|Z| = %f\n", norm);
334 
335  for (int i = 1; i <= m; i++)
336  {
337  memset(A[i], 0, sizeof(double) * (m + 1));
338  D[i] = 1;
339  A[i][i] = 1;
340  }
341  }

451 void learn(OjaNewton& ON, base_learner& base, example& ec)
452 {

454  predict(ON, base, ec);
455 
456  update_data& data = ON.data;
457  data.g = ON.all->loss->first_derivative(ON.all->sd, ec.pred.scalar, ec.l.simple.label) * ec.l.simple.weight;
458  data.g /= 2; // for half square loss
459 
460  if (ON.normalize)
461  GD::foreach_feature<update_data, update_normalization>(*ON.all, ec, data);
462 
463  ON.buffer[ON.cnt] = &ec;
464  ON.weight_buffer[ON.cnt++] = data.g / 2;
465 
466  if (ON.cnt == ON.epoch_size)
467  {
468  for (int k = 0; k < ON.epoch_size; k++, ON.t++)
469  {
470  example& ex = *(ON.buffer[k]);
471  data.sketch_cnt = ON.weight_buffer[k];
472 
473  data.norm2_x = 0;
474  memset(data.Zx, 0, sizeof(float) * (ON.m + 1));
475  GD::foreach_feature<update_data, compute_Zx_and_norm>(*ON.all, ex, data);
476  ON.compute_AZx();
477 
478  ON.update_eigenvalues();
479  ON.compute_delta();
480 
481  ON.update_K();
482 
483  GD::foreach_feature<update_data, update_Z_and_wbar>(*ON.all, ex, data);
484  }
485 
486  ON.update_A();
487  // ON.update_D();
488  }
489 
490  memset(data.Zx, 0, sizeof(float) * (ON.m + 1));
491  GD::foreach_feature<update_data, update_wbar_and_Zx>(*ON.all, ec, data);
492  ON.compute_AZx();
493 
494  ON.update_b();
495  ON.check();
496 
497  if (ON.cnt == ON.epoch_size)
498  {
499  ON.cnt = 0;
