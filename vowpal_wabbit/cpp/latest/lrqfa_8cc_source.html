32 void predict_or_learn(LRQFAstate& lrq, single_learner& base, example& ec)
33 {
34  vw& all = *lrq.all;
35 
36  memset(lrq.orig_size, 0, sizeof(lrq.orig_size));
37  for (namespace_index i : ec.indices) lrq.orig_size[i] = ec.feature_space[i].size();
38 
39  size_t which = ec.example_counter;
40  float first_prediction = 0;
41  float first_loss = 0;
42  unsigned int maxiter = (is_learn && !example_is_test(ec)) ? 2 : 1;
43  unsigned int k = lrq.k;
44  float sqrtk = (float)std::sqrt(k);
45 
46  uint32_t stride_shift = lrq.all->weights.stride_shift();
47  uint64_t weight_mask = lrq.all->weights.mask();
48  for (unsigned int iter = 0; iter < maxiter; ++iter, ++which)
49  {
50  // Add left LRQ features, holding right LRQ features fixed
51  // and vice versa
52 
53  for (std::string::const_iterator i1 = lrq.field_name.begin(); i1 != lrq.field_name.end(); ++i1)
54  {
55  for (std::string::const_iterator i2 = i1 + 1; i2 != lrq.field_name.end(); ++i2)
56  {
57  unsigned char left = (which % 2) ? *i1 : *i2;
58  unsigned char right = ((which + 1) % 2) ? *i1 : *i2;
59  unsigned int lfd_id = lrq.field_id[left];
60  unsigned int rfd_id = lrq.field_id[right];
61  for (unsigned int lfn = 0; lfn < lrq.orig_size[left]; ++lfn)
62  {
63  features& fs = ec.feature_space[left];
64  float lfx = fs.values[lfn];
65  uint64_t lindex = fs.indicies[lfn];
66  for (unsigned int n = 1; n <= k; ++n)
67  {
68  uint64_t lwindex =
69  (lindex + ((uint64_t)(rfd_id * k + n) << stride_shift)); // a feature has k weights in each field
70  float* lw = &all.weights[lwindex & weight_mask];
71  // perturb away from saddle point at (0, 0)
72  if (is_learn && !example_is_test(ec) && *lw == 0)
73  *lw = cheesyrand(lwindex) * 0.5f / sqrtk;
74 
75  for (unsigned int rfn = 0; rfn < lrq.orig_size[right]; ++rfn)
76  {
77  features& rfs = ec.feature_space[right];
78  // feature* rf = ec.atomics[right].begin + rfn;
79  // NB: ec.ft_offset added by base learner
80  float rfx = rfs.values[rfn];
81  uint64_t rindex = rfs.indicies[rfn];
82  uint64_t rwindex = (rindex + ((uint64_t)(lfd_id * k + n) << stride_shift));
83 
84  rfs.push_back(*lw * lfx * rfx, rwindex);
85  if (all.audit || all.hash_inv)
86  {
87  std::stringstream new_feature_buffer;
88  new_feature_buffer << right << '^' << rfs.space_names[rfn].get()->second << '^' << n;
89 #ifdef _WIN32
90  char* new_space = _strdup("lrqfa");
91  char* new_feature = _strdup(new_feature_buffer.str().c_str());
92 #else
93  char* new_space = strdup("lrqfa");
94  char* new_feature = strdup(new_feature_buffer.str().c_str());
95 #endif
96  rfs.space_names.push_back(audit_strings_ptr(new audit_strings(new_space, new_feature)));
97  }
98  }
99  }
100  }
101  }
102  }
103 
104  if (is_learn)
105  base.learn(ec);
106  else
107  base.predict(ec);
108 
109  // Restore example
110  if (iter == 0)
111  {
112  first_prediction = ec.pred.scalar;
113  first_loss = ec.loss;
114  }
115  else
116  {
117  ec.pred.scalar = first_prediction;
118  ec.loss = first_loss;
119  }
120 
121  for (char i : lrq.field_name)
122  {
123  namespace_index right = i;
124  features& rfs = ec.feature_space[right];
125  rfs.values.end() = rfs.values.begin() + lrq.orig_size[right];
126 
127  if (all.audit || all.hash_inv)
128  {
129  for (size_t j = lrq.orig_size[right]; j < rfs.space_names.size(); ++j) rfs.space_names[j].~audit_strings_ptr();
130 
131  rfs.space_names.end() = rfs.space_names.begin() + lrq.orig_size[right];
132  }
133  }
134  }
135 }
