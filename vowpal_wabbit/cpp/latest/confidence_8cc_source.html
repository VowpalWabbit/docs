19 void predict_or_learn_with_confidence(confidence& /* c */, single_learner& base, example& ec)
20 {
21  float threshold = 0.f;
22  float sensitivity = 0.f;
23 
24  float existing_label = ec.l.simple.label;
25  if (existing_label == FLT_MAX)
26  {
27  base.predict(ec);
28  float opposite_label = 1.f;
29  if (ec.pred.scalar > 0)
30  opposite_label = -1.f;
31  ec.l.simple.label = opposite_label;
32  }
33 
34  if (!is_confidence_after_training)
35  sensitivity = base.sensitivity(ec);
36 
37  ec.l.simple.label = existing_label;
38  if (is_learn)
39  base.learn(ec);
40  else
41  base.predict(ec);
42 
43  if (is_confidence_after_training)
44  sensitivity = base.sensitivity(ec);
45 
46  ec.confidence = fabsf(ec.pred.scalar - threshold) / sensitivity;
47 }
