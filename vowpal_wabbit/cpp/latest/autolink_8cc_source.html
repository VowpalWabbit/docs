53 void VW::autolink::prepare_example(LEARNER::single_learner& base, example& ec)
54 {
55  base.predict(ec);
56  float base_pred = ec.pred.scalar;
57 
58  // Add features of label.
59  ec.indices.push_back(autolink_namespace);
60  features& fs = ec.feature_space[autolink_namespace];
61  for (size_t i = 0; i < _poly_degree; i++)
62  {
63  if (base_pred != 0.)
64  {
65  fs.push_back(base_pred, AUTOCONSTANT + (i << _stride_shift));
66  base_pred *= ec.pred.scalar;
67  }
68  }
69  ec.total_sum_feat_sq += fs.sum_feat_sq;
70 }
